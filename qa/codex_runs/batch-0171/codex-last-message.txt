            <h2 class="text-2xl font-semibold">How to Use This Calculator</h2>
            <p>Enter your most recent systolic and diastolic pressure readings in mmHg. Click Calculate or wait for automatic updates to see the categorized result along with guidance on what it means for your cardiovascular health.</p>
            <h2 class="text-2xl font-semibold">Methodology</h2>
            <div class="optional-note">
              <p class="mb-1"><strong>Interpretation:</strong> The calculator follows the 2017 guideline thresholds from the American Heart Association. It evaluates your reading hierarchically, checking crisis ranges first before assigning other stages.</p>
              <p>Validations prevent impossible combinations (systolic must exceed diastolic, and values remain inside medically plausible bounds) so you always see a meaningful classification.</p>
            </div>
            <ul class="list-disc pl-5 space-y-2 text-gray-700">
              <li>Hypertensive Crisis is flagged when the systolic is above 180 or diastolic exceeds 120; emergency guidance is recommended.</li>
              <li>Stage 2 and Stage 1 thresholds follow the standard AHA cutoffs, covering both systolic and diastolic conditions.</li>
              <li>Elevated indicates higher-than-desirable systolic pressure with normal diastolic; Normal stays under 120/80.</li>
            </ul>
            <div class="mt-4 bg-gray-50 border border-gray-200 rounded-lg p-4 text-sm text-gray-700">
              This tool is informational only. Consult a healthcare provider for clinical advice, especially if your reading falls in the elevated, high blood pressure, or crisis ranges.
            </div>

            <details id="formulaDetails">
              <summary class="cursor-pointer font-bold text-gray-800">Formulas</summary>
              <div class="mt-3 p-3 bg-white border border-gray-200 rounded-lg text-sm">
                <p><strong>Blood pressure category rule:</strong></p>
                <div class="formula-box mt-2">
                  \[
                  \text{Category} = \begin{cases}
                  \text{Hypertensive Crisis} & \text{if } S > 180 \text{ or } D > 120 \\
                  \text{High BP (Stage 2)} & \text{if } S \ge 140 \text{ or } D \ge 90 \\
                  \text{High BP (Stage 1)} & \text{if } (130 \le S \le 139) \text{ or } (80 \le D \le 89) \\
                  \text{Elevated} & \text{if } (120 \le S \le 129) \text{ and } D < 80 \\
                  \text{Normal} & \text{if } S < 120 \text{ and } D < 80
                  \end{cases}
                  \]
                </div>
                <p class="mt-3 text-gray-600">S represents systolic mmHg and D represents diastolic mmHg. The calculator evaluates the highest-priority condition that matches your reading.</p>
              </div>

      (function () {

        const debounce = (fn, delay = 100) => {
          let timeout;
          return (...args) => {
            clearTimeout(timeout);
            timeout = setTimeout(() => fn(...args), delay);

            action: 'Talk to your doctor about medications and lifestyle changes.',

            action: 'Adopt healthy habits like reducing sodium and exercising.',

        function parseInputs() {
          return {
            systolic: Number.parseInt(els.systolic.value, 10),
            diastolic: Number.parseInt(els.diastolic.value, 10)
          };
        }

        function validate(inputs) {
          const errors = [];
          const s = inputs.systolic;
          const d = inputs.diastolic;
          if (!Number.isFinite(s)) {
            errors.push('Systolic pressure is required and must be a number.');
          } else if (s < 40 || s > 300) {
            errors.push('Please enter a systolic value between 40 and 300.');
          }
          if (!Number.isFinite(d)) {
            errors.push('Diastolic pressure is required and must be a number.');
          } else if (d < 30 || d > 200) {
            errors.push('Please enter a diastolic value between 30 and 200.');
          }
          if (Number.isFinite(s) && Number.isFinite(d) && s <= d) {
            errors.push('Systolic must be higher than diastolic.');
          }
          return { ok: errors.length === 0, errors };
        }

        function compute(inputs) {
          const s = inputs.systolic;
          const d = inputs.diastolic;
          if (!Number.isFinite(s) || !Number.isFinite(d)) return null;

          let category = null;
          if (s > 180 || d > 120) {
            category = 'Hypertensive Crisis';
          } else if (s >= 140 || d >= 90) {
            category = 'High Blood Pressure (Stage 2)';
          } else if ((s >= 130 && s <= 139) || (d >= 80 && d <= 89)) {
            category = 'High Blood Pressure (Stage 1)';
          } else if (s >= 120 && s <= 129 && d < 80) {
            category = 'Elevated';
          } else if (s < 120 && d < 80) {
            category = 'Normal';
          } else if (s >= 130 && d < 80) {
            category = 'High Blood Pressure (Stage 1)';
          } else if (s < 120 && d >= 80) {
            category = 'High Blood Pressure (Stage 1)';
          }
          if (!category) return null;
          const copy = categoryCopy[category] || categoryCopy['Normal'];
          return { category, explanation: copy.explanation, action: copy.action, colorClass: copy.colorClass };
        }

        function format(outputs, inputs) {
          const defaultFormatted = {
            category: '—',
            explanation: 'Enter both systolic and diastolic values to see your category.',
            action: 'Awaiting readings',
            colorClass: 'text-gray-500',
            systolicLabel: Number.isFinite(inputs.systolic) ? `${inputs.systolic} mmHg` : '—',
            diastolicLabel: Number.isFinite(inputs.diastolic) ? `${inputs.diastolic} mmHg` : '—'
          };
          if (!outputs) return defaultFormatted;
          return {
            category: outputs.category,
            explanation: outputs.explanation,
            action: outputs.action,
            colorClass: outputs.colorClass || 'text-gray-500',
            systolicLabel: `${inputs.systolic} mmHg`,
            diastolicLabel: `${inputs.diastolic} mmHg`
          };
        }

        function render(formatted, errors) {
          renderError(errors);
          els.resultCategory.textContent = formatted.category;
          els.resultExplanation.textContent = formatted.explanation;
          els.resultAction.textContent = formatted.action;
          els.displaySystolic.textContent = formatted.systolicLabel;
          els.displayDiastolic.textContent = formatted.diastolicLabel;
          els.resultCategory.className = 'result-category';
          if (formatted.colorClass) {
            els.resultCategory.classList.add(formatted.colorClass);
          } else {
            els.resultCategory.classList.add('text-gray-500');
          }
        }

        function renderError(errors) {
          if (!errors || errors.length === 0) {
            els.errorBox.style.display = 'none';
            els.errorBox.textContent = '';
            return;
          }
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render(format(null, inputs), validation.errors);
            return;
          }
          const outputs = compute(inputs);
          if (!outputs) {
            render(format(null, inputs), ['Unable to classify this reading.']);
            return;
          }
          render(format(outputs, inputs), []);
        }

        els.calcBtn.addEventListener('click', update);
        els.resetBtn.addEventListener('click', () => {
          els.systolic.value = `${defaults.systolic}`;
          els.diastolic.value = `${defaults.diastolic}`;
          update();
        });

        const debouncedUpdate = debounce(update, 100);
        document.querySelectorAll('#inputsCard input').forEach((input) => {
          input.addEventListener('input', debouncedUpdate);
          input.addEventListener('change', debouncedUpdate);
        });

        update();
      })();
