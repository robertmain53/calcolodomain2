        <h2>How to use</h2>
        <p>Input the measured oxygen consumption (VO2) and the arterial/venous oxygen contents (CaO2 and CvO2) using consistent units. Hit Calculate or rely on the instant recalculation to derive the cardiac output in liters per minute.</p>
        <h2>Methodology</h2>
        <p>This calculator implements the Fick principle: cardiac output equals oxygen consumption divided by the arteriovenous oxygen content difference. It assumes steady-state measurements and yields a single, steady-volume output for the provided inputs.</p>
        <ul>
          <li>Ensure VO2 is in mL/min and oxygen contents are in mL O2/L for consistency.</li>
          <li>The denominator is CaO2 − CvO2; a positive difference is required for a meaningful result.</li>
          <li>Values are rounded to two decimals for clarity.</li>
        </ul>
        <h2>Full original guide (expanded)</h2>
        <p>This tool is designed for healthcare professionals needing a quick cardiac output estimate. When the measurement inputs are precise, the result helps confirm or question hemodynamic assumptions.</p>
        <p>The calculator mirrors the classic formula:</p>
        <div class="formula-block">
          <pre>Cardiac Output = VO2 / (CaO2 − CvO2)</pre>
        </div>
        <p>For example, VO2 = 250 mL/min, CaO2 = 200 mL/L, CvO2 = 150 mL/L yields 5 L/min. Repeat the calculation with updated measurements as needed.</p>
        <h3>FAQ</h3>
        <h4>What is the Fick principle?</h4>
        <p>It equates the oxygen uptake rate to the blood flow multiplied by the oxygen content difference across the pulmonary circulation.</p>
        <h4>Why is cardiac output important?</h4>
        <p>It reflects heart pump performance and guides critical care decisions.</p>
        <h4>How can I improve accuracy?</h4>
        <p>Use calibrated devices for VO2 and consistent lab values for CaO2/CvO2.</p>
        <h4>Can this be used in pediatrics?</h4>
        <p>Yes, ensure pediatric norms for the inputs.</p>

  <script>
    (function() {
      'use strict';

      const debounce = (fn, delay = 100) => {
        let t;
        return (...args) => {
          clearTimeout(t);
          t = setTimeout(() => fn(...args), delay);
        };
      };

      const round2 = (x) => Math.round((x + Number.EPSILON) * 100) / 100;
      const fmtNumber = (val) => Number.isFinite(val) ? round2(val).toFixed(2) : '—';

      const els = {
        vo2: document.getElementById('vo2'),
        cao2: document.getElementById('cao2'),
        cvo2: document.getElementById('cvo2'),
        totalOutput: document.getElementById('totalOutput'),
        avDiff: document.getElementById('avDiff'),
        vo2Display: document.getElementById('vo2Display'),
        formulaDisplay: document.getElementById('formulaDisplay'),
        errorBox: document.getElementById('errorBox'),
        calcBtn: document.getElementById('calcBtn'),
        resetBtn: document.getElementById('resetBtn')
      };

      const defaults = {
        vo2: 250,
        cao2: 200,
        cvo2: 150
      };

      function parseInputs() {
        return {
          vo2: parseFloat(els.vo2.value),
          cao2: parseFloat(els.cao2.value),
          cvo2: parseFloat(els.cvo2.value)
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!Number.isFinite(inputs.vo2) || inputs.vo2 <= 0) errors.push('VO2 must be greater than 0.');
        if (!Number.isFinite(inputs.cao2) || inputs.cao2 <= 0) errors.push('CaO2 must be greater than 0.');
        if (!Number.isFinite(inputs.cvo2) || inputs.cvo2 <= 0) errors.push('CvO2 must be greater than 0.');
        if (Number.isFinite(inputs.cao2) && Number.isFinite(inputs.cvo2) && inputs.cao2 <= inputs.cvo2) {
          errors.push('CaO2 must exceed CvO2 to compute a positive output.');
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const difference = inputs.cao2 - inputs.cvo2;
        if (!Number.isFinite(difference) || difference <= 0) return null;
        const cardiacOutput = inputs.vo2 / difference;
        if (!Number.isFinite(cardiacOutput)) return null;
        return {
          cardiacOutput: round2(cardiacOutput),
          avDifference: round2(difference)
        };
      }

      function format(outputs, inputs) {
        return {
          total: `${fmtNumber(outputs.cardiacOutput)} L/min`,
          diff: `${fmtNumber(outputs.avDifference)} mL/L`,
          vo2: `${fmtNumber(inputs.vo2)} mL/min`,
          formula: `${fmtNumber(inputs.vo2)} / (${fmtNumber(inputs.cao2)} - ${fmtNumber(inputs.cvo2)})`
        };
      }

      function render(formatted = {}, errors = []) {
        if (errors.length === 0) {
          els.errorBox.style.display = 'none';
          els.errorBox.textContent = '';
        } else {
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
        }
        els.totalOutput.textContent = formatted.total || '—';
        els.avDiff.textContent = formatted.diff || '—';
        els.vo2Display.textContent = formatted.vo2 || '—';
        els.formulaDisplay.textContent = formatted.formula || '—';
      }

      function update() {
        const inputs = parseInputs();
        const validated = validate(inputs);
        if (!validated.ok) {
          render({}, validated.errors);
          return;
        }
        const outputs = compute(inputs);
        if (!outputs) {
          render({}, ['Unable to compute cardiac output with the provided values.']);
          return;
        }
        const formatted = format(outputs, inputs);
        render(formatted);
      }

      const debouncedUpdate = debounce(update, 100);

      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', () => {
        els.vo2.value = defaults.vo2;
        els.cao2.value = defaults.cao2;
        els.cvo2.value = defaults.cvo2;
        update();
      });

      document.querySelectorAll('#inputsCard input').forEach((el) => {
        el.addEventListener('input', debouncedUpdate);
        el.addEventListener('change', debouncedUpdate);
      });

      update();
    })();
  </script>
