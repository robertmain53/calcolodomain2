        <ul class="list-disc pl-5 text-gray-700 space-y-1">
          <li><strong>Azimuth:</strong> A horizontal angle measured clockwise from a north baseline.</li>
          <li><strong>Bearing:</strong> The angle between a reference direction (north or south) and a line.</li>
        </ul>
        <div class="ghost-divider"></div>
        <h2 class="text-2xl font-semibold">Step-by-Step Example</h2>
        <p>If you start with an azimuth of 120°, add 180° and take the remainder modulo 360° to arrive at a bearing of 300°. The tool does the same math automatically and keeps both the input and output normalized.</p>

      (function () {
        'use strict';

        const defaults = { angle: 45, conversionType: 'toBearing' };
        const els = {
          angleInput: document.getElementById('angleInput'),
          convertedDisplay: document.getElementById('convertedDisplay'),
          normalizedInput: document.getElementById('normalizedInput'),
          normalizedOutput: document.getElementById('normalizedOutput'),
          conversionLabel: document.getElementById('conversionLabel'),
          errorBox: document.getElementById('errorBox'),
          calcBtn: document.getElementById('calcBtn'),
          resetBtn: document.getElementById('resetBtn')
        };
        const conversionRadios = Array.from(document.querySelectorAll('input[name="conversionType"]'));

        const round = (value, decimals = 2) => {
          if (!Number.isFinite(value)) return 0;
          const factor = Math.pow(10, decimals);
          return Math.round((value + Number.EPSILON) * factor) / factor;
        };

        const formatDegree = (value) => `${round(value).toFixed(2)}°`;

        function parseInputs() {
          const raw = els.angleInput.value.trim();
          const angle = raw === '' ? NaN : Number(raw);
          const conversionType = conversionRadios.find((radio) => radio.checked)?.value || defaults.conversionType;
          return { angle, conversionType };
        }

        function validate(inputs) {
          const errors = [];
          if (!Number.isFinite(inputs.angle)) {
            errors.push('Please enter a valid angle in degrees.');
          }
          return { ok: errors.length === 0, errors };
        }

        function compute(inputs) {
          const normalizedInput = ((inputs.angle % 360) + 360) % 360;
          let converted;
          if (inputs.conversionType === 'toBearing') {
            converted = (normalizedInput + 180) % 360;
          } else {
            converted = (normalizedInput - 180 + 360) % 360;
          }
          const normalizedOutput = ((converted % 360) + 360) % 360;
          return { converted, normalizedInput, normalizedOutput };
        }

        function format(outputs, inputs) {
          const conversionLabel = inputs.conversionType === 'toBearing' ? 'Azimuth → Bearing' : 'Bearing → Azimuth';
          return {
            convertedDisplay: formatDegree(outputs.converted),
            normalizedInput: formatDegree(outputs.normalizedInput),
            normalizedOutput: formatDegree(outputs.normalizedOutput),
            conversionLabel
          };
        }

        function render(formatted, validation) {
          if (!validation.ok) {
            els.errorBox.textContent = validation.errors.join(' ');
            els.errorBox.style.display = 'block';
            els.convertedDisplay.textContent = '—';
            els.normalizedInput.textContent = '—';
            els.normalizedOutput.textContent = '—';
            els.conversionLabel.textContent = '—';
            return;
          }

          els.errorBox.style.display = 'none';
          els.errorBox.textContent = '';
          if (!formatted) return;

          els.convertedDisplay.textContent = formatted.convertedDisplay;
          els.normalizedInput.textContent = formatted.normalizedInput;
          els.normalizedOutput.textContent = formatted.normalizedOutput;
          els.conversionLabel.textContent = formatted.conversionLabel;
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          const outputs = validation.ok ? compute(inputs) : null;
          const formatted = outputs ? format(outputs, inputs) : null;
          render(formatted, validation);
        }

        function debounce(fn, delay = 100) {
          let timer;
          return function (...args) {
            if (timer) {
              clearTimeout(timer);
            }
            timer = setTimeout(() => fn.apply(this, args), delay);
          };
        }

        const interactiveInputs = document.querySelectorAll('#inputsCard input, #inputsCard select, #inputsCard textarea');
        const debouncedUpdate = debounce(update, 100);

        els.calcBtn.addEventListener('click', update);
        els.resetBtn.addEventListener('click', () => {
          els.angleInput.value = defaults.angle;
          conversionRadios.forEach((radio) => {
            radio.checked = radio.value === defaults.conversionType;
          });
          update();
        });

        interactiveInputs.forEach((el) => {
          el.addEventListener('input', debouncedUpdate);
          el.addEventListener('change', debouncedUpdate);
        });

        update();
      })();
    </script>
