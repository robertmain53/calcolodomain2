  <script>
    (function() {

      'use strict';

      const debounce = (fn, delay = 100) => {
        let timer;
        return (...args) => {
          clearTimeout(timer);
          timer = setTimeout(() => fn(...args), delay);
        };
      };

      const round2 = (value) => Math.round((value + Number.EPSILON) * 100) / 100;

      const formatNumber = (value, decimals = 2) => {
        if (!Number.isFinite(value)) return '—';
        return value.toLocaleString('en-US', {
          minimumFractionDigits: decimals,
          maximumFractionDigits: decimals
        });
      };

      const els = {
        area: document.getElementById('area'),
        yieldPerHectare: document.getElementById('yieldPerHectare'),
        totalYieldMain: document.getElementById('totalYieldMain'),
        areaKpi: document.getElementById('areaKpi'),
        yieldKpi: document.getElementById('yieldKpi'),
        errorBox: document.getElementById('errorBox'),
        calcBtn: document.getElementById('calcBtn'),
        resetBtn: document.getElementById('resetBtn')
      };

      const defaults = {
        area: 10,
        yieldPerHectare: 4
      };

      function setDefaults() {
        els.area.value = defaults.area;
        els.yieldPerHectare.value = defaults.yieldPerHectare;
      }

      function parseInputs() {
        return {
          area: Number.parseFloat(els.area.value),
          yieldPerHectare: Number.parseFloat(els.yieldPerHectare.value)
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!Number.isFinite(inputs.area) || inputs.area <= 0) {
          errors.push('Area must be greater than 0.');
        }
        if (!Number.isFinite(inputs.yieldPerHectare) || inputs.yieldPerHectare < 0) {
          errors.push('Yield per hectare must be 0 or greater.');
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const totalYield = inputs.area * inputs.yieldPerHectare;
        if (!Number.isFinite(totalYield)) return null;
        return { totalYield: round2(totalYield) };
      }

      function format(outputs, inputs) {
        return {
          totalYieldLabel: `${formatNumber(outputs.totalYield)} tons`,
          areaLabel: `${formatNumber(inputs.area)} ha`,
          yieldPerHectareLabel: `${formatNumber(inputs.yieldPerHectare)} tons/ha`
        };
      }

      function renderError(errors) {
        if (!errors || errors.length === 0) {
          els.errorBox.style.display = 'none';
          els.errorBox.textContent = '';
          return;
        }
        els.errorBox.style.display = 'block';
        els.errorBox.textContent = errors.join(' ');
      }

      function render(formatted) {
        els.totalYieldMain.textContent = formatted.totalYieldLabel;
        els.areaKpi.textContent = formatted.areaLabel;
        els.yieldKpi.textContent = formatted.yieldPerHectareLabel;
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          renderError(validation.errors);
          return;
        }
        renderError([]);
        const outputs = compute(inputs);
        if (!outputs) {
          renderError(['Unable to compute results for the current inputs.']);
          return;
        }
        const formatted = format(outputs, inputs);
        render(formatted);
      }

      const debouncedUpdate = debounce(update, 100);

      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', () => {
        setDefaults();
        update();
      });

      document.querySelectorAll('#inputsCard input, #inputsCard select, #inputsCard textarea').forEach((el) => {
        el.addEventListener('input', debouncedUpdate);
        el.addEventListener('change', debouncedUpdate);
      });

      setDefaults();
      update();
    })();
  </script>
