        <h2 class="text-2xl font-semibold mt-6">Full original guide (expanded)</h2>
        <div class="full-guide-block">
          <h3 class="text-xl font-semibold">Data Source</h3>
          <p>We rely on the standard HVAC conversion where 1 Watt equals 3.41214 BTU/hr (and the inverse). Energy professionals, HVAC suppliers, and electrical engineers use the same ratio when mapping thermal capacity to electrical power.</p>
        </div>
        <div class="full-guide-block">
          <h3 class="text-xl font-semibold">Step-by-Step Conversion Example</h3>
          <ol class="list-decimal list-inside space-y-2 text-gray-700">
            <li><strong>Start with the value:</strong> 3,412.14 BTU/hr.</li>
            <li><strong>Apply the formula:</strong> Watts = 3,412.14 ÷ 3.41214.</li>
            <li><strong>Calculate:</strong> Watts = 1,000.</li>
            <li><strong>Result:</strong> 3,412.14 BTU/hr = 1,000 Watts (rounded to two decimals by the calculator).</li>
          </ol>

        <div class="full-guide-block">
          <h3 class="text-xl font-semibold">Glossary</h3>
          <dl class="space-y-4">
            <div>
              <dt class="font-semibold text-gray-800">BTU/hr (British Thermal Unit per hour)</dt>
              <dd class="text-gray-600 ml-4">A unit of power commonly used in HVAC to measure heating and cooling capacity. It represents the energy transfer rate in BTU per hour.</dd>
            </div>
            <div>
              <dt class="font-semibold text-gray-800">Watt (W)</dt>
              <dd class="text-gray-600 ml-4">The SI unit of power, equal to one joule per second. It measures how rapidly energy is transferred or consumed.</dd>
            </div>
            <div>
              <dt class="font-semibold text-gray-800">Power</dt>
              <dd class="text-gray-600 ml-4">The rate at which energy is transferred or converted. Unlike energy, it tells you how quickly work is performed.</dd>
            </div>
          </dl>
        </div>
        <div class="full-guide-block">
          <h3 class="text-xl font-semibold">Frequently Asked Questions</h3>
          <div class="space-y-4">
            <div>
              <p class="font-semibold">How do you convert BTU/hr to Watts?</p>
              <p class="text-gray-600">Divide the BTU/hr value by 3.41214 or multiply by 0.293071.</p>
            </div>
            <div>
              <p class="font-semibold">How many BTU/hr is 1 Watt?</p>
              <p class="text-gray-600">One Watt equals 3.41214 BTU/hr, so every 1,000 Watts correspond to 3,412.14 BTU/hr.</p>
            </div>
            <div>
              <p class="font-semibold">Why is this conversion important?</p>
              <p class="text-gray-600">It helps compare HVAC capacity (BTU/hr) to electrical consumption (Watts), which matters for sizing equipment and calculating operating costs.</p>
            </div>
            <div>
              <p class="font-semibold">Can I use this for air conditioners?</p>
              <p class="text-gray-600">Yes. Air conditioners are rated in BTU/hr; converting to Watts clarifies the electrical power required.</p>
            </div>
            <div>
              <p class="font-semibold">What's the difference between BTU and BTU/hr?</p>
              <p class="text-gray-600">BTU is energy; BTU/hr is power (energy per time). BTU/hr expresses the rate energy is delivered, matching the concept of Watts.</p>
            </div>
          </div>

      (function() {
        const CONVERSION_FACTOR = 3.41214;
        const DEFAULT_VALUE = '3412.14';
        const DEBOUNCE_DELAY = 100;
        const formatter = new Intl.NumberFormat('en-US', {
          minimumFractionDigits: 2,
          maximumFractionDigits: 2
        });

        const els = {
          btuInput: document.getElementById('btuInput'),
          calcBtn: document.getElementById('calcBtn'),
          resetBtn: document.getElementById('resetBtn'),
          wattsResult: document.getElementById('wattsResult'),
          btuDisplay: document.getElementById('btuDisplay'),
          ratioDisplay: document.getElementById('ratioDisplay'),
          inverseDisplay: document.getElementById('inverseDisplay'),
          errorBox: document.getElementById('errorBox')
        };

        const debounce = (fn, delay = DEBOUNCE_DELAY) => {
          let timeout;
          return (...args) => {
            clearTimeout(timeout);
            timeout = setTimeout(() => fn(...args), delay);
          };
        };

        function parseInputs() {
          const sanitized = els.btuInput.value.replace(/,/g, '').trim();
          const raw = parseFloat(sanitized);
          return {
            btu: Number.isFinite(raw) ? raw : NaN
          };
        }

        function validate(inputs) {
          const errors = [];
          if (!Number.isFinite(inputs.btu)) {
            errors.push('BTU/hr must be a valid number.');
          } else if (inputs.btu < 0) {
            errors.push('BTU/hr must be 0 or greater.');
          }
          return { ok: errors.length === 0, errors };
        }

        function compute(inputs) {
          return {
            watts: inputs.btu / CONVERSION_FACTOR,
            btu: inputs.btu,
            ratio: CONVERSION_FACTOR,
            inverse: 1 / CONVERSION_FACTOR
          };
        }

        function format(outputs, inputs) {
          return {
            result: `${formatter.format(outputs.watts)} W`,
            btuValue: formatter.format(inputs.btu),
            ratioText: `${outputs.ratio.toFixed(5)} BTU/hr`,
            inverseText: `${outputs.inverse.toFixed(6)} W`
          };
        }

        function render(formatted, errors) {
          const ratioDefault = `${CONVERSION_FACTOR.toFixed(5)} BTU/hr`;
          const inverseDefault = `${(1 / CONVERSION_FACTOR).toFixed(6)} W`;
          if (errors && errors.length) {
            els.errorBox.style.display = 'block';
            els.errorBox.textContent = errors.join(' ');
            els.wattsResult.textContent = '—';
            els.btuDisplay.textContent = '—';
            els.ratioDisplay.textContent = ratioDefault;
            els.inverseDisplay.textContent = inverseDefault;
            return;
          }
          els.errorBox.style.display = 'none';
          els.errorBox.textContent = '';
          els.wattsResult.textContent = formatted.result;
          els.btuDisplay.textContent = formatted.btuValue;
          els.ratioDisplay.textContent = formatted.ratioText;
          els.inverseDisplay.textContent = formatted.inverseText;
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render(null, validation.errors);
            return;
          }
          const outputs = compute(inputs);
          const formatted = format(outputs, inputs);
          render(formatted, []);
        }

        const debouncedUpdate = debounce(update, DEBOUNCE_DELAY);
        ['input', 'change'].forEach((evt) => {
          els.btuInput.addEventListener(evt, debouncedUpdate);
        });

        els.calcBtn.addEventListener('click', update);
        els.resetBtn.addEventListener('click', () => {
          els.btuInput.value = DEFAULT_VALUE;
          update();
        });

        update();
      })();
