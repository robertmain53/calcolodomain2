    (function() {
      'use strict';

      const FACTOR = 0.0630902;
      const EXAMPLE_GPM = 50;
      const EXAMPLE_LPS = EXAMPLE_GPM * FACTOR;

      const els = {
        gpmInput: document.getElementById('gpmInput'),
        calcBtn: document.getElementById('calcBtn'),
        resetBtn: document.getElementById('resetBtn'),
        lpsValue: document.getElementById('lpsValue'),
        lpmValue: document.getElementById('lpmValue'),
        factorValue: document.getElementById('factorValue'),
        exampleValue: document.getElementById('exampleValue'),
        errorBox: document.getElementById('errorBox')
      };

      const defaults = {
        gpm: 50
      };

      const debounce = (fn, delay = 100) => {
        let timer;
        return (...args) => {
          clearTimeout(timer);
          timer = setTimeout(() => fn(...args), delay);
        };
      };

      function parseInputs() {
        return {
          gpm: parseFloat(els.gpmInput.value)
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!Number.isFinite(inputs.gpm) || inputs.gpm < 0) {
          errors.push('Enter a non-negative gallons per minute value.');
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const lps = inputs.gpm * FACTOR;
        return {
          lps,
          lpm: lps * 60
        };
      }

      function format(outputs) {
        return {
          lps: outputs.lps.toFixed(5),
          lpm: outputs.lpm.toFixed(3),
          factor: FACTOR.toFixed(7),
          example: `${EXAMPLE_LPS.toFixed(5)} LPS`
        };
      }

      function render(formatted, errors) {
        if (errors && errors.length) {
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
          return;
        }
        els.errorBox.style.display = 'none';
        els.errorBox.textContent = '';
        els.lpsValue.textContent = formatted.lps;
        els.lpmValue.textContent = formatted.lpm;
        els.factorValue.textContent = formatted.factor;
        els.exampleValue.textContent = formatted.example;
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          render(null, validation.errors);
          return;
        }
        render(format(compute(inputs)), []);
      }

      function resetForm() {
        els.gpmInput.value = defaults.gpm;
        update();
      }

      const debouncedUpdate = debounce(update, 100);
      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', resetForm);
      els.gpmInput.addEventListener('input', debouncedUpdate);
      els.gpmInput.addEventListener('change', debouncedUpdate);

      resetForm();
    })();
  </script>
