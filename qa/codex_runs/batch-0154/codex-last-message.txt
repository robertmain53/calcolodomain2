            <h2 class="text-2xl font-semibold">How to Use This Calculator</h2>
            <p>Provide the cue ball and target ball coordinates on your table layout. The calculator interprets your positions as Cartesian points and instantly reveals the angle you must strike to line up the shot. Use decimals for precise aiming, and reset anytime to return to the example setup.</p>

            <h2 class="text-2xl font-semibold mt-6">Methodology</h2>
            <div class="bg-gray-50 border border-gray-200 rounded-lg p-4">
              <p>The engine measures the vector from the cue ball to the target and applies the <code>atan2</code> function to return the angle in degrees. The sign tells you whether the cue must travel clockwise or counterclockwise from the positive X axis (your reference direction), while the distance and delta values show how far to move the cue for a precise line.</p>
            </div>
            <ul class="list-disc pl-5 mt-4 space-y-2 text-gray-700">
              <li>Angle output is in degrees and respects the full −180° to 180° range provided by <code>Math.atan2</code>.</li>
              <li>Delta values help you visualize horizontal and vertical adjustments before touching the cue.</li>
              <li>The distance metric is the straight-line separation between the cue and target balls.</li>
            </ul>

            <div class="mt-4 bg-gray-50 border border-gray-200 rounded-lg p-4 text-sm text-gray-700">
              <p>Figures assume a planar table and neglect spin, friction, or cushion interactions. Use them as aiming guidance and adjust for house-specific dynamics.</p>
            </div>

                <li><strong>Shot Angle:</strong> The angle at which you need to strike the cue ball (measured from the positive X axis).</li>

<script>
(function() {
  'use strict';

  const round2 = (value) => Math.round((value + Number.EPSILON) * 100) / 100;
  const formatNumber = (value) => {
    if (!Number.isFinite(value)) return '0.00';
    return round2(value).toFixed(2);
  };

  const els = {
    cueBallX: document.getElementById('cueBallX'),
    cueBallY: document.getElementById('cueBallY'),
    targetBallX: document.getElementById('targetBallX'),
    targetBallY: document.getElementById('targetBallY'),
    shotAngleValue: document.getElementById('shotAngleValue'),
    deltaXVal: document.getElementById('deltaXVal'),
    deltaYVal: document.getElementById('deltaYVal'),
    distanceVal: document.getElementById('distanceVal'),
    errorBox: document.getElementById('errorBox'),
    scheduleWrap: document.getElementById('scheduleWrap'),
    scheduleBody: document.getElementById('scheduleBody'),
    toggleSchedule: document.getElementById('toggleSchedule'),
    downloadCsv: document.getElementById('downloadCsv'),
    calcBtn: document.getElementById('calcBtn'),
    resetBtn: document.getElementById('resetBtn')
  };

  let lastVector = null;
  let scheduleVisible = false;

  function parseInputs() {
    return {
      cueBallX: parseFloat(els.cueBallX.value),
      cueBallY: parseFloat(els.cueBallY.value),
      targetBallX: parseFloat(els.targetBallX.value),
      targetBallY: parseFloat(els.targetBallY.value)
    };
  }

  function validate(inputs) {
    const errors = [];
    ['cueBallX', 'cueBallY', 'targetBallX', 'targetBallY'].forEach((key) => {
      if (!Number.isFinite(inputs[key])) {
        errors.push('All coordinates must be valid numbers.');
      }
    });
    if (Number.isFinite(inputs.cueBallX) && Number.isFinite(inputs.cueBallY)
        && Number.isFinite(inputs.targetBallX) && Number.isFinite(inputs.targetBallY)) {
      if (inputs.cueBallX === inputs.targetBallX && inputs.cueBallY === inputs.targetBallY) {
        errors.push('Cue and target cannot occupy the same point.');
      }
    }
    return { ok: errors.length === 0, errors };
  }

  function compute(inputs) {
    const dx = inputs.targetBallX - inputs.cueBallX;
    const dy = inputs.targetBallY - inputs.cueBallY;
    const angle = Math.atan2(dy, dx) * (180 / Math.PI);
    const distance = Math.hypot(dx, dy);
    if (!Number.isFinite(angle) || !Number.isFinite(distance)) {
      return null;
    }
    return { angle, deltaX: dx, deltaY: dy, distance };
  }

  function format(outputs) {
    return {
      angleLabel: `${formatNumber(outputs.angle)}°`,
      deltaX: formatNumber(outputs.deltaX),
      deltaY: formatNumber(outputs.deltaY),
      distance: formatNumber(outputs.distance)
    };
  }

  function render(formatted, errors) {
    if (errors && errors.length) {
      els.errorBox.style.display = 'block';
      els.errorBox.textContent = errors.join(' ');
    } else {
      els.errorBox.style.display = 'none';
      els.errorBox.textContent = '';
    }

    const safe = formatted || { angleLabel: '0.00°', deltaX: '0.00', deltaY: '0.00', distance: '0.00' };
    els.shotAngleValue.textContent = safe.angleLabel;
    els.deltaXVal.textContent = safe.deltaX;
    els.deltaYVal.textContent = safe.deltaY;
    els.distanceVal.textContent = safe.distance;

    lastVector = formatted ? {
      angle: safe.angleLabel,
      deltaX: safe.deltaX,
      deltaY: safe.deltaY,
      distance: safe.distance
    } : null;

    els.downloadCsv.disabled = !lastVector;
    if (scheduleVisible && lastVector) {
      renderSchedule();
    }
  }

  function renderSchedule() {
    els.scheduleBody.innerHTML = '';
    if (!lastVector) return;
    const row = document.createElement('tr');
    row.innerHTML = `
      <td>1</td>
      <td>Shot vector</td>
      <td>${lastVector.deltaX}</td>
      <td>${lastVector.deltaY}</td>
      <td>${lastVector.angle}</td>
      <td>${lastVector.distance}</td>
    `;
    els.scheduleBody.appendChild(row);
  }

  function update() {
    const inputs = parseInputs();
    const validation = validate(inputs);
    if (!validation.ok) {
      render(null, validation.errors);
      return;
    }
    const outputs = compute(inputs);
    if (!outputs) {
      render(null, ['Calculation failed for the current inputs. Please revise the coordinates.']);
      return;
    }
    const formatted = format(outputs);
    render(formatted, []);
  }

  els.calcBtn.addEventListener('click', update);
  els.resetBtn.addEventListener('click', () => {
    els.cueBallX.value = '0';
    els.cueBallY.value = '0';
    els.targetBallX.value = '5';
    els.targetBallY.value = '3';
    update();
  });

  els.toggleSchedule.addEventListener('click', () => {
    scheduleVisible = !scheduleVisible;
    els.scheduleWrap.style.display = scheduleVisible ? 'block' : 'none';
    els.toggleSchedule.textContent = scheduleVisible ? 'Hide Calculation Notes' : 'View Calculation Notes';
    if (scheduleVisible) {
      renderSchedule();
    }
  });

  els.downloadCsv.addEventListener('click', () => {
    if (!lastVector) return;
    let csv = 'Step,Description,Delta X,Delta Y,Angle,Distance\n';
    csv += `1,Shot vector,${lastVector.deltaX},${lastVector.deltaY},${lastVector.angle},${lastVector.distance}\n`;
    const blob = new Blob([csv], { type: 'text/csv' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'shot-vector.csv';
    a.click();
    URL.revokeObjectURL(url);
  });

  const debounce = (fn, delay = 100) => {
    let timeout;
    return (...args) => {
      clearTimeout(timeout);
      timeout = setTimeout(() => fn(...args), delay);
    };
  };

  const debouncedUpdate = debounce(update, 100);
  document.querySelectorAll('#inputsCard input').forEach((el) => {
    el.addEventListener('input', debouncedUpdate);
    el.addEventListener('change', debouncedUpdate);
  });

  update();
})();
</script>
