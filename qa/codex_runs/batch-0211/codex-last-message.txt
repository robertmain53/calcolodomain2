            <h2 class="text-2xl font-semibold mt-6">Frequently Asked Questions</h2>
            <h3 class="text-xl font-semibold mt-4">What is the formula for BTU to Joules (J)?</h3>
            <p>The formula is: <strong>Joules = BTU × 1055.056</strong>.</p>

            <h3 class="text-xl font-semibold mt-4">How many Joules are in 1 BTU?</h3>
            <p>There are <strong>1,055.056 Joules</strong> (or 1.055 kilojoules) in 1 British Thermal Unit (BTU).</p>

            <h3 class="text-xl font-semibold mt-4">What is the difference between BTU/hr and Joules/second (Watts)?</h3>
            <p>BTU/hr and Watts both measure power (the rate of energy flow). A Watt equals 1 Joule per second, so <strong>1 BTU/hr = 0.293071 Watts</strong>.</p>

            <h3 class="text-xl font-semibold mt-4">How many Joules are in a kBTU?</h3>
            <p><strong>1 kBTU</strong> equals <strong>1,055,056 Joules</strong> because 1,000 BTU × 1,055.056 J/BTU = 1,055,056 J.</p>

      (function() {
        'use strict';

        const formatNumber = (num) => {
          if (!Number.isFinite(num) || num === 0) return '';
          return parseFloat(num.toPrecision(7)).toString();
        };

        const roundTo = (value, decimals) => {
          const factor = Math.pow(10, decimals);
          return Math.round((value + Number.EPSILON) * factor) / factor;
        };

        const constants = {
          BTU_TO_J: 1055.05585,
          BTU_TO_KJ: 1.05505585,
          BTUHR_TO_W: 0.29307107,
          BTUHR_TO_KW: 0.000293071
        };

        const els = {
          energyBtu: document.getElementById('energyBtu'),
          energyJ: document.getElementById('energyJ'),
          energyKj: document.getElementById('energyKj'),
          powerBtuHr: document.getElementById('powerBtuHr'),
          powerW: document.getElementById('powerW'),
          powerKw: document.getElementById('powerKw'),
          errorBox: document.getElementById('errorBox'),
          energyPrimary: document.getElementById('energyPrimary'),
          energyHint: document.getElementById('energyHint'),
          energyBtuDisplay: document.getElementById('energyBtuDisplay'),
          energyJDisplay: document.getElementById('energyJDisplay'),
          energyKjDisplay: document.getElementById('energyKjDisplay'),
          powerPrimary: document.getElementById('powerPrimary'),
          powerHint: document.getElementById('powerHint'),
          powerBtuHrDisplay: document.getElementById('powerBtuHrDisplay'),
          powerWDisplay: document.getElementById('powerWDisplay'),
          powerKwDisplay: document.getElementById('powerKwDisplay'),
          calcBtn: document.getElementById('calcBtn'),
          resetBtn: document.getElementById('resetBtn')
        };

        const inputMap = {
          energy: {
            btu: els.energyBtu,
            j: els.energyJ,
            kj: els.energyKj
          },
          power: {
            btu_hr: els.powerBtuHr,
            w: els.powerW,
            kw: els.powerKw
          }
        };

        const unitDefinitions = {
          energy: {
            btu: { label: 'BTU', key: 'btu' },
            j: { label: 'Joules', key: 'j' },
            kj: { label: 'Kilojoules', key: 'kj' }
          },
          power: {
            btu_hr: { label: 'BTU/hr', key: 'btu_hr' },
            w: { label: 'Watts', key: 'w' },
            kw: { label: 'Kilowatts', key: 'kw' }
          }
        };

        const summaryTargets = {
          energy: { btu: 'j', j: 'btu', kj: 'btu' },
          power: { btu_hr: 'w', w: 'btu_hr', kw: 'btu_hr' }
        };

        const state = {
          energySource: 'btu',
          powerSource: 'w'
        };

        const energyInputs = [els.energyBtu, els.energyJ, els.energyKj];
        const powerInputs = [els.powerBtuHr, els.powerW, els.powerKw];

        let isSyncing = false;

        const safeParse = (value) => {
          const num = parseFloat(value);
          return Number.isFinite(num) ? num : null;
        };

        function parseInputs() {
          const energySourceEl = inputMap.energy[state.energySource];
          const powerSourceEl = inputMap.power[state.powerSource];
          return {
            energy: {
              source: state.energySource,
              value: safeParse(energySourceEl ? energySourceEl.value : '')
            },
            power: {
              source: state.powerSource,
              value: safeParse(powerSourceEl ? powerSourceEl.value : '')
            }
          };
        }

        function validate(inputs) {
          const errors = [];
          if (inputs.energy.value !== null && inputs.energy.value < 0) {
            errors.push('Energy values must be zero or greater.');
          }
          if (inputs.power.value !== null && inputs.power.value < 0) {
            errors.push('Power values must be zero or greater.');
          }
          return { ok: errors.length === 0, errors };
        }

        function compute(inputs) {
          const energyValue = inputs.energy.value;
          const powerValue = inputs.power.value;
          const results = { energy: null, power: null };

          if (energyValue !== null) {
            let btu = 0;
            if (inputs.energy.source === 'btu') btu = energyValue;
            if (inputs.energy.source === 'j') btu = energyValue / constants.BTU_TO_J;
            if (inputs.energy.source === 'kj') btu = (energyValue * 1000) / constants.BTU_TO_J;
            const joules = btu * constants.BTU_TO_J;
            const kilojoules = joules / 1000;
            results.energy = {
              btu: roundTo(btu, 7),
              j: roundTo(joules, 7),
              kj: roundTo(kilojoules, 7)
            };
          }

          if (powerValue !== null) {
            let btuHr = 0;
            if (inputs.power.source === 'btu_hr') btuHr = powerValue;
            if (inputs.power.source === 'w') btuHr = powerValue / constants.BTUHR_TO_W;
            if (inputs.power.source === 'kw') btuHr = powerValue / constants.BTUHR_TO_KW;
            const watts = btuHr * constants.BTUHR_TO_W;
            const kilowatts = btuHr * constants.BTUHR_TO_KW;
            results.power = {
              btu_hr: roundTo(btuHr, 7),
              w: roundTo(watts, 7),
              kw: roundTo(kilowatts, 7)
            };
          }

          return results;
        }

        function buildSummary(values, source, value, type) {
          if (value === null) return '—';
          const targetKey = summaryTargets[type][source];
          const defs = unitDefinitions[type];
          const sourceDef = defs[source];
          const targetDef = defs[targetKey];
          const targetValue = formatNumber(values[targetKey]);
          if (!targetValue) return '—';
          const sourceValue = formatNumber(value) || '0';
          return `${sourceValue} ${sourceDef.label} ≈ ${targetValue} ${targetDef.label}`;
        }

        function format(outputs, inputs) {
          const formatted = { energy: null, power: null };
          if (outputs.energy) {
            formatted.energy = {
              summary: buildSummary(outputs.energy, inputs.energy.source, inputs.energy.value, 'energy'),
              btu: formatNumber(outputs.energy.btu),
              j: formatNumber(outputs.energy.j),
              kj: formatNumber(outputs.energy.kj)
            };
          }
          if (outputs.power) {
            formatted.power = {
              summary: buildSummary(outputs.power, inputs.power.source, inputs.power.value, 'power'),
              btu_hr: formatNumber(outputs.power.btu_hr),
              w: formatNumber(outputs.power.w),
              kw: formatNumber(outputs.power.kw)
            };
          }
          return formatted;
        }

        function render(formatted, errors) {
          if (errors && errors.length) {
            els.errorBox.textContent = errors.join(' ');
            els.errorBox.style.display = 'block';
          } else {
            els.errorBox.style.display = 'none';
            els.errorBox.textContent = '';
          }

          const energy = formatted.energy;
          els.energyPrimary.textContent = energy ? energy.summary : '—';
          els.energyHint.textContent = energy ? 'Energy equivalents shown below.' : 'Enter a value to see conversions.';
          els.energyBtuDisplay.textContent = energy && energy.btu ? energy.btu : '—';
          els.energyJDisplay.textContent = energy && energy.j ? energy.j : '—';
          els.energyKjDisplay.textContent = energy && energy.kj ? energy.kj : '—';

          const power = formatted.power;
          els.powerPrimary.textContent = power ? power.summary : '—';
          els.powerHint.textContent = power ? 'Power equivalents shown below.' : 'Enter a power value to see equivalents.';
          els.powerBtuHrDisplay.textContent = power && power.btu_hr ? power.btu_hr : '—';
          els.powerWDisplay.textContent = power && power.w ? power.w : '—';
          els.powerKwDisplay.textContent = power && power.kw ? power.kw : '—';

          syncInputs(formatted);
        }

        function syncInputs(formatted) {
          if (isSyncing) return;
          isSyncing = true;
          if (formatted.energy) {
            energyInputs.forEach((input) => {
              if (input.dataset.unit === state.energySource) return;
              const valueKey = input.dataset.unit;
              input.value = formatted.energy[valueKey] || '';
            });
          }
          if (formatted.power) {
            powerInputs.forEach((input) => {
              if (input.dataset.unit === state.powerSource) return;
              const valueKey = input.dataset.unit;
              input.value = formatted.power[valueKey] || '';
            });
          }
          isSyncing = false;
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render({ energy: null, power: null }, validation.errors);
            return;
          }
          const results = compute(inputs);
          const formatted = format(results, inputs);
          render(formatted, []);
        }

        const debouncedUpdate = ((fn, delay = 100) => {
          let timer;
          return (...args) => {
            clearTimeout(timer);
            timer = setTimeout(() => fn(...args), delay);
          };
        })(update);

        energyInputs.forEach((input) => {
          const unit = input.dataset.unit;
          input.addEventListener('input', () => {
            if (isSyncing) return;
            state.energySource = unit;
            debouncedUpdate();
          });
          input.addEventListener('change', () => {
            state.energySource = unit;
            debouncedUpdate();
          });
        });

        powerInputs.forEach((input) => {
          const unit = input.dataset.unit;
          input.addEventListener('input', () => {
            if (isSyncing) return;
            state.powerSource = unit;
            debouncedUpdate();
          });
          input.addEventListener('change', () => {
            state.powerSource = unit;
            debouncedUpdate();
          });
        });

        els.calcBtn.addEventListener('click', update);
        els.resetBtn.addEventListener('click', () => {
          energyInputs.forEach(input => input.value = '');
          powerInputs.forEach(input => input.value = '');
          state.energySource = 'btu';
          state.powerSource = 'w';
          update();
        });

        update();
      })();
