        <div>
          <h2 class="text-2xl font-semibold">How to Use This Calculator</h2>
          <p>Enter your fixed monthly (or annual normalized) costs, the price you charge per unit, and the variable cost that scales with each sale. The results refresh as soon as you click "Calculate" to show how many units and how much revenue are required before profit begins.</p>
          <ul class="list-disc list-inside mt-3 text-gray-700 space-y-1">
            <li>Fixed costs are expenses that stay flat regardless of production volume (rent, leases, insurance, salaries).</li>
            <li>Sales price should reflect the average transaction, including any discounts handeled regularly.</li>
            <li>Variable costs include direct materials, hourly labor, or commissions tied directly to each unit sold.</li>
          </ul>
        </div>
        <div class="bg-gray-50 border border-gray-200 rounded-lg p-4">
          <h3 class="text-xl font-semibold mb-2">Methodology</h3>
          <p>The calculator uses standard contribution-margin logic: subtract the variable cost per unit from the sales price to find the contribution margin, then divide total fixed costs by that margin. The result is rounded up to the next whole unit, reflecting how many complete sales are needed to cover every fixed dollar.</p>
        </div>
        <div class="bg-gray-50 border border-gray-200 rounded-lg p-4 text-sm text-gray-700">
          Results are estimates for planning and may differ from lender, investor, or adviser projections. Always check key inputs and consider rounding preferences when reviewing budgets.
        </div>
        <div class="bg-white border border-gray-200 rounded-lg p-4">
          <h3 class="text-xl font-semibold mb-3">Why Use This Calculator?</h3>
          <ul class="list-disc list-inside space-y-2 text-gray-700 text-sm">
            <li>Instantly calculate the sales volume required to break even.</li>
            <li>Test pricing and cost scenarios to make proactive adjustments.</li>
            <li>Understand the core drivers behind profitability.</li>
            <li>Use the contribution margin insight to track risk and margin of safety.</li>
          </ul>
        </div>

      (function() {
        'use strict';

        const round2 = (value) => Math.round((value + Number.EPSILON) * 100) / 100;
        const fmtCurrency = (value) => {
          const safe = Number.isFinite(value) ? value : 0;
          return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(safe);
        };
        const fmtNumber = (value) => {
          const safe = Number.isFinite(value) ? value : 0;
          return new Intl.NumberFormat('en-US').format(safe);
        };

        const defaults = {
          fixedCosts: '2000',
          salesPrice: '150',
          variableCost: '100'
        };

        const els = {
          fixedCosts: document.getElementById('fixedCosts'),
          salesPrice: document.getElementById('salesPrice'),
          variableCost: document.getElementById('variableCost'),
          errorFixedCosts: document.getElementById('error-fixed-costs'),
          errorSalesPrice: document.getElementById('error-sales-price'),
          errorVariableCost: document.getElementById('error-variable-cost'),
          errorBox: document.getElementById('errorBox'),
          resultUnits: document.getElementById('resultUnits'),
          resultRevenue: document.getElementById('resultRevenue'),
          contributionMargin: document.getElementById('contributionMargin'),
          contributionRatio: document.getElementById('contributionRatio'),
          calcBtn: document.getElementById('calcBtn'),
          resetBtn: document.getElementById('resetBtn'),
          tooltipButtons: document.querySelectorAll('[data-tooltip-target]'),
          fixedCostsTooltip: document.getElementById('tooltip-fixed-costs'),
          salesPriceTooltip: document.getElementById('tooltip-sales-price'),
          variableCostTooltip: document.getElementById('tooltip-variable-cost'),
          chartCanvas: document.getElementById('cvpChart')
        };

        const chartCtx = els.chartCanvas ? els.chartCanvas.getContext('2d') : null;

        const debounce = (fn, delay = 100) => {
          let timer;
          return (...args) => {
            clearTimeout(timer);
            timer = setTimeout(() => fn(...args), delay);
          };
        };

        const tooltipButtons = Array.from(els.tooltipButtons);
        tooltipButtons.forEach((button) => {
          const targetId = button.getAttribute('data-tooltip-target');
          const tooltip = document.getElementById(targetId);
          if (!tooltip) return;

          button.addEventListener('click', () => {
            const isActive = tooltip.classList.toggle('active');
            button.setAttribute('aria-expanded', isActive ? 'true' : 'false');
          });
          button.addEventListener('blur', () => {
            tooltip.classList.remove('active');
            button.setAttribute('aria-expanded', 'false');
          });

          tooltip.addEventListener('mouseenter', () => {
            tooltip.classList.add('active');
            button.setAttribute('aria-expanded', 'true');
          });
          tooltip.addEventListener('mouseleave', () => {
            tooltip.classList.remove('active');
            button.setAttribute('aria-expanded', 'false');
          });
        });

        function parseInputs() {
          return {
            fixedCosts: parseFloat(els.fixedCosts.value),
            salesPrice: parseFloat(els.salesPrice.value),
            variableCost: parseFloat(els.variableCost.value)
          };
        }

        function setFieldError(inputEl, errorEl, message) {
          if (!inputEl || !errorEl) return;
          if (message) {
            inputEl.classList.add('border-red-500');
            inputEl.classList.remove('border-gray-300');
            errorEl.textContent = message;
            errorEl.classList.remove('hidden');
          } else {
            inputEl.classList.remove('border-red-500');
            inputEl.classList.add('border-gray-300');
            errorEl.textContent = '';
            errorEl.classList.add('hidden');
          }
        }

        function clearFieldErrors() {
          setFieldError(els.fixedCosts, els.errorFixedCosts, '');
          setFieldError(els.salesPrice, els.errorSalesPrice, '');
          setFieldError(els.variableCost, els.errorVariableCost, '');
        }

        function validate(inputs) {
          const errors = [];
          clearFieldErrors();

          if (!Number.isFinite(inputs.fixedCosts) || inputs.fixedCosts < 0) {
            errors.push('Fixed costs must be 0 or greater.');
            setFieldError(els.fixedCosts, els.errorFixedCosts, 'Fixed costs must be 0 or greater.');
          }
          if (!Number.isFinite(inputs.salesPrice) || inputs.salesPrice <= 0) {
            errors.push('Sales price must be greater than 0.');
            setFieldError(els.salesPrice, els.errorSalesPrice, 'Sales price must be greater than 0.');
          }
          if (!Number.isFinite(inputs.variableCost) || inputs.variableCost < 0) {
            errors.push('Variable cost must be 0 or greater.');
            setFieldError(els.variableCost, els.errorVariableCost, 'Variable cost must be 0 or greater.');
          }
          if (Number.isFinite(inputs.salesPrice) && Number.isFinite(inputs.variableCost) && inputs.salesPrice <= inputs.variableCost) {
            errors.push('Sales Price per Unit must be greater than Variable Cost per Unit to be profitable.');
          }
          return { ok: errors.length === 0, errors };
        }

        function compute(inputs) {
          const contributionMargin = inputs.salesPrice - inputs.variableCost;
          if (!Number.isFinite(contributionMargin) || contributionMargin <= 0) return null;
          const exactUnits = inputs.fixedCosts / contributionMargin;
          const units = Math.max(0, Math.ceil(Number.isFinite(exactUnits) ? exactUnits : 0));
          const revenue = units * inputs.salesPrice;
          const ratio = inputs.salesPrice ? contributionMargin / inputs.salesPrice : 0;
          return { contributionMargin, units, revenue, ratio };
        }

        function format(results) {
          if (!results) {
            return {
              units: '—',
              revenue: '$0.00',
              contributionMargin: '$0.00',
              contributionRatio: '—'
            };
          }
          return {
            units: fmtNumber(results.units),
            revenue: fmtCurrency(results.revenue),
            contributionMargin: fmtCurrency(results.contributionMargin),
            contributionRatio: `${round2(results.ratio * 100).toFixed(1)}%`
          };
        }

        function render(formatted) {
          els.resultUnits.textContent = formatted.units;
          els.resultRevenue.textContent = formatted.revenue;
          els.contributionMargin.textContent = formatted.contributionMargin;
          els.contributionRatio.textContent = formatted.contributionRatio;
        }

        function renderError(messages) {
          if (!messages || messages.length === 0) {
            els.errorBox.style.display = 'none';
            els.errorBox.textContent = '';
            return;
          }
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = messages.join(' ');
        }

        function clearChart() {
          if (!chartCtx) return;
          const canvas = els.chartCanvas;
          chartCtx.clearRect(0, 0, canvas.width, canvas.height);
          chartCtx.fillStyle = '#e2e8f0';
          chartCtx.font = '13px ui-sans-serif';
          chartCtx.fillText('Enter inputs to render the break-even chart.', 20, 30);
        }

        function renderChart(results, inputs) {
          if (!chartCtx || !results) return;
          const canvas = els.chartCanvas;
          const width = canvas.width;
          const height = canvas.height;
          chartCtx.clearRect(0, 0, width, height);
          const padding = 40;
          const maxUnits = Math.max(results.units * 2, 10);
          const maxRevenue = Math.max(inputs.fixedCosts + maxUnits * inputs.variableCost, maxUnits * inputs.salesPrice);
          const usableWidth = width - padding * 2;
          const usableHeight = height - padding * 2;
          const scaleX = usableWidth / maxUnits;
          const scaleY = maxRevenue > 0 ? usableHeight / maxRevenue : 1;

          chartCtx.strokeStyle = '#cbd5f5';
          chartCtx.lineWidth = 1;
          chartCtx.beginPath();
          chartCtx.moveTo(padding, padding);
          chartCtx.lineTo(padding, height - padding);
          chartCtx.lineTo(width - padding, height - padding);
          chartCtx.stroke();

          const plotLine = (color, yAt) => {
            chartCtx.strokeStyle = color;
            chartCtx.lineWidth = 2;
            chartCtx.beginPath();
            chartCtx.moveTo(padding, height - padding - yAt(0) * scaleY);
            chartCtx.lineTo(padding + maxUnits * scaleX, height - padding - yAt(maxUnits) * scaleY);
            chartCtx.stroke();
          };

          plotLine('#2563eb', (units) => units * inputs.salesPrice);
          plotLine('#f97316', (units) => inputs.fixedCosts + units * inputs.variableCost);

          const bepX = padding + results.units * scaleX;
          chartCtx.setLineDash([6, 4]);
          chartCtx.strokeStyle = '#14b8a6';
          chartCtx.beginPath();
          chartCtx.moveTo(bepX, padding);
          chartCtx.lineTo(bepX, height - padding);
          chartCtx.stroke();
          chartCtx.setLineDash([]);

          chartCtx.fillStyle = '#0f172a';
          chartCtx.font = '12px ui-sans-serif';
          chartCtx.fillText(`${fmtNumber(results.units)} units`, bepX + 6, padding + 16);
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            renderError(validation.errors);
            render(format(null));
            clearChart();
            return;
          }

          const results = compute(inputs);
          if (!results) {
            const message = 'Sales Price per Unit must be greater than Variable Cost per Unit to be profitable.';
            renderError([message]);
            render(format(null));
            clearChart();
            return;
          }

          renderError([]);
          render(format(results));
          renderChart(results, inputs);
        }

        els.calcBtn.addEventListener('click', update);
        els.resetBtn.addEventListener('click', () => {
          els.fixedCosts.value = defaults.fixedCosts;
          els.salesPrice.value = defaults.salesPrice;
          els.variableCost.value = defaults.variableCost;
          update();
        });

        const debouncedUpdate = debounce(update, 100);
        document.querySelectorAll('#inputsCard input, #inputsCard select, #inputsCard textarea').forEach((input) => {
          input.addEventListener('input', debouncedUpdate);
          input.addEventListener('change', debouncedUpdate);
        });

        clearChart();
        update();
      })();
