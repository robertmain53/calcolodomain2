        <h2>How to use</h2>
        <p>
          Enter the numeric value and choose which magnetic units you are converting from and to. The calculator updates automatically with a slight debounce and also responds to the Calculate button when you prefer direct control.
        </p>
        <h2>Methodology</h2>
        <p>
          The tool normalizes every quantity against a base (tesla for flux density, ampere/meter for field strength, weber for flux) and applies deterministic rounding to avoid reporting inconsistent decimals. Precision settings merely affect the formatted output.
        </p>
        <ul>
          <li>Conversions remain within the same quantity family to avoid cross-type mistakes (e.g., B ↔ H is not auto-mapped).</li>
          <li>Displayed precision updates the result summary while keeping the underlying math exact.</li>
          <li>Error handling prevents NaN/Infinity from ever reaching the UI.</li>
        </ul>
      </section>

    (function() {
      'use strict';
      const debounce = (fn, delay = 100) => {
        let timer;
        return (...args) => {
          clearTimeout(timer);
          timer = setTimeout(() => fn(...args), delay);
        };
      };
      const units = {
        tesla: { label: 'tesla (T)', factor: 1 },
        gauss: { label: 'gauss (G)', factor: 1e-4 },
        oersted: { label: 'oersted (Oe)', factor: 79.5774715 },
        amperePerMeter: { label: 'ampere per meter (A/m)', factor: 1 },
        weber: { label: 'weber (Wb)', factor: 1 },
        maxwell: { label: 'maxwell (Mx)', factor: 1e-8 }
      };
      const DEFAULTS = {
        value: 1000,
        from: 'gauss',
        to: 'tesla',
        precision: 3
      };
      const els = {
        valueInput: document.getElementById('valueInput'),
        unitFrom: document.getElementById('unitFrom'),
        unitTo: document.getElementById('unitTo'),
        precision: document.getElementById('precision'),
        calcBtn: document.getElementById('calcBtn'),
        resetBtn: document.getElementById('resetBtn'),
        convertedValue: document.getElementById('convertedValue'),
        resultSummary: document.getElementById('resultSummary'),
        precisionLabel: document.getElementById('precisionLabel'),
        errorBox: document.getElementById('errorBox')
      };
      function round(val, decimals) {
        const factor = Math.pow(10, decimals);
        return Math.round((val + Number.EPSILON) * factor) / factor;
      }
      function fmtNumber(val, precision) {
        if (!Number.isFinite(val)) return '—';
        return round(val, precision).toFixed(precision);
      }
      function parseInputs() {
        return {
          value: parseFloat(els.valueInput.value),
          from: els.unitFrom.value,
          to: els.unitTo.value,
          precision: parseInt(els.precision.value, 10)
        };
      }
      function validate(inputs) {
        const errors = [];
        if (!Number.isFinite(inputs.value) || inputs.value < 0) {
          errors.push('Please enter a non-negative number.');
        }
        if (!units[inputs.from] || !units[inputs.to]) {
          errors.push('Please choose valid units.');
        }
        if (!Number.isFinite(inputs.precision) || inputs.precision < 0 || inputs.precision > 6) {
          errors.push('Precision must be between 0 and 6.');
        }
        return { ok: errors.length === 0, errors };
      }
      function compute(inputs) {
        const from = units[inputs.from];
        const to = units[inputs.to];
        if (!from || !to) return null;
        const normalized = inputs.value * from.factor;
        const converted = normalized / to.factor;
        return { converted };
      }
      function format(output, inputs) {
        if (!output) return null;
        return {
          convertedLabel: `${fmtNumber(output.converted, inputs.precision)} ${units[inputs.to].label}`,
          summary: `${fmtNumber(inputs.value, inputs.precision)} ${units[inputs.from].label} = ${fmtNumber(output.converted, inputs.precision)} ${units[inputs.to].label}`,
          precisionLabel: `${inputs.precision} decimal${inputs.precision === 1 ? '' : 's'}`
        };
      }
      function render(formatted, errors) {
        if (errors && errors.length) {
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
          els.convertedValue.textContent = '—';
          els.resultSummary.textContent = 'Error prevents conversion.';
          return;
        }
        els.errorBox.style.display = 'none';
        els.errorBox.textContent = '';
        if (!formatted) return;
        els.convertedValue.textContent = formatted.convertedLabel;
        els.resultSummary.textContent = formatted.summary;
        els.precisionLabel.textContent = formatted.precisionLabel;
      }
      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          render(null, validation.errors);
          return;
        }
        const result = compute(inputs);
        if (!result) {
          render(null, ['Unable to compute conversion.']);
          return;
        }
        const formatted = format(result, inputs);
        render(formatted, []);
      }
      const debouncedUpdate = debounce(update, 100);
      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', () => {
        els.valueInput.value = DEFAULTS.value;
        els.unitFrom.value = DEFAULTS.from;
        els.unitTo.value = DEFAULTS.to;
        els.precision.value = DEFAULTS.precision;
        update();
      });
      document.querySelectorAll('#inputsCard input, #inputsCard select').forEach((el) => {
        el.addEventListener('input', debouncedUpdate);
        el.addEventListener('change', debouncedUpdate);
      });
      update();
    })();
  </script>
