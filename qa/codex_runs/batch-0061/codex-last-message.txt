        <h2>How to use</h2>
        <p>
          Enter the superficial velocity, particle diameter, viscosity, and voidage. Click Calculate or rely on the debounced inputs to recompute the pressure drop instantly.
        </p>
        <h2>Methodology</h2>
        <p>
          The calculator applies the Ergun equation to quantify the combined viscous and inertial resistance in packed beds assuming steady, incompressible flow with water density fixed at 1000 kg/m³.
        </p>
        <h3>Full original guide (expanded)</h3>
        <p>
          This version preserves the classic Ergun derivation and tracks each term so the pressure drop contributions remain transparent for chemical engineering scenarios.
        </p>
        <h3>Illustrative example</h3>
        <p>
          Using v = 0.1 m/s, d = 0.01 m, μ = 0.001 Pa·s, and ε = 0.4 yields a pressure drop near 200 Pa. Smaller particles or higher velocities increase the inertia term dramatically.
        </p>
        <h3>FAQ</h3>
        <ul>
          <li><strong>What dominates the pressure drop?</strong> Low Reynolds numbers make the viscous term dominant, while higher velocities favor the inertial term.</li>
          <li><strong>Why does voidage matter?</strong> Voidage appears cubed in the denominator, so small changes have a large effect on the loss.</li>
        </ul>

          <div style="margin-top:12px; padding:12px; background:#fff; border:1px solid var(--line); border-radius:8px;">
            <p><strong>Ergun equation:</strong></p>
            \[\Delta P = \frac{150(1-\varepsilon)}{\varepsilon^3}\frac{\mu v}{d^2} + \frac{1.75(1-\varepsilon)}{\varepsilon^3}\frac{\rho v^2}{d}\]
            <ul style="padding-left:18px; margin-top:8px;">
              <li>$\Delta P$: Pressure drop (Pa)</li>
              <li>$v$: Superficial velocity (m/s)</li>
              <li>$d$: Particle diameter (m)</li>
              <li>$\mu$: Fluid viscosity (Pa·s)</li>
              <li>$\varepsilon$: Bed voidage (–)</li>
              <li>$\rho$: Fluid density (kg/m³), fixed at 1000 for this calculator</li>

  <script>
    (function() {

      'use strict';

      const debounce = (fn, delay = 100) => {
        let timer;
        return (...args) => {
          clearTimeout(timer);
          timer = setTimeout(() => fn(...args), delay);
        };
      };

      const round2 = (value) => Math.round((value + Number.EPSILON) * 100) / 100;
      const fmtPressure = (val) => {
        const safe = Number.isFinite(val) ? val : 0;
        return `${round2(safe).toFixed(2)} Pa`;
      };

      const els = {
        fluidVelocity: document.getElementById('fluidVelocity'),
        particleDiameter: document.getElementById('particleDiameter'),
        fluidViscosity: document.getElementById('fluidViscosity'),
        bedVoidage: document.getElementById('bedVoidage'),
        pressureDropValue: document.getElementById('pressureDropValue'),
        viscousTermValue: document.getElementById('viscousTermValue'),
        inertialTermValue: document.getElementById('inertialTermValue'),
        errorBox: document.getElementById('errorBox'),
        calcBtn: document.getElementById('calcBtn'),
        resetBtn: document.getElementById('resetBtn')
      };

      const defaultValues = {
        fluidVelocity: '0.1',
        particleDiameter: '0.01',
        fluidViscosity: '0.001',
        bedVoidage: '0.4'
      };

      function parseInputs() {
        return {
          velocity: parseFloat(els.fluidVelocity.value),
          diameter: parseFloat(els.particleDiameter.value),
          viscosity: parseFloat(els.fluidViscosity.value),
          voidage: parseFloat(els.bedVoidage.value)
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!Number.isFinite(inputs.velocity) || inputs.velocity <= 0) {
          errors.push('Velocity must be greater than 0.');
        }
        if (!Number.isFinite(inputs.diameter) || inputs.diameter <= 0) {
          errors.push('Particle diameter must be greater than 0.');
        }
        if (!Number.isFinite(inputs.viscosity) || inputs.viscosity <= 0) {
          errors.push('Viscosity must be greater than 0.');
        }
        if (!Number.isFinite(inputs.voidage) || inputs.voidage <= 0 || inputs.voidage >= 1) {
          errors.push('Voidage must be greater than 0 and less than 1.');
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const { velocity: v, diameter: d, viscosity: mu, voidage: eps } = inputs;
        const rho = 1000;
        const denom = Math.pow(eps, 3);
        if (!Number.isFinite(denom) || denom === 0) return null;
        const factor = (1 - eps) / denom;
        const viscousTerm = 150 * factor * (mu * v) / (d * d);
        const inertialTerm = 1.75 * factor * (rho * v * v) / d;
        const deltaP = viscousTerm + inertialTerm;
        if (!Number.isFinite(deltaP)) return null;
        return {
          pressureDrop: deltaP,
          viscousTerm,
          inertialTerm
        };
      }

      function format(outputs) {
        return {
          total: fmtPressure(outputs.pressureDrop),
          viscous: fmtPressure(outputs.viscousTerm),
          inertial: fmtPressure(outputs.inertialTerm)
        };
      }

      function render(formatted, errors) {
        if (errors && errors.length) {
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
          return;
        }
        els.errorBox.style.display = 'none';
        els.errorBox.textContent = '';
        els.pressureDropValue.textContent = formatted.total;
        els.viscousTermValue.textContent = formatted.viscous;
        els.inertialTermValue.textContent = formatted.inertial;
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          render(null, validation.errors);
          return;
        }

        const outputs = compute(inputs);
        if (!outputs) {
          render(null, ['Calculation produced an invalid result.']);
          return;
        }

        const formatted = format(outputs);
        render(formatted, []);
      }

      function reset() {
        Object.entries(defaultValues).forEach(([key, value]) => {
          const el = els[key];
          if (el) el.value = value;
        });
        update();
      }

      const debouncedUpdate = debounce(update, 100);

      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', reset);
      document.querySelectorAll('#inputsCard input, #inputsCard select, #inputsCard textarea')
        .forEach((input) => {
          input.addEventListener('input', debouncedUpdate);
          input.addEventListener('change', debouncedUpdate);
        });

      update();
    })();
  </script>
