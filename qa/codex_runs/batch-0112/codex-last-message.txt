        <h2 class="text-2xl font-semibold">How to Use This Calculator</h2>
        <p>This tool supports the three common quadratic representations. Switch the tab to focus on your equation form, fill in the coefficients, and click Calculate to see the axis of symmetry and a transparent, numbered walkthrough.</p>
        <div class="bg-gray-50 border border-gray-200 rounded-lg p-4 space-y-3">
          <h3 class="text-lg font-semibold">Methodology</h3>
          <p>The calculator uses the standard axis formulas for each form and interprets extra formatting (fractions or decimals) safely. Results are deterministic: the same inputs always produce the same rounded axis, and the steps mirror the math shown to you.</p>
        </div>

        <h3 class="text-xl font-semibold">1. Standard Form: <span class="font-normal">y = ax² + bx + c</span></h3>
        <p>The axis of symmetry is the vertical line that passes through the vertex. Compute it with <span class="font-semibold">x = -b / (2a)</span>. The value of <code>a</code> must not be zero or the equation is not a parabola.</p>
        <div class="formula-box">
          x = \frac{-b}{2a}
        </div>
        <p><strong>Example:</strong> For y = 2x² + 8x - 5, plug in a=2, b=8:</p>
        <p><strong>Result:</strong> x = -8 / (2×2) = -8 / 4 = -2. The axis is x = -2.</p>

        <h3 class="text-xl font-semibold">2. Vertex Form: <span class="font-normal">y = a(x - h)² + k</span></h3>
        <p>The vertex explicit in this form gives the axis directly: <span class="font-semibold">x = h</span>. The calculator reads your <code>h</code> value and displays this line.</p>
        <p><strong>Example:</strong> For y = 3(x - 4)² + 1, the axis of symmetry is x = 4 (if the equation were y = 3(x + 4)² + 1, interpret h = -4).</p>

        <h3 class="text-xl font-semibold">3. Intercept Form: <span class="font-normal">y = a(x - p)(x - q)</span></h3>
        <p>The axis runs halfway between the x-intercepts p and q. Use the midpoint formula <span class="font-semibold">x = (p + q) / 2</span> to find it.</p>
        <div class="formula-box">
          x = \frac{p + q}{2}
        </div>
        <p><strong>Example:</strong> For y = -2(x - 1)(x - 7), the intercepts are p=1 and q=7, so x = (1 + 7) / 2 = 4.</p>

        <h2 class="text-2xl font-semibold">Frequently Asked Questions (FAQ)</h2>
        <div class="space-y-4">
          <div>
            <h3 class="font-semibold">What is the axis of symmetry?</h3>
            <div class="faq-content">
              <p>The axis of symmetry is a vertical line that divides a parabola into two perfect mirror images. It passes through the vertex, the highest or lowest point of the curve.</p>
            </div>
          </div>
          <div>
            <h3 class="font-semibold">How is the axis of symmetry related to the vertex of a parabola?</h3>
            <div class="faq-content">
              <p>The axis of symmetry is the x-coordinate of the vertex. If the vertex is at (h, k), the axis is x = h, and this calculator identifies that value.</p>
            </div>
          </div>
          <div>
            <h3 class="font-semibold">What is the axis of symmetry for a horizontal parabola?</h3>
            <div class="faq-content">
              <p>Horizontal parabolas (functions of y) are not supported here. Their axis is horizontal: y = -b / (2a). This calculator focuses on vertical parabolas (functions of x).</p>
            </div>
          </div>
          <div>
            <h3 class="font-semibold">What does the 'a' value do?</h3>
            <div class="faq-content">
              <p>The 'a' coefficient determines the opening direction and steepness. If a is positive, the parabola opens upward; if negative, it opens downward. In vertex and intercept forms, a does not shift the axis, but it is essential for standard form calculations.</p>
            </div>
          </div>
        </div>

  const debounce = (fn, delay = 100) => {
    let timer;
    return (...args) => {
      clearTimeout(timer);
      timer = setTimeout(() => fn(...args), delay);
    };
  };

  const roundForDisplay = (value) => {
    if (!Number.isFinite(value)) return '0';
    return parseFloat(value.toFixed(10)).toString().replace(/\.0+$/, (match) => '');
  };

  const parseValue = (raw) => {
    const value = String(raw || '').trim();
    if (value === '') return NaN;
    if (value.includes('/')) {
      const parts = value.split('/').map((v) => v.trim());
      if (parts.length === 2) {
        const num = parseFloat(parts[0]);
        const den = parseFloat(parts[1]);
        if (!isNaN(num) && !isNaN(den) && den !== 0) {
          return num / den;
        }
      }
    }
    return parseFloat(value);
  };

  const els = {
    stdA: document.getElementById('std-a'),

      vtxH: parseValue(els.vtxH.value),

      intA: parseValue(els.intA.value),
      intP: parseValue(els.intP.value),
      intQ: parseValue(els.intQ.value)
    };

  function validate(inputs) {
    const errors = [];
    if (inputs.form === 'standard-form') {
      if (!Number.isFinite(inputs.stdA) || Math.abs(inputs.stdA) < 1e-12) {
        errors.push("'a' must be a non-zero number in standard form.");
      }
      if (!Number.isFinite(inputs.stdB)) {
        errors.push("'b' must be a valid number in standard form.");
      }
      if (!Number.isFinite(inputs.stdC)) {
        errors.push("'c' must be a valid number in standard form.");
      }
    } else if (inputs.form === 'vertex-form') {
      if (!Number.isFinite(inputs.vtxH)) {
        errors.push("'h' must be a valid number in vertex form.");
      }
    } else if (inputs.form === 'intercept-form') {
      if (!Number.isFinite(inputs.intP) || !Number.isFinite(inputs.intQ)) {
        errors.push("'p' and 'q' must be valid numbers in intercept form.");
      }
    }

  }

  function compute(inputs) {
    let axis = 0;
    let stepsHtml = '';
    let stepsCount = 0;
    if (inputs.form === 'standard-form') {
      axis = -inputs.stdB / (2 * inputs.stdA);
      const formattedA = roundForDisplay(inputs.stdA);
      const formattedB = roundForDisplay(inputs.stdB);
      const formattedNegB = roundForDisplay(-inputs.stdB);
      const formattedDen = roundForDisplay(2 * inputs.stdA);

        <p>1. Equation: y = ax² + bx + c</p>
        <p>2. Substituting a = ${formattedA}, b = ${formattedB}</p>
        <p>3. Apply x = -b / (2a)</p>
        <p>4. x = ${formattedNegB} / ${formattedDen}</p>
        <p>5. Result: x = ${roundForDisplay(axis)}</p>
      `;

        <p>3. Axis of symmetry: x = h</p>
        <p>4. Result: x = ${formattedH}</p>
      `;
      stepsCount = 4;

      axis = (inputs.intP + inputs.intQ) / 2;

      const formattedSum = roundForDisplay(inputs.intP + inputs.intQ);

        <p>2. Identified intercepts: p = ${formattedP}, q = ${formattedQ}</p>
        <p>3. Axis uses midpoint: x = (p + q) / 2</p>
        <p>4. x = (${formattedP} + ${formattedQ}) / 2 = ${formattedSum} / 2</p>
        <p>5. Result: x = ${roundForDisplay(axis)}</p>
      `;
      stepsCount = 5;

  }

      equation: `x = ${roundForDisplay(outputs.axis)}`,
      steps: outputs.stepsHtml,
      stepsCount: outputs.stepsCount,

    }
  }

    const validation = validate(inputs);
    if (!validation.ok) {
      render({}, validation.errors);
      return;
    }
    const outputs = compute(inputs);
    if (!Number.isFinite(outputs.axis)) {
      render({}, ['Calculation failed. Please check the inputs.']);
      return;
    }
    const formatted = format(outputs);
    render(formatted, []);
  }

  });
