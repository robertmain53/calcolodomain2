              <h2>Full original guide (expanded)</h2>
              <section class="space-y-4 mt-4">
                <div>
                  <h3>Data Source and Methodology</h3>
                  <p>All calculations are based on the Beer-Lambert Law as detailed in <em>Principles of Instrumental Analysis</em> by Skoog et al. (ISBN: 978-1-305-57721-3). The same formulas and data guide every multiplication step.</p>
                </div>
                <div>
                  <h3>The Formula Explained</h3>
                  <div class="formula-box">
                    <p>\( A = \epsilon \cdot c \cdot l \)</p>
                    <p>Where:</p>
                    <ul>
                      <li>\( A \) is the dimensionless absorbance</li>
                      <li>\( \epsilon \) is molar absorptivity in L/mol·cm</li>
                      <li>\( c \) is concentration in mol/L</li>
                      <li>\( l \) is path length in cm</li>
                    </ul>
                  </div>
                </div>
                <div>
                  <h3>Glossary of Variables</h3>
                  <ul>
                    <li><strong>Concentration:</strong> The amount of solute per liter of solution (mol/L).</li>
                    <li><strong>Path Length:</strong> The cuvette thickness, usually 1 cm, through which light passes.</li>
                    <li><strong>Molar Absorptivity:</strong> A constant that measures how strongly a molecule absorbs light at a specific wavelength, expressed in L/mol·cm.</li>
                  </ul>
                </div>
                <div>
                  <h3>How It Works: A Step-by-Step Example</h3>
                  <p>Consider a solution with concentration 0.5 mol/L, path length 1 cm, and molar absorptivity 200 L/mol·cm. Multiply the three values:</p>
                  <p>\( A = 200 \times 0.5 \times 1 = 100 \)</p>
                  <p>The solution therefore has an absorbance of 100 under those conditions.</p>
                </div>
                <div>
                  <h3>Frequently Asked Questions (FAQ)</h3>
                  <div class="faq-item">
                    <h4 class="text-base font-semibold">What is the Beer-Lambert Law?</h4>
                    <p>The law relates light attenuation to concentration, molar absorptivity, and path length with a linear relationship for dilute solutions.</p>
                  </div>
                  <div class="faq-item">
                    <h4 class="text-base font-semibold">Why is molar absorptivity important?</h4>
                    <p>It defines how strongly a species absorbs light at a given wavelength and must be known or measured before calculating absorbance.</p>
                  </div>
                  <div class="faq-item">
                    <h4 class="text-base font-semibold">How do I measure path length?</h4>
                    <p>Use the cuvette width, typically 1 cm in spectrophotometers. Enter the exact value if you have a non-standard cuvette.</p>
                  </div>
                  <div class="faq-item">
                    <h4 class="text-base font-semibold">Can the Beer-Lambert Law be applied to all solutions?</h4>
                    <p>It applies best to dilute, non-scattering solutions where absorbance varies linearly with concentration.</p>
                  </div>
                  <div class="faq-item">
                    <h4 class="text-base font-semibold">What units should I use for concentration?</h4>
                    <p>Always use mol/L to stay consistent with molar absorptivity and keep the formula valid.</p>
                  </div>
                </div>

    <script>
      (function() {
        'use strict';

        const debounce = (fn, delay = 100) => {
          let timer;
          return (...args) => {
            clearTimeout(timer);
            timer = setTimeout(() => fn(...args), delay);
          };
        };

        const round2 = (value) => Math.round((value + Number.EPSILON) * 100) / 100;

        const els = {
          concentration: document.getElementById('concentration'),
          pathLength: document.getElementById('pathLength'),
          molarAbsorptivity: document.getElementById('molarAbsorptivity'),
          absorbanceValue: document.getElementById('absorbanceValue'),
          displayConcentration: document.getElementById('displayConcentration'),
          displayPathLength: document.getElementById('displayPathLength'),
          displayMolarAbsorptivity: document.getElementById('displayMolarAbsorptivity'),
          errorBox: document.getElementById('errorBox'),
          calcBtn: document.getElementById('calcBtn'),
          resetBtn: document.getElementById('resetBtn')
        };

        function parseInputs() {
          return {
            concentration: parseFloat(els.concentration.value),
            pathLength: parseFloat(els.pathLength.value),
            molarAbsorptivity: parseFloat(els.molarAbsorptivity.value)
          };
        }

        function validate(inputs) {
          const errors = [];
          if (!Number.isFinite(inputs.concentration) || inputs.concentration <= 0) {
            errors.push('Concentration must be greater than 0.');
          }
          if (!Number.isFinite(inputs.pathLength) || inputs.pathLength <= 0) {
            errors.push('Path length must be greater than 0.');
          }
          if (!Number.isFinite(inputs.molarAbsorptivity) || inputs.molarAbsorptivity <= 0) {
            errors.push('Molar absorptivity must be greater than 0.');
          }
          return { ok: errors.length === 0, errors };
        }

        function compute(inputs) {
          const { concentration, pathLength, molarAbsorptivity } = inputs;
          if (![concentration, pathLength, molarAbsorptivity].every((val) => Number.isFinite(val) && val > 0)) {
            return null;
          }
          return { absorbance: round2(concentration * pathLength * molarAbsorptivity) };
        }

        function format(outputs, inputs) {
          const safeNumber = (val) => Number.isFinite(val) ? val : null;
          return {
            absorbance: outputs && Number.isFinite(outputs.absorbance) ? outputs.absorbance.toFixed(2) : '—',
            concentration: safeNumber(inputs.concentration) ? round2(inputs.concentration).toFixed(2) : '—',
            pathLength: safeNumber(inputs.pathLength) ? round2(inputs.pathLength).toFixed(2) : '—',
            molarAbsorptivity: safeNumber(inputs.molarAbsorptivity) ? round2(inputs.molarAbsorptivity).toFixed(2) : '—'
          };
        }

        function render(formatted, errors) {
          els.absorbanceValue.textContent = formatted.absorbance;
          els.displayConcentration.textContent = formatted.concentration;
          els.displayPathLength.textContent = formatted.pathLength;
          els.displayMolarAbsorptivity.textContent = formatted.molarAbsorptivity;

          if (!errors || errors.length === 0) {
            els.errorBox.style.display = 'none';
            els.errorBox.textContent = '';
            return;
          }
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render(format(null, inputs), validation.errors);
            return;
          }

          const outputs = compute(inputs);
          if (!outputs) {
            render(format(null, inputs), ['Calculation failed. Please check your entries.']);
            return;
          }

          render(format(outputs, inputs), []);
        }

        const debouncedUpdate = debounce(update, 100);
        document.querySelectorAll('#inputsCard input').forEach((input) => {
          input.addEventListener('input', debouncedUpdate);
          input.addEventListener('change', debouncedUpdate);
        });

        els.calcBtn.addEventListener('click', update);
        els.resetBtn.addEventListener('click', () => {
          els.concentration.value = els.concentration.defaultValue;
          els.pathLength.value = els.pathLength.defaultValue;
          els.molarAbsorptivity.value = els.molarAbsorptivity.defaultValue;
          update();
        });

        update();
      })();
    </script>
