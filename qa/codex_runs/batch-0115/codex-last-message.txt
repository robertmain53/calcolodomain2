      <div>
        <h2 class="text-2xl font-semibold">How to Use This Tool</h2>
        <p>Paste or type the message you want to transform, choose whether you are encoding to a Baconian alphabet or decoding from it, then click Calculate. The interface automatically translates using the same uppercase mapping used by classic Baconian exercises, so uppercase letters spread into five-letter blocks while punctuation and numbers pass through untouched.</p>
        <p>You can use slash characters (<code>/</code>) to mimic word boundaries when decoding and the field accommodates longer passages by wrapping content inside the output card.</p>
      </div>
      <div>
        <h2 class="text-2xl font-semibold mt-6">Methodology</h2>
        <div class="bg-gray-50 border border-gray-200 rounded-lg p-4">
          <p>This calculator keeps a fixed substitution map, matching each letter to five letters composed of <code>A</code> and <code>B</code>. Letters I/J and U/V share the same codes, mirroring the 24-letter Baconian alphabet used historically. The encoder uppercases the source text, emits <code>/</code> where spaces occur, and leaves punctuation, digits, and other characters untouched. The decoder normalizes the input to <code>A</code>, <code>B</code>, and word separators, then reconstructs the text block by block, reporting an error if an invalid code appears.</p>
        </div>
        <ul class="list-disc pl-5 mt-4 space-y-2 text-gray-700">
          <li>The output card shows the raw encoded or decoded string plus quick KPIs so you can confirm the mode and volume of data processed.</li>
          <li>Encoding keeps punctuation, digits, and casing (after uppercasing letters) intact so you can see where human-readable bits remain.</li>
          <li>Decoding strips non-<code>A</code>/<code>B</code> characters and treats consecutive slashes/spaces as separators between codes.</li>
        </ul>
        <div class="mt-4 bg-gray-50 border border-gray-200 rounded-lg p-4 text-sm text-gray-700">
          Results are deterministic and intended for learning; treat them as approximations and verify with traditional methods when security matters.
        </div>
      </div>

      reverseBaconianMap['ABAAA'] = 'I/J';
      reverseBaconianMap['BAABB'] = 'U/V';

      const els = {
        textInput: document.getElementById('textInput'),
        cipherMode: document.getElementById('cipherMode'),
        textInputError: document.getElementById('textInputError'),
        resultText: document.getElementById('resultText'),
        modeLabel: document.getElementById('modeLabel'),
        charsProcessed: document.getElementById('charsProcessed'),
        codeGroups: document.getElementById('codeGroups'),
        errorBox: document.getElementById('errorBox'),
        calcBtn: document.getElementById('calcBtn'),
        resetBtn: document.getElementById('resetBtn')
      };

      const defaults = {
        text: 'HELLO WORLD',
        mode: 'encode'
      };

      const debounce = (fn, delay = 100) => {
        let timeout;
        return (...args) => {
          clearTimeout(timeout);
          timeout = setTimeout(() => fn(...args), delay);
        };
      };

      function parseInputs() {
        return {
          text: els.textInput.value || '',
          mode: els.cipherMode.value
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!inputs.text.trim()) {
          errors.push('Please enter some text to encode or decode.');
        }
        if (!['encode', 'decode'].includes(inputs.mode)) {
          errors.push('Mode selection is invalid.');
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const { text, mode } = inputs;
        try {
          if (mode === 'encode') {
            const normalized = text.toUpperCase();
            const tokens = [];
            for (const char of normalized) {
              if (char === ' ') {
                tokens.push('/');
                continue;
              }
              if (baconianMap[char]) {
                tokens.push(baconianMap[char]);
                continue;
              }
              if (char.match(/[A-Z]/)) {
                throw new Error(`Character '${char}' not in Baconian alphabet map.`);
              }
              tokens.push(char);
            }
            return {
              result: tokens.join(' '),
              processed: normalized.length,
              groups: tokens.length
            };
          }

          if (mode === 'decode') {
            const cleaned = text.toUpperCase().replace(/[^AB\s\/]/g, '');
            const chunks = cleaned.split(/[\s\/]+/);
            const decoded = [];
            for (const chunk of chunks) {
              if (!chunk) continue;
              if (reverseBaconianMap[chunk]) {
                decoded.push(reverseBaconianMap[chunk]);
                continue;
              }
              if (chunk.length < 5 && !/[AB]/.test(chunk)) {
                decoded.push(chunk);
                continue;
              }
              throw new Error(`Invalid Baconian code: '${chunk}'`);
            }
            return {
              result: decoded.join(''),
              processed: text.length,
              groups: chunks.filter(Boolean).length
            };
          }

          return null;
        } catch (err) {
          return { error: err.message };
        }
      }

      function format(outputs, inputs) {
        const emptyValue = '—';
        if (!outputs || outputs.error) {
          return {
            result: emptyValue,
            modeLabel: inputs.mode === 'decode' ? 'Decoding' : 'Encoding',
            processed: emptyValue,
            groups: emptyValue
          };
        }
        return {
          result: outputs.result || emptyValue,
          modeLabel: inputs.mode === 'encode' ? 'Encoding' : 'Decoding',
          processed: outputs.processed ?? 0,
          groups: outputs.groups ?? 0
        };
      }

      function render(formatted, errors) {
        if (errors && errors.length) {
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
        } else {
          els.errorBox.style.display = 'none';
          els.errorBox.textContent = '';
        }
        els.textInputError.textContent = '';
        els.resultText.textContent = formatted.result;
        els.modeLabel.textContent = formatted.modeLabel;
        els.charsProcessed.textContent = typeof formatted.processed === 'number' ? formatted.processed.toString() : formatted.processed;
        els.codeGroups.textContent = typeof formatted.groups === 'number' ? formatted.groups.toString() : formatted.groups;
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          render(format(null, inputs), validation.errors);
          return;
        }
        const outputs = compute(inputs);
        if (!outputs || outputs.error) {
          render(format({ error: outputs ? outputs.error : 'Calculation failed.' }, inputs), [outputs ? outputs.error : 'Calculation failed.']);
          return;
        }
        render(format(outputs, inputs));
      }

      els.calcBtn.addEventListener('click', update);
      const debouncedUpdate = debounce(update, 100);
      ['input', 'change'].forEach((evt) => {
        els.textInput.addEventListener(evt, debouncedUpdate);
        els.cipherMode.addEventListener(evt, debouncedUpdate);
      });

      els.resetBtn.addEventListener('click', () => {
        els.textInput.value = defaults.text;
        els.cipherMode.value = defaults.mode;
        update();
      });

      update();
    })();
  </script>
