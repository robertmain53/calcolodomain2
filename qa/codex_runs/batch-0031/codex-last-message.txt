        <h2>How to use</h2>
        <p>Enter exactly one of the circle measurement fields (radius, diameter, circumference, or area). Leave the other fields empty so the calculator can derive the radius and update the remaining values automatically.</p>
        <p>Set the circle equation inputs (center h, center k, and radius) to display the analytic form—the standard equation uses $(x-h)^2+(y-k)^2=r^2$ while the general form rearranges it into $x^2+y^2+Dx+Ey+F=0$.</p>
        <h2>Methodology</h2>
        <p>The engine first resolves the supplied measurement into the canonical radius using the appropriate formula, then recalculates diameter, circumference, and area with deterministic rounding so the UI never shows NaN or Infinity.</p>
        <ul>
          <li>Diameter inputs derive the radius by halving the value, circumference inputs divide by $2\pi$, and area inputs use $\sqrt{A/\pi}$.</li>
          <li>Output values (radius, diameter, circumference, area) are rounded consistently to six decimal places to keep comparisons stable between browsers.</li>
          <li>The equation builder reuses the explicitly provided equation radius and center coordinates to render both the standard and general forms.</li>
        </ul>

      }

      function validate(inputs) {
        const provided = measurementKeys.filter((key) => Number.isFinite(inputs[key]) && inputs[key] > 0);
        const errors = [];
        if (provided.length === 0) {
          errors.push('Provide exactly one filled measurement field (radius, diameter, circumference, or area).');
        }
        if (provided.length > 1) {
          errors.push('Only one measurement may be entered at a time.');
        }
        if (!Number.isFinite(inputs.radiusEq) || inputs.radiusEq <= 0) {
          errors.push('The circle equation radius must be greater than 0.');
        }
        return { ok: errors.length === 0, errors, measurementKey: provided[0] };
      }

      function compute(inputs, measurementKey) {
        if (!measurementKey) return null;
        const value = inputs[measurementKey];
        let radius = null;
        const steps = [];
        let stepNumber = 1;
        if (measurementKey === 'radius') {
          radius = value;
          steps.push(`<p>${stepNumber++}. Radius input: $r = ${cleanNumber(radius)}$</p>`);
        } else if (measurementKey === 'diameter') {
          radius = value / 2;
          steps.push(`<p>${stepNumber++}. Radius from diameter: $r = d/2 = ${cleanNumber(value)}/2 = ${cleanNumber(radius)}$</p>`);
        } else if (measurementKey === 'circumference') {
          radius = value / (2 * Math.PI);
          steps.push(`<p>${stepNumber++}. Radius from circumference: $r = C/(2\\pi) = ${cleanNumber(value)}/(2\\pi) \\approx ${cleanNumber(radius)}$</p>`);
        } else if (measurementKey === 'area') {
          radius = Math.sqrt(Math.max(0, value / Math.PI));
          steps.push(`<p>${stepNumber++}. Radius from area: $r = \\sqrt{A/\\pi} = \\sqrt{${cleanNumber(value)}/\\pi} \\approx ${cleanNumber(radius)}$</p>`);
        }
        if (!Number.isFinite(radius) || radius <= 0) return null;
        const diameter = 2 * radius;
        const circumference = 2 * Math.PI * radius;
        const area = Math.PI * radius * radius;
        if (measurementKey !== 'diameter') {
          steps.push(`<p>${stepNumber++}. Diameter: $d = 2r = 2 \\times ${cleanNumber(radius)} = ${cleanNumber(diameter)}$</p>`);
        }
        if (measurementKey !== 'circumference') {
          steps.push(`<p>${stepNumber++}. Circumference: $C = 2\\pi r = 2\\pi \\times ${cleanNumber(radius)} \\approx ${cleanNumber(circumference)}$</p>`);
        }
        if (measurementKey !== 'area') {
          steps.push(`<p>${stepNumber++}. Area: $A = \\pi r^2 = \\pi \\times ${cleanNumber(radius)}^2 \\approx ${cleanNumber(area)}$</p>`);
        }
        const eqRadius = inputs.radiusEq;
        const centerH = inputs.centerH;
        const centerK = inputs.centerK;
        const radiusSq = eqRadius * eqRadius;
        const formatCenter = (variable, coord) => {
          if (Math.abs(coord) < 1e-11) return variable;
          const sign = coord >= 0 ? '-' : '+';
          return `(${variable} ${sign} ${cleanNumber(Math.abs(coord))})`;
        };
        const standardEquation = `$${formatCenter('x', centerH)}^2 + ${formatCenter('y', centerK)}^2 = ${cleanNumber(radiusSq)}$$`;
        const generalEquation = ['x^2 + y^2', term(-2 * centerH, 'x'), term(-2 * centerK, 'y'), constantTerm(centerH * centerH + centerK * centerK - radiusSq)]
          .filter(Boolean)
          .join(' ')
          .replace(/\s+/g, ' ')
          + ' = 0';
        return {
          radius,
          diameter,
          circumference,
          area,
          steps: steps.join(''),
          equation: {
            standard: standardEquation,
            general: generalEquation
          }
        };
      }

      function format(outputs) {
        return {
          radius: cleanNumber(outputs.radius),
          diameter: cleanNumber(outputs.diameter),
          circumference: cleanNumber(outputs.circumference),
          area: cleanNumber(outputs.area),
          standardEquation: outputs.equation.standard,
          generalEquation: outputs.equation.general,
          steps: outputs.steps
        };
      }

      function renderError(errors) {
        if (!errors || errors.length === 0) {
          els.errorBox.style.display = 'none';
          els.errorBox.textContent = '';
          return;
        }
        els.errorBox.style.display = 'block';
        els.errorBox.textContent = errors.join(' ');
      }

      function render(formatted) {
        els.totalMonthly.textContent = formatted.radius;
        els.displayDiameter.textContent = formatted.diameter;
        els.displayCircumference.textContent = formatted.circumference;
        els.displayArea.textContent = formatted.area;
        els.resultEquationStd.innerHTML = formatted.standardEquation;
        els.resultEquationGen.textContent = formatted.generalEquation;
        els.stepByStep.innerHTML = formatted.steps || 'Derived values will appear here once you provide a measurement.';
        if (window.MathJax && window.MathJax.typesetPromise) {
          window.MathJax.typesetPromise([els.stepByStep, els.resultEquationStd]).catch(() => {});
        }
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          renderError(validation.errors);
          return;
        }
        const computed = compute(inputs, validation.measurementKey);
        if (!computed) {
          renderError(['Unable to derive circle properties with the supplied measurement.']);
          return;
        }
        renderError([]);
        const formatted = format(computed);
        render(formatted);
      }

      const debouncedUpdate = debounce(update, 100);

      document.querySelectorAll('#inputsCard input').forEach((input) => {
        input.addEventListener('input', debouncedUpdate);
        input.addEventListener('change', debouncedUpdate);
      });

      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', () => {
        els.radius.value = defaultValues.radius;
        els.diameter.value = defaultValues.diameter;
        els.circumference.value = defaultValues.circumference;
        els.area.value = defaultValues.area;
        els.centerH.value = defaultValues.centerH;
        els.centerK.value = defaultValues.centerK;
        els.radiusEq.value = defaultValues.radiusEq;
        update();
      });

      update();
    })();

  </script>
