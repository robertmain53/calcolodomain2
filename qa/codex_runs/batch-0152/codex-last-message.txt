        <h2>How to Use This Calculator</h2>
        <p>Measure your standing height and inseam in centimeters, then enter those values above. The calculator multiplies your inseam by 0.65, the industry-standard ratio for road and hybrid bike sizing, to deliver a frame size recommendation.</p>
        <p>Click Calculate to update the results, or make a measurement change and the interface will refresh automatically after a brief debounce. Reset restores the defaults if you want to start from scratch.</p>

        <h3>Methodology</h3>
        <p>We rely on height-to-frame size ratios gathered from trusted cycling authorities and expanded by the bikecalc.com guide. The inseam determines how the frame geometry aligns with your leg extension, so a correctly sized bike keeps your joints happy and power delivery efficient.</p>
        <p>The result is a suggestion for an upright road/hybrid frame; mountain bikes often require a slightly shorter reach and more standover clearance, so treat the number as a baseline.</p>

        <h3>Glossary of Terms</h3>
        <ul>
          <li><strong>Height:</strong> Your full standing height in centimeters.</li>
          <li><strong>Inseam:</strong> The distance from crotch to the ground while standing barefoot.</li>
          <li><strong>Frame Size:</strong> The recommended road/hybrid frame size derived from your inseam.</li>
        </ul>

        <h3>Step-by-Step Example</h3>
        <p>For a rider who is 175 cm tall with an 80 cm inseam, the calculator multiplies the inseam by 0.65:</p>
        <p><code>Frame Size = 80 × 0.65 = 52 cm</code></p>

        <h3>Frequently Asked Questions</h3>
        <div>
          <h4>What if my measurements fall between sizes?</h4>
          <p>If you sit between two frame sizes, pick based on riding style. A smaller frame leans toward agility while a larger frame provides comfort, especially for long-distance riding.</p>
          <h4>Can I use this calculator for all bike types?</h4>
          <p>This calculator suits road and hybrid builds. Mountain, gravel, and aerodynamic frames may need nuanced fitting; consider professional fitting for those bikes.</p>
          <h4>How often should I re-evaluate my bike size?</h4>
          <p>Recheck your tooling if you change riding disciplines, experience discomfort, or suspect your current fit is outdated.</p>

          errorBox: document.getElementById('errorBox'),
          calcBtn: document.getElementById('calcBtn'),
          resetBtn: document.getElementById('resetBtn')
        };

        const defaults = { height: 175, inseam: 80 };

        function parseInputs() {
          return {
            height: Number.parseFloat(els.height.value),
            inseam: Number.parseFloat(els.inseam.value)
          };
        }

        function isFinitePositive(value) {
          return Number.isFinite(value) && value > 0;
        }

        function validate(inputs) {
          const errors = [];
          if (!isFinitePositive(inputs.height)) {
            errors.push('Height must be a positive number.');
          }
          if (!isFinitePositive(inputs.inseam)) {
            errors.push('Inseam must be a positive number.');
          }
          return { ok: errors.length === 0, errors };
        }

        function compute(inputs) {
          const frameSize = inputs.inseam * 0.65;
          const ratio = inputs.inseam / inputs.height;
          if (!Number.isFinite(frameSize) || !Number.isFinite(ratio)) {
            return null;
          }
          return { frameSize, ratio };
        }

        function formatNumber(value, decimals) {
          if (!Number.isFinite(value)) {
            return '—';
          }
          const factor = Math.pow(10, decimals);
          const rounded = Math.round((value + Number.EPSILON) * factor) / factor;
          const str = rounded.toFixed(decimals);
          return str.replace(/\.0+$/, '');
        }

        function format(outputs, inputs) {
          return {
            frameSize: formatNumber(outputs.frameSize, 1),
            frameDetail: `Based on ${formatNumber(inputs.inseam, 1)} cm inseam and ${formatNumber(inputs.height, 1)} cm height.`,
            ratio: formatNumber(outputs.ratio, 3),
            note: 'Rounded to one decimal place, trailing zeros removed for readability.'
          };
        }

        function render(formatted, errors) {
          if (errors && errors.length) {
            els.errorBox.innerHTML = `<strong>Please fix:</strong><ul>${errors.map(err => `<li>${err}</li>`).join('')}</ul>`;
            els.errorBox.style.display = 'block';
            els.frameSizeResult.textContent = '—';
            els.frameSizeDetail.textContent = 'Enter valid measurements to see a size.';
            els.ratioDetail.textContent = 'Height-to-inseam ratio: —';
            els.frameAdvice.textContent = 'Sizing follows the 0.65 × inseam rule from the cited source.';
            els.frameSizeNote.textContent = 'Values appear once the calculator has valid input.';
            return;
          }

          els.errorBox.textContent = '';
          els.errorBox.style.display = 'none';
          els.frameSizeResult.textContent = formatted.frameSize;
          els.frameSizeDetail.textContent = formatted.frameDetail;
          els.ratioDetail.textContent = `Height-to-inseam ratio: ${formatted.ratio}`;
          els.frameAdvice.textContent = 'Sizing follows the 0.65 × inseam rule from the cited source.';
          els.frameSizeNote.textContent = formatted.note;
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render({}, validation.errors);
            return;
          }

          const outputs = compute(inputs);
          if (!outputs) {
            render({}, ['Unable to calculate a frame size with the current inputs.']);
            return;
          }

          const formatted = format(outputs, inputs);
          render(formatted, []);
        }

        const debouncedUpdate = debounce(update, 100);
        document.querySelectorAll('#inputsCard input, #inputsCard select, #inputsCard textarea').forEach((el) => {
          el.addEventListener('input', debouncedUpdate);
          el.addEventListener('change', debouncedUpdate);
        });

        els.calcBtn.addEventListener('click', update);
        els.resetBtn.addEventListener('click', () => {
          els.height.value = defaults.height;
          els.inseam.value = defaults.inseam;
          update();
        });

        update();
      })();
    </script>
