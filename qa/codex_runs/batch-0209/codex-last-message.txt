          <article class="prose mt-4 text-sm text-gray-700">
            <h4>Conversion formulas</h4>
            <div class="formula-box">
              <p>General method: convert everything through joules (J).</p>
              <p><strong>BTU → J:</strong> J = BTU × BTU<sub>factor</sub></p>
              <p><strong>J → cal:</strong> cal = J ÷ cal<sub>factor</sub></p>
              <p><strong>BTU → cal:</strong> cal = BTU × BTU<sub>factor</sub> ÷ cal<sub>factor</sub></p>
              <p>Example: 1 BTU (IT) × 1055.05585262 ÷ 4.184 = 252.1644 cal</p>
            </div>
            <h4 class="mt-6">When to pick which unit</h4>
            <ul>
              <li><strong>BTU (IT):</strong> most common in HVAC, US engineering tables.</li>
              <li><strong>BTU (thermochemical):</strong> used in some older thermodynamics literature.</li>
              <li><strong>cal (thermochemical):</strong> standard for scientific conversion (4.184 J).</li>
              <li><strong>kcal:</strong> food labels, nutrition, “Calories” with capital C.</li>
            </ul>
            <h4 class="mt-6">FAQ</h4>
            <h5 class="font-semibold">1. Why do different sites show slightly different results?</h5>
            <p>They may be mixing BTU (IT) and BTU (thermochemical), or using 4.1868 J per calorie instead of 4.184 J. Our calculator shows you the exact factors used.</p>
            <h5 class="font-semibold">2. Are “nutrition calories” the same as kilocalories?</h5>
            <p>Yes. 1 kcal = 1 Cal (with capital C) = 1000 small calories.</p>
            <h5 class="font-semibold">3. Can I go from m³ gas → BTU → calories?</h5>
            <p>Yes, but you first need the energy content of the gas in BTU or MJ. This page handles the BTU ↔ calorie part.</p>

          btuType2: 'btu_it'
        };
        let activeMode = defaults.mode;
        let debounceTimer = null;

        function setMode(mode) {
          activeMode = mode;
          inputPanels.forEach(panel => {
            panel.classList.toggle('hidden', panel.getAttribute('data-mode-panel') !== mode);
          });
          resultPanels.forEach(panel => {
            panel.classList.toggle('hidden', panel.getAttribute('data-result-pane') !== mode);
          });
          modeButtons.forEach(button => {
            const isActive = button.getAttribute('data-mode-btn') === mode;
            button.classList.toggle('is-active', isActive);
          });
        }

        function parseNumber(element, fallback) {
          const parsed = parseFloat(element.value);
          return Number.isFinite(parsed) ? parsed : fallback;
        }

        function parseInputs() {
          return {
            mode: activeMode,
            btuValue: parseNumber(btuValueEl, defaults.btuValue),
            btuType: btuTypeEl.value,
            calType: calTypeEl.value,
            calValue: parseNumber(calValueEl, defaults.calValue),
            calType2: calType2El.value,
            btuType2: btuType2El.value
          };
        }

        function validate(inputs) {
          const errors = [];
          if (inputs.btuValue < 0) {
            errors.push('BTU values must be zero or greater.');
          }
          if (inputs.calValue < 0) {
            errors.push('Calorie values must be zero or greater.');
          }
          return { ok: errors.length === 0, errors };
        }

        function btuLabel(key) {
          if (key === 'btu_th') return 'BTU (thermochemical)';
          if (key === 'btu_mean') return 'BTU (mean)';
          return 'BTU (IT)';
        }

        function calLabel(key) {
          if (key === 'cal_it') return 'cal (IT)';
          if (key === 'cal_15') return 'cal (15°C)';
          if (key === 'kcal') return 'kcal (nutritional)';
          return 'cal (th)';
        }

        function compute(inputs) {
          const btuFactor = BTU_J[inputs.btuType];
          const calFactor = CAL_J[inputs.calType];
          const calFactor2 = CAL_J[inputs.calType2];
          const btuFactor2 = BTU_J[inputs.btuType2];
          const btuJoules = inputs.btuValue * btuFactor;
          const calories = btuJoules / calFactor;
          const calJoules = inputs.calValue * calFactor2;
          const calToBtu = calJoules / btuFactor2;
          const tableRows = [1, 2, 5, 10].map(value => {
            const joules = value * BTU_J.btu_it;
            return {
              multiplier: value,
              cal: joules / CAL_J.cal_th,
              kcal: joules / CAL_J.kcal
            };
          });
          return {
            btuToCal: { joules: btuJoules, calories, btuLabel: btuLabel(inputs.btuType), calLabel: calLabel(inputs.calType), btuFactor, calFactor },
            calToBtu: { joules: calJoules, btu: calToBtu, btuLabel: btuLabel(inputs.btuType2) },
            table: tableRows
          };
        }

        function formatNumber(value, decimals) {
          if (!Number.isFinite(value)) {
            return '0';
          }
          const formatted = value.toFixed(decimals);
          return formatted.replace(/\.?0+$/, '');
        }

        function format(outputs) {
          if (!outputs) {
            return null;
          }
          return {
            btuCalories: formatNumber(outputs.btuToCal.calories, 6),
            btuJoules: formatNumber(outputs.btuToCal.joules, 4),
            factor: `Using 1 ${outputs.btuToCal.btuLabel} = ${formatNumber(outputs.btuToCal.btuFactor, 6)} J and 1 ${outputs.btuToCal.calLabel} = ${formatNumber(outputs.btuToCal.calFactor, 6)} J`,
            calBtu: formatNumber(outputs.calToBtu.btu, 6),
            calJoules: formatNumber(outputs.calToBtu.joules, 4),
            table: outputs.table.map(row => ({ multiplier: row.multiplier, cal: formatNumber(row.cal, 3), kcal: formatNumber(row.kcal, 6) }))
          };
        }

        function render(formatted, errors) {
          if (errors && errors.length) {
            errorBox.textContent = errors.join(' ');
            errorBox.classList.add('show');
          } else {
            errorBox.textContent = '';
            errorBox.classList.remove('show');
          }
          if (!formatted) {
            return;
          }
          btuCalResultEl.textContent = formatted.btuCalories;
          btuJoulesResultEl.textContent = formatted.btuJoules;
          factorLabelEl.textContent = formatted.factor;
          calBtuResultEl.textContent = formatted.calBtu;
          calJoulesResultEl.textContent = formatted.calJoules;
          formatted.table.forEach(row => {
            const cells = tableCells[row.multiplier];
            if (cells) {
              cells.kcal.textContent = row.kcal;
              cells.cal.textContent = row.cal;
            }
          });
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render(null, validation.errors);
            return;
          }
          const outputs = compute(inputs);
          const formatted = format(outputs);
          render(formatted, []);
        }

        function scheduleUpdate() {
          if (debounceTimer) {
            clearTimeout(debounceTimer);
          }
          debounceTimer = setTimeout(update, 100);
        }

        modeButtons.forEach(button => {
          button.addEventListener('click', () => {
            const targetMode = button.getAttribute('data-mode-btn');
            if (targetMode === activeMode) { return; }
            setMode(targetMode);
            update();
          });
        });

        [btuValueEl, btuTypeEl, calTypeEl, calValueEl, calType2El, btuType2El].forEach(element => {
          element.addEventListener('input', scheduleUpdate);
          element.addEventListener('change', scheduleUpdate);
        });

        btnCalc.addEventListener('click', update);
        btnReset.addEventListener('click', () => {
          btuValueEl.value = defaults.btuValue;
          calValueEl.value = defaults.calValue;
          btuTypeEl.value = defaults.btuType;
          calTypeEl.value = defaults.calType;
          calType2El.value = defaults.calType2;
          btuType2El.value = defaults.btuType2;
          setMode(defaults.mode);
          update();
        });

        setMode(defaults.mode);
        update();
