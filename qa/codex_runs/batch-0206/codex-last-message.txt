        <details id="formulaDetails">
          <summary class="font-bold text-gray-800">Formulas</summary>
          <div class="mt-3 p-3 bg-white border border-gray-200 rounded-lg text-sm text-gray-700">
            <p><strong>Mosteller</strong></p>
            <div class="formula-box">\\[\\text{BSA}_{\\text{Mosteller}} = \\sqrt{\\frac{H_{\\text{cm}}\\times W_{\\text{kg}}}{3600}}\\]</div>
            <p class="mt-2"><strong>Du Bois &amp; Du Bois</strong></p>
            <div class="formula-box">\\[\\text{BSA}_{\\text{DuBois}} = 0.007184 \\times H_{\\text{cm}}^{0.725}\\times W_{\\text{kg}}^{0.425}\\]</div>
            <p class="mt-2"><strong>Haycock</strong></p>
            <div class="formula-box">\\[\\text{BSA}_{\\text{Haycock}} = 0.024265 \\times H_{\\text{cm}}^{0.3964}\\times W_{\\text{kg}}^{0.5378}\\]</div>
            <p class="mt-2"><strong>Gehan–George</strong></p>
            <div class="formula-box">\\[\\text{BSA}_{\\text{Gehan}} = 0.0235 \\times H_{\\text{cm}}^{0.42246}\\times W_{\\text{kg}}^{0.51456}\\]</div>
            <p class="mt-3 text-gray-600 text-xs">Variables: H = height in centimeters, W = weight in kilograms. The engine converts US inputs before applying each exponent.</p>

(function() {

  const els = {

    bsaGehan: document.getElementById('bsaGehan'),
    errorBox: document.getElementById('errorBox'),
    chartEl: document.getElementById('bsaChart')
  };

  let chart = null;

  const debounce = (fn, delay = 100) => {
    let timeout;
    return (...args) => {
      clearTimeout(timeout);
      timeout = setTimeout(() => fn(...args), delay);
    };
  };

  const round3 = (value) => Math.round((value + Number.EPSILON) * 1000) / 1000;

  const togglePanel = (button, panel) => {
    const expanded = button.getAttribute('aria-expanded') === 'true';
    button.setAttribute('aria-expanded', String(!expanded));
    panel.style.display = expanded ? 'none' : 'block';
  };

  els.heightHelpBtn.addEventListener('click', () => togglePanel(els.heightHelpBtn, els.heightHelp));
  els.weightHelpBtn.addEventListener('click', () => togglePanel(els.weightHelpBtn, els.weightHelp));

  function parseInputs() {
    return {
      units: els.unitsMetric.checked ? 'metric' : 'us',
      height: parseFloat(els.height.value),
      weight: parseFloat(els.weight.value)
    };
  }

  function validate(inputs) {
    const errors = [];
    const heightRange = inputs.units === 'metric' ? [30, 250] : [12, 98];
    const weightRange = inputs.units === 'metric' ? [1, 300] : [2, 660];
    let heightError = '';
    let weightError = '';

    if (!Number.isFinite(inputs.height)) {
      const unitLabel = inputs.units === 'metric' ? 'cm' : 'in';
      heightError = `Please enter height in ${unitLabel}.`;
      errors.push('Height is required.');
    } else if (inputs.height < heightRange[0] || inputs.height > heightRange[1]) {
      heightError = `Height must be between ${heightRange[0]} and ${heightRange[1]} ${inputs.units === 'metric' ? 'cm' : 'in'}.`;
      errors.push(heightError);
    }

    if (!Number.isFinite(inputs.weight)) {
      const unitLabel = inputs.units === 'metric' ? 'kg' : 'lb';
      weightError = `Please enter weight in ${unitLabel}.`;
      errors.push('Weight is required.');
    } else if (inputs.weight < weightRange[0] || inputs.weight > weightRange[1]) {
      weightError = `Weight must be between ${weightRange[0]} and ${weightRange[1]} ${inputs.units === 'metric' ? 'kg' : 'lb'}.`;
      errors.push(weightError);
    }

    return { ok: errors.length === 0, errors, heightError, weightError };
  }

  function renderValidation({ heightError, weightError }) {
    els.heightError.textContent = heightError || '';
    els.weightError.textContent = weightError || '';
  }

  function compute(inputs) {
    const heightCm = inputs.units === 'metric' ? inputs.height : inputs.height * 2.54;
    const weightKg = inputs.units === 'metric' ? inputs.weight : inputs.weight / 2.2046226218;
    if (![heightCm, weightKg].every(Number.isFinite)) return null;

    return {
      mosteller: Math.sqrt((heightCm * weightKg) / 3600),
      duBois: 0.007184 * Math.pow(heightCm, 0.725) * Math.pow(weightKg, 0.425),
      haycock: 0.024265 * Math.pow(heightCm, 0.3964) * Math.pow(weightKg, 0.5378),
      gehan: 0.0235 * Math.pow(heightCm, 0.42246) * Math.pow(weightKg, 0.51456)
    };
  }

  function format(results) {
    if (!results) return {
      mosteller: '—',
      duBois: '—',
      haycock: '—',
      gehan: '—'
    };
    return {
      mosteller: `${round3(results.mosteller).toFixed(3)} m²`,
      duBois: `${round3(results.duBois).toFixed(3)} m²`,
      haycock: `${round3(results.haycock).toFixed(3)} m²`,
      gehan: `${round3(results.gehan).toFixed(3)} m²`
    };
  }

  function renderError(errors) {
    if (!errors || errors.length === 0) {
      els.errorBox.style.display = 'none';
      els.errorBox.textContent = '';
      return;
    }
    els.errorBox.style.display = 'block';
    els.errorBox.textContent = errors.join(' ');
  }

    els.bsaMosteller.textContent = formatted.mosteller;
    els.bsaDuBois.textContent = formatted.duBois;
    els.bsaHaycock.textContent = formatted.haycock;
    els.bsaGehan.textContent = formatted.gehan;
  }

  function clearOutputs() {
    const placeholder = '—';
    els.bsaMosteller.textContent = placeholder;
    els.bsaDuBois.textContent = placeholder;
    els.bsaHaycock.textContent = placeholder;
    els.bsaGehan.textContent = placeholder;
  }

  function renderChart(data) {
    const labels = ['Mosteller', 'Du Bois', 'Haycock', 'Gehan–George'];
    const values = [
      round3(data.mosteller),
      round3(data.duBois),
      round3(data.haycock),
      round3(data.gehan)
    ];
    const cfg = {
      type: 'bar',
      data: {
        labels,
        datasets: [{
          label: 'BSA (m²)',
          data: values,
          backgroundColor: ['#2563eb', '#1d4ed8', '#1d4ed8', '#2563eb'],
          borderRadius: 8
        }]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: { legend: { display: false } },
        scales: {
          y: { beginAtZero: true },
          x: { grid: { display: false } }
        }
      }
    };
    if (chart) {
      chart.data.labels = labels;
      chart.data.datasets[0].data = values;
      chart.update();
    } else {
      chart = new Chart(els.chartEl, cfg);

    if (chart) {
      chart.destroy();
      chart = null;

    renderError(errors);
    if (!formatted || errors.length) {
      clearOutputs();
      clearChart();
      return;
    }
    updateOutputs(formatted);
    if (numericResults) {
      renderChart(numericResults);
    }
  }

  function update() {
    const inputs = parseInputs();
    const validation = validate(inputs);
    renderValidation(validation);
    if (!validation.ok) {
      render(null, validation.errors, null);
      return;
    }
    const results = compute(inputs);
    if (!results) {
      render(null, ['Unable to calculate with the provided values.'], null);
      return;
    }
    const formatted = format(results);
    render(formatted, [], results);
