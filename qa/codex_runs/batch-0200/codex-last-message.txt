    <section id="howToUse" class="card content-card space-y-6 mt-10" aria-label="How to use and methodology">
      <h2 class="text-2xl font-semibold">How to Use This Calculator</h2>
      <p>Set the wall length and height, choose the number of skins, and supply the brick dimensions plus the mortar joint thickness for your chosen unit system. The calculator factors in wastage automatically.</p>
      <p>Click Calculate after adjusting inputs; the results card to the right refreshes with bricks, mortar, cement, and sand requirements.</p>

      <h2 class="text-2xl font-semibold mt-6">Methodology</h2>
      <div class="bg-gray-50 border border-gray-200 rounded-xl p-4 text-gray-700">
        <p>The tool converts every measurement to metric meters before computing. The effective area of a brick equals its length plus joint multiplied by its height plus joint. Bricks per square meter follow from the reciprocal of that area and then multiply by wall area, skin multiplier, and wastage percentage.</p>
        <p>Mortar volume is derived from the number of bricks (0.23 m³ per 1,000 bricks), then split into cement and sand using a 1:4 mix ratio. Imperial results are converted from the metric totals for user convenience.</p>
      </div>
      <ul class="list-disc pl-5 space-y-2 text-gray-700">
        <li>Wall type sets the skin multiplier: single skin = 1, double = 2.</li>
        <li>A higher wastage percentage gives a larger ordering cushion for cuts and breakage.</li>
        <li>Cement and sand are reported in weight units (kg or lbs) after applying standard bulk densities.</li>
      </ul>

        <summary class="cursor-pointer font-bold text-gray-800">Formulas</summary>
        <div class="mt-3 p-3 bg-white border border-gray-200 rounded-lg text-sm space-y-2">
          <p><strong>Effective brick area</strong> = (brick length + joint) × (brick height + joint)</p>
          <p><strong>Bricks per area</strong> = 1 / effective area</p>
          <p><strong>Total bricks</strong> = wall area × bricks per area × skin multiplier × (1 + wastage %)</p>
          <p><strong>Mortar, cement, sand</strong> = (total bricks / 1000) × 0.23 m³; split into a 1:4 mix ratio then multiply by densities (cement 1440 kg/m³, sand 1600 kg/m³).</p>

    (function () {
      'use strict';

      const CEMENT_DENSITY_KG_M3 = 1440;
      const SAND_DENSITY_KG_M3 = 1600;
      const MORTAR_PER_1000_BRICKS_M3 = 0.23;

      const FT_TO_M = 0.3048;

      const DEFAULTS = {
        unitSystem: 'metric',

        brickLength: 215,

        mortarJoint: 10,
        wallType: 'single',
        wastage: 8
      };

      const UNIT_DEFAULTS = {

        imperial: { brickLength: 7.625, brickHeight: 2.25, mortarJoint: 0.375 }
      };

      const UNIT_LABELS = {
        metric: { wallLength: 'm', wallHeight: 'm', brickLength: 'mm', brickHeight: 'mm', mortarJoint: 'mm' },

        brickLengthError: document.getElementById('brickLengthError'),

        mortarJointError: document.getElementById('mortarJointError'),

      const numberFormatter = new Intl.NumberFormat('en-US');
      const errorFields = ['wallLength', 'wallHeight', 'brickLength', 'brickHeight', 'mortarJoint', 'wastagePercentage'];

      function debounce(fn, wait = 100) {
        let timeout;
        return function () {
          clearTimeout(timeout);
          timeout = setTimeout(() => fn(), wait);
        };
      }

      function applyUnitLabels(system) {
        const labels = UNIT_LABELS[system] || UNIT_LABELS.metric;
        elements.wallLengthUnit.textContent = labels.wallLength;
        elements.wallHeightUnit.textContent = labels.wallHeight;

        elements.mortarJointUnit.textContent = labels.mortarJoint;
      }

      function applyUnitDefaults(system) {
        const defaults = UNIT_DEFAULTS[system];
        if (!defaults) return;
        elements.brickLength.value = defaults.brickLength;
        elements.brickHeight.value = defaults.brickHeight;
        elements.mortarJoint.value = defaults.mortarJoint;
      }

      function resetErrors() {
        errorFields.forEach((field) => {
          const target = elements[`${field}Error`];
          if (target) target.textContent = '';
        });
        elements.errorBox.textContent = '';
        elements.errorBox.classList.remove('alert-visible');
      }

      function resetResults() {
        elements.totalBricks.textContent = '—';
        elements.totalMortar.textContent = '—';
        elements.cementNeeded.textContent = '—';
        elements.sandNeeded.textContent = '—';

      function parseInputs() {
        return {
          unitSystem: elements.unitMetric.checked ? 'metric' : 'imperial',
          wallLength: Number(elements.wallLength.value),
          wallHeight: Number(elements.wallHeight.value),

          mortarJoint: Number(elements.mortarJoint.value),
          wallType: elements.wallType.value,
          wastage: Number(elements.wastagePercentage.value)
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!Number.isFinite(inputs.wallLength) || inputs.wallLength <= 0) {
          errors.push({ field: 'wallLength', message: 'Enter a length greater than zero.' });
        }
        if (!Number.isFinite(inputs.wallHeight) || inputs.wallHeight <= 0) {
          errors.push({ field: 'wallHeight', message: 'Enter a height greater than zero.' });
        }
        if (!Number.isFinite(inputs.brickLength) || inputs.brickLength <= 0) {
          errors.push({ field: 'brickLength', message: 'Enter a brick length greater than zero.' });
        }
        if (!Number.isFinite(inputs.brickHeight) || inputs.brickHeight <= 0) {
          errors.push({ field: 'brickHeight', message: 'Enter a brick height greater than zero.' });
        }
        if (!Number.isFinite(inputs.mortarJoint) || inputs.mortarJoint < 0) {
          errors.push({ field: 'mortarJoint', message: 'Enter a joint thickness of zero or more.' });
        }
        if (!Number.isFinite(inputs.wastage) || inputs.wastage < 0) {
          errors.push({ field: 'wastagePercentage', message: 'Enter a wastage percentage of zero or more.' });
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const isMetric = inputs.unitSystem === 'metric';
        const wallLengthM = isMetric ? inputs.wallLength : inputs.wallLength * FT_TO_M;
        const wallHeightM = isMetric ? inputs.wallHeight : inputs.wallHeight * FT_TO_M;
        const brickLengthM = isMetric ? inputs.brickLength * MM_TO_M : inputs.brickLength * IN_TO_M;
        const brickHeightM = isMetric ? inputs.brickHeight * MM_TO_M : inputs.brickHeight * IN_TO_M;
        const jointM = isMetric ? inputs.mortarJoint * MM_TO_M : inputs.mortarJoint * IN_TO_M;

        const wallArea = wallLengthM * wallHeightM;
        const effectiveLength = brickLengthM + jointM;
        const effectiveHeight = brickHeightM + jointM;
        const effectiveArea = effectiveLength * effectiveHeight;
        const bricksPerM2 = effectiveArea > 0 ? 1 / effectiveArea : 0;
        const skinMultiplier = inputs.wallType === 'double' ? 2 : 1;
        const wastageMultiplier = 1 + inputs.wastage / 100;
        const totalBricks = Math.max(0, Math.ceil(wallArea * bricksPerM2 * skinMultiplier * wastageMultiplier));
        const totalMortar = (totalBricks / 1000) * MORTAR_PER_1000_BRICKS_M3;
        const totalParts = 1 + MORTAR_MIX_RATIO;
        const cementVolume = totalMortar / totalParts;
        const sandVolume = cementVolume * MORTAR_MIX_RATIO;
        const cementWeightKg = cementVolume * CEMENT_DENSITY_KG_M3;
        const sandWeightKg = sandVolume * SAND_DENSITY_KG_M3;

        return {
          isMetric,
          totalBricks,
          totalMortar,
          cementWeightKg,
          sandWeightKg
        };
      }

      function format(outputs, inputs) {
        if (!outputs) return null;
        const totalMortarMetric = `${outputs.totalMortar.toFixed(2)} m³`;
        const cementMetric = `${outputs.cementWeightKg.toFixed(1)} kg`;
        const sandMetric = `${outputs.sandWeightKg.toFixed(1)} kg`;

        if (outputs.isMetric) {
          return {
            totalBricks: numberFormatter.format(outputs.totalBricks),
            totalMortar: totalMortarMetric,
            cementNeeded: cementMetric,
            sandNeeded: sandMetric
          };
        }

        const totalMortarFt3 = outputs.totalMortar * M3_TO_FT3;
        const cementLbs = outputs.cementWeightKg * KG_TO_LBS;
        const sandLbs = outputs.sandWeightKg * KG_TO_LBS;
        return {
          totalBricks: numberFormatter.format(outputs.totalBricks),
          totalMortar: `${totalMortarFt3.toFixed(2)} ft³`,
          cementNeeded: `${cementLbs.toFixed(1)} lbs`,
          sandNeeded: `${sandLbs.toFixed(1)} lbs`
        };
      }

      function render(formatted, validation) {
        resetErrors();
        if (!formatted) {
          resetResults();
          if (validation && validation.errors.length) {
            validation.errors.forEach((error, index) => {
              const target = elements[`${error.field}Error`];
              if (target) target.textContent = error.message;
              if (index === 0) {
                elements.errorBox.textContent = error.message;
                elements.errorBox.classList.add('alert-visible');
              }
            });
          }
          return;
        }
        elements.totalBricks.textContent = formatted.totalBricks;
        elements.totalMortar.textContent = formatted.totalMortar;
        elements.cementNeeded.textContent = formatted.cementNeeded;
        elements.sandNeeded.textContent = formatted.sandNeeded;
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          render(null, validation);
          return;
        }
        const outputs = compute(inputs);
        const formatted = format(outputs, inputs);
        render(formatted);
      }

      function applyDefaults() {
        elements.unitMetric.checked = DEFAULTS.unitSystem === 'metric';
        elements.unitImperial.checked = DEFAULTS.unitSystem === 'imperial';
        elements.wallLength.value = DEFAULTS.wallLength;
        elements.wallHeight.value = DEFAULTS.wallHeight;
        elements.wallType.value = DEFAULTS.wallType;
        elements.wastagePercentage.value = DEFAULTS.wastage;
        applyUnitLabels(DEFAULTS.unitSystem);
        applyUnitDefaults(DEFAULTS.unitSystem);
      }

      const debouncedUpdate = debounce(update, 100);

      document.querySelectorAll('#inputsCard input, #inputsCard select, #inputsCard textarea').forEach((el) => {
        el.addEventListener('input', debouncedUpdate);
        el.addEventListener('change', debouncedUpdate);
      });

      elements.unitMetric.addEventListener('change', () => {
        if (elements.unitMetric.checked) {
          applyUnitLabels('metric');
          applyUnitDefaults('metric');
          debouncedUpdate();
        }
      });

      elements.unitImperial.addEventListener('change', () => {
        if (elements.unitImperial.checked) {
          applyUnitLabels('imperial');
          applyUnitDefaults('imperial');
          debouncedUpdate();
        }
      });

      elements.calcBtn.addEventListener('click', update);
      elements.resetBtn.addEventListener('click', () => {
        applyDefaults();
        update();
      });

      applyDefaults();
      update();
    })();
