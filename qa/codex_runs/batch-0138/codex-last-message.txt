          <section class="card content-card space-y-6" id="howToUse" aria-label="How to use and methodology">
            <h2 class="text-2xl font-semibold">How to Use This Calculator</h2>
            <p>Enter the start point and end point angles in degrees; the calculator subtracts the start from the end, then normalizes the result so you always see a bearing between 0° and 360°.</p>

            <h2 class="text-2xl font-semibold mt-6">Methodology</h2>
            <div class="bg-gray-50 border border-gray-200 rounded-lg p-4">
              <p>The calculator applies the standard trigonometric difference between the two angles and then wraps the outcome inside the 0°–360° range. This matches how navigational bearings are typically reported, and it mirrors the behavior described in authoritative engineering references.</p>
            </div>

            <section>
              <h3 class="text-xl font-semibold mt-6">How It Works: A Step-by-Step Example</h3>
              <p>Suppose the start point is 45 degrees and the end point is 135 degrees. The bearing is calculated as 135 − 45 = 90 degrees, which already sits inside the 0°–360° range.</p>
            </section>

            <section>
              <h3 class="text-xl font-semibold mt-4">Glossary of Variables</h3>
              <ul class="mt-2 space-y-1 text-gray-700">
                <li><strong>Start Point:</strong> The initial angle in degrees from which the bearing is measured.</li>
                <li><strong>End Point:</strong> The final angle in degrees for the point you are heading toward.</li>
                <li><strong>Bearing:</strong> The resulting direction inside 0°–360° after subtracting the two angles.</li>
              </ul>
            </section>

            <section>
              <h3 class="text-xl font-semibold mt-4">Data Source and Methodology</h3>
              <p>All calculations are based on standard trigonometric formulas and principles. Consult reliable engineering sources for further details.</p>
            </section>

            <section>
              <h3 class="text-xl font-semibold mt-4">Frequently Asked Questions</h3>
              <div class="space-y-4">
                <div>
                  <p class="font-semibold">What is a bearing in trigonometry?</p>
                  <p>In trigonometry, a bearing is a direction or path along which something moves or lies.</p>
                </div>
                <div>
                  <p class="font-semibold">How do you calculate a bearing?</p>
                  <p>Bearing can be calculated using trigonometric functions such as sine and cosine, based on the known angles and distances.</p>
                </div>
                <div>
                  <p class="font-semibold">Why are bearings important?</p>
                  <p>Bearings are critical in navigation and engineering to determine the direction between two points.</p>
                </div>
                <div>
                  <p class="font-semibold">Can this calculator handle negative angles?</p>
                  <p>Yes, negative angles are converted to their positive equivalents in the normalization step.</p>
                </div>
              </div>
            </section>

              <summary class="cursor-pointer font-bold text-gray-800">Formulas</summary>
              <div class="mt-3 p-3 bg-white border border-gray-200 rounded-lg text-sm">
                <p><strong>Normalized Bearing:</strong></p>
                <p class="font-mono mt-2">bearing = ((endPoint − startPoint) mod 360 + 360) mod 360</p>
                <ul class="list-disc list-inside mt-2 text-gray-600">
                  <li><strong>startPoint:</strong> Starting angle in degrees.</li>
                  <li><strong>endPoint:</strong> Ending angle in degrees.</li>
                  <li><strong>Normalized bearing:</strong> Result forced into 0°–360° using modulo arithmetic.</li>

<script>
(function () {
  'use strict';

  const elements = {
    start: document.getElementById('startPoint'),
    end: document.getElementById('endPoint'),
    calcBtn: document.getElementById('calcBtn'),
    resetBtn: document.getElementById('resetBtn'),
    bearing: document.getElementById('bearingResult'),
    displayStart: document.getElementById('displayStart'),
    displayEnd: document.getElementById('displayEnd'),
    displayRaw: document.getElementById('displayRaw'),
    displayNormalized: document.getElementById('displayNormalized'),
    errorBox: document.getElementById('errorBox')
  };

  const defaultValues = {
    startPoint: '0',
    endPoint: '90'
  };

  const inputs = [elements.start, elements.end];
  const DEBOUNCE_DELAY = 100;

  function parseNumber(value) {
    if (value === null || value === undefined) {
      return null;
    }
    const trimmed = value.toString().trim();
    if (trimmed.length === 0) {
      return null;
    }
    const parsed = Number(trimmed);
    return Number.isFinite(parsed) ? parsed : null;
  }

  function parseInputs() {
    return {
      startPoint: parseNumber(elements.start.value),
      endPoint: parseNumber(elements.end.value)
    };
  }

  function validate(inputs) {
    const errors = [];
    if (inputs.startPoint === null) {
      errors.push('Start point is required and must be numeric.');
    }
    if (inputs.endPoint === null) {
      errors.push('End point is required and must be numeric.');
    }
    return { ok: errors.length === 0, errors };
  }

  function compute(inputs) {
    const rawDifference = inputs.endPoint - inputs.startPoint;
    const normalized = ((rawDifference % 360) + 360) % 360;
    return {
      startPoint: inputs.startPoint,
      endPoint: inputs.endPoint,
      rawDifference,
      normalized
    };
  }

  function roundNumber(value, precision = 2) {
    if (!Number.isFinite(value)) {
      return 0;
    }
    const factor = 10 ** precision;
    return Math.round((value + Number.EPSILON) * factor) / factor;
  }

  function format(outputs) {
    const formattedBearing = roundNumber(outputs.normalized, 2).toFixed(2);
    const formattedRaw = roundNumber(outputs.rawDifference, 2).toFixed(2);
    const formattedStart = roundNumber(outputs.startPoint, 2).toFixed(2);
    const formattedEnd = roundNumber(outputs.endPoint, 2).toFixed(2);
    return {
      bearing: `${formattedBearing}°`,
      rawDiff: `${formattedRaw}°`,
      start: `${formattedStart}°`,
      end: `${formattedEnd}°`
    };
  }

  function clearResults() {
    elements.bearing.textContent = '—';
    elements.displayStart.textContent = '—';
    elements.displayEnd.textContent = '—';
    elements.displayRaw.textContent = '—';
    elements.displayNormalized.textContent = '—';
  }

  function render(formatted, errors = []) {
    if (errors.length) {
      elements.errorBox.textContent = errors.join(' ');
      elements.errorBox.classList.add('show');
    } else {
      elements.errorBox.textContent = '';
      elements.errorBox.classList.remove('show');
    }

    if (!formatted) {
      clearResults();
      return;
    }

    elements.bearing.textContent = formatted.bearing;
    elements.displayStart.textContent = formatted.start;
    elements.displayEnd.textContent = formatted.end;
    elements.displayRaw.textContent = formatted.rawDiff;
    elements.displayNormalized.textContent = formatted.bearing;
  }

  function update() {
    const parsed = parseInputs();
    const validation = validate(parsed);
    if (!validation.ok) {
      render(null, validation.errors);
      return;
    }
    const outputs = compute(parsed);
    const formatted = format(outputs);
    render(formatted);
  }

  function debounce(fn, delay = DEBOUNCE_DELAY) {
    let timer;
    return (...args) => {
      if (timer) {
        clearTimeout(timer);
      }
      timer = setTimeout(() => fn(...args), delay);
    };
  }

  const debouncedUpdate = debounce(update, DEBOUNCE_DELAY);

  elements.calcBtn.addEventListener('click', update);
  elements.resetBtn.addEventListener('click', () => {
    elements.start.value = defaultValues.startPoint;
    elements.end.value = defaultValues.endPoint;
    update();
  });

  inputs.forEach((input) => {
    input.addEventListener('input', debouncedUpdate);
    input.addEventListener('change', debouncedUpdate);
  });

  update();
})();
</script>
