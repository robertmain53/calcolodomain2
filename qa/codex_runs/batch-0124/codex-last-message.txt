            <h3 class="text-xl font-semibold">Understanding the Barrel Variants</h3>
            <p>Unlike standardized units such as meters or kilograms, a "barrel" can hold different volumes depending on the commodity. The default US Oil Barrel (bbl) is 42 US gallons, but there are other definitions for liquid chemicals, beer, dry goods, and the British Imperial system.</p>

            <h3 class="text-xl font-semibold">Conversion Factors Used</h3>
            <ul class="list-disc pl-5 text-gray-700 space-y-2">
              <li><strong>1 US Oil Barrel (bbl) = 42 US Gallons</strong><br>This is the standard for measuring crude oil and other petroleum products.</li>
              <li><strong>1 US Liquid Barrel = 31.5 US Gallons</strong><br>Used for a few liquid commodities such as chemicals or food products.</li>
              <li><strong>1 US Beer Barrel = 31 US Gallons</strong><br>This volume is legally defined in the United States for taxation on beer production.</li>
              <li><strong>1 US Dry Barrel = 28.37759 US Gallons</strong><br>Equivalent to 105 dry quarts, commonly used for dry commodities.</li>
              <li><strong>1 UK (Imperial) Barrel = 43.23405 US Gallons</strong><br>Based on 36 Imperial gallons, the UK unit is bigger than the US gallon.</li>
            </ul>

            <h3 class="text-xl font-semibold">Frequently Asked Questions (FAQ)</h3>
            <h4 class="text-lg font-semibold">How many gallons are in a barrel of oil?</h4>
            <p>One standard US Oil Barrel equals exactly <strong>42 US gallons</strong>. This is the default used by international crude oil markets.</p>
            <h4 class="text-lg font-semibold">How many gallons are in a barrel of beer?</h4>
            <p>A US beer barrel is defined by federal law as <strong>31 US gallons</strong>; a keg is often a half-barrel (15.5 gallons).</p>
            <h4 class="text-lg font-semibold">Why is an oil barrel 42 gallons?</h4>
            <p>The 42-gallon size dates back to the 1870s in Pennsylvania, when oil producers repurposed whiskey barrels of that capacity for shipping crude. The size stuck to avoid disputes and create a consistent trading unit.</p>

      (function () {
        'use strict';

        const conversionFactors = {

          beer: 31.0,
          dry: 28.37759,
          uk: 43.23405
        };

        const barrelLabels = {

          liquid: 'US Liquid Barrel',
          beer: 'US Beer Barrel',
          dry: 'US Dry Barrel',
          uk: 'UK (Imperial) Barrel'
        };

        const defaults = {
          barrelValue: '1',

        const els = {
          barrelValue: document.getElementById('barrelValue'),
          gallonValue: document.getElementById('gallonValue'),
          barrelType: document.getElementById('barrelType'),
          calcBtn: document.getElementById('calcBtn'),
          resetBtn: document.getElementById('resetBtn'),
          convertedValue: document.getElementById('convertedValue'),
          convertedUnit: document.getElementById('convertedUnit'),
          baseValue: document.getElementById('baseValue'),
          typeValue: document.getElementById('typeValue'),
          factorValue: document.getElementById('factorValue'),
          inverseValue: document.getElementById('inverseValue'),
          insightText: document.getElementById('insightText'),
          errorBox: document.getElementById('errorBox')
        };

        let lastSource = 'barrel';

        const roundTo = (value, digits = 6) => {
          if (!Number.isFinite(value)) return 0;
          const factor = Math.pow(10, digits);
          return Math.round((value + Number.EPSILON) * factor) / factor;
        };

        const formatDecimal = (value, digits = 6) => {
          const rounded = roundTo(value, digits);
          const text = rounded.toFixed(digits);
          return text.replace(/\.0+$/, '').replace(/(\.\d*?[1-9])0+$/, '$1');
        };

        const parseNumber = (value) => {
          if (value === undefined || value === null || value === '') return null;
          const parsed = parseFloat(value);
          return Number.isFinite(parsed) ? parsed : null;
        };

        function parseInputs() {
          return {
            barrelValue: parseNumber(els.barrelValue.value),
            gallonValue: parseNumber(els.gallonValue.value),
            barrelType: els.barrelType.value,
            source: lastSource
          };
        }

        function validate(inputs) {
          const errors = [];
          const hasBarrel = Number.isFinite(inputs.barrelValue) && inputs.barrelValue > 0;
          const hasGallon = Number.isFinite(inputs.gallonValue) && inputs.gallonValue > 0;
          if (!hasBarrel && !hasGallon) {
            errors.push('Enter a positive value for barrels or gallons.');
          }
          if (!conversionFactors[inputs.barrelType]) {
            errors.push('Select a valid barrel type.');
          }
          if (Number.isFinite(inputs.barrelValue) && inputs.barrelValue < 0) {
            errors.push('Barrels must be 0 or greater.');
          }
          if (Number.isFinite(inputs.gallonValue) && inputs.gallonValue < 0) {
            errors.push('Gallons must be 0 or greater.');
          }
          return { ok: errors.length === 0, errors };
        }

        function compute(inputs) {
          const factor = conversionFactors[inputs.barrelType];
          if (!factor) return null;

          let barrels = null;
          let gallons = null;
          let source = inputs.source;

          const useGallonSource = source === 'gallon' && Number.isFinite(inputs.gallonValue) && inputs.gallonValue > 0;
          const useBarrelSource = Number.isFinite(inputs.barrelValue) && inputs.barrelValue > 0;

          if (useGallonSource) {
            gallons = inputs.gallonValue;
            barrels = gallons / factor;
          } else if (useBarrelSource) {
            barrels = inputs.barrelValue;
            gallons = barrels * factor;
            source = 'barrel';
          } else if (Number.isFinite(inputs.gallonValue) && inputs.gallonValue > 0) {
            gallons = inputs.gallonValue;
            barrels = gallons / factor;
            source = 'gallon';
          }

          if (!Number.isFinite(barrels) || !Number.isFinite(gallons)) return null;

          return {
            barrels: roundTo(barrels),
            gallons: roundTo(gallons),
            factor: roundTo(factor, 6),
            source
          };
        }

        function format(outputs, inputs) {
          if (!outputs) return null;
          const baseLabel = barrelLabels[inputs.barrelType] || 'Barrels';
          const convertedValue = outputs.source === 'barrel' ? outputs.gallons : outputs.barrels;
          const convertedUnit = outputs.source === 'barrel' ? 'US Gallons (Liquid)' : baseLabel;
          const baseValue = outputs.source === 'barrel' ? outputs.barrels : outputs.gallons;
          const baseUnit = outputs.source === 'barrel' ? baseLabel : 'US Gallons (Liquid)';
          const inverse = outputs.factor === 0 ? 0 : 1 / outputs.factor;
          return {
            convertedValue: formatDecimal(convertedValue, 6),
            convertedUnit,
            baseDisplay: `${formatDecimal(baseValue, 6)} ${baseUnit}`,
            typeDisplay: baseLabel,
            factorDisplay: `1 ${baseLabel} = ${formatDecimal(outputs.factor, 6)} US Gallons (Liquid)`,
            inverseDisplay: `1 US Gallon = ${formatDecimal(inverse, 6)} ${baseLabel}`
          };
        }

            els.errorBox.style.display = 'none';
            els.errorBox.textContent = '';
            return;
          }
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
        }

        function render(formatted, errors) {
          renderError(errors);
          if (!formatted) {
            els.convertedValue.textContent = '—';
            els.convertedUnit.textContent = 'US Gallons (Liquid)';
            els.baseValue.textContent = '—';
            els.typeValue.textContent = barrelLabels[els.barrelType.value] || 'Barrels';
            els.factorValue.textContent = '—';
            els.inverseValue.textContent = '—';
            els.insightText.textContent = 'Set your preferred barrel type and enter either barrels or gallons to expose the exact ratio used for the conversion.';
            return;
          }
          els.convertedValue.textContent = formatted.convertedValue;
          els.convertedUnit.textContent = formatted.convertedUnit;
          els.baseValue.textContent = formatted.baseDisplay;
          els.typeValue.textContent = formatted.typeDisplay;
          els.factorValue.textContent = formatted.factorDisplay;
          els.inverseValue.textContent = formatted.inverseDisplay;
          els.insightText.textContent = `Last computed using ${formatted.typeDisplay} and a deterministic rounding strategy.`;
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render(null, validation.errors);
            return;
          }
          const outputs = compute(inputs);
          if (!outputs) {
            render(null, ['Unable to compute with the current inputs.']);
            return;
          }
          render(format(outputs, inputs), []);
        }

          let timeout;
          return () => {
            clearTimeout(timeout);
            timeout = setTimeout(update, 100);
          };
