            <h2 class="text-2xl font-semibold">How to Use This Calculator</h2>
            <p>This calculator helps medical professionals determine the risk of hyperbilirubinemia in newborns. It is designed for use by pediatricians, neonatologists, and healthcare providers caring for newborns, offering a quick and reliable assessment tool.</p>
            <p>Enter the newborn's age in hours and the measured bilirubin level. Click Calculate to classify the risk tier and review the interpretation to guide follow-up or treatment decisions.</p>
            <ul class="list-disc pl-5 text-gray-700">
              <li>Use the age in hours to place the infant on the AAP phototherapy/risk nomogram.</li>
              <li>Bilirubin values are evaluated against the same nomogram ranges: high if above the 95th percentile or if the infant is less than 24 hours old, moderate when the value is elevated but below high risk, and low otherwise.</li>
              <li>Reset restores the default example inputs when you want to start a new evaluation.</li>
            </ul>

            <h2 class="text-2xl font-semibold mt-6">Methodology</h2>
            <p>All calculations are strictly based on the guidelines provided by the American Academy of Pediatrics. For more information, visit <a href="https://pediatrics.aappublications.org/" rel="noopener noreferrer" target="_blank" class="text-blue-600 hover:underline">AAP Pediatrics</a>. The calculator uses a risk assessment model based on bilirubin levels and age in hours; the formula and its application are detailed in the same medical literature accessible through the AAP.</p>
            <div class="bg-gray-50 border border-gray-200 rounded-lg p-4 text-sm text-gray-700">
              <p>The engine applies the following classification logic:</p>
              <ol class="list-decimal pl-5 mt-2 space-y-1 text-gray-600">
                <li>If the bilirubin level exceeds 15 mg/dL or the infant is under 24 hours old, mark the reading as <strong>High Risk</strong>.</li>
                <li>If the bilirubin level exceeds 10 mg/dL but does not meet the high-risk criteria, report <strong>Moderate Risk</strong>.</li>
                <li>All other combinations default to <strong>Low Risk</strong>, representing routine monitoring.</li>
              </ol>
            </div>

                <li><strong>Age in hours:</strong> The baby's age measured in hours since birth.</li>
                <li><strong>Bilirubin Level:</strong> The concentration of bilirubin in the blood, measured in milligrams per deciliter (mg/dL).</li>
                <li><strong>Risk Level:</strong> The calculated risk of developing hyperbilirubinemia.</li>

              <h3 class="text-xl font-semibold">Frequently Asked Questions (FAQ)</h3>
              <div class="space-y-6 pt-3">
                <div class="faq-item">
                  <h4>What is bilirubin?</h4>
                  <p>Bilirubin is a yellow compound that occurs in the normal catabolic pathway that breaks down heme in vertebrates.</p>
                </div>
                <div class="faq-item">
                  <h4>Why is bilirubin important in newborns?</h4>
                  <p>High levels of bilirubin in newborns can lead to jaundice and, in severe cases, can cause brain damage.</p>
                </div>
                <div class="faq-item">
                  <h4>How is the risk level determined?</h4>
                  <p>The risk is assessed using age-specific nomograms which are based on the bilirubin level and the infant's age in hours.</p>
                </div>
                <div class="faq-item">
                  <h4>Can this tool replace medical advice?</h4>
                  <p>No, this tool is not a substitute for professional medical advice. Always consult a healthcare provider for medical concerns.</p>
                </div>
                <div class="faq-item">
                  <h4>Is this calculator suitable for all newborns?</h4>
                  <p>This tool is specifically designed for newborns 35 weeks gestation or older. Consult your healthcare provider for other cases.</p>

<script>
(function() {
  'use strict';

  const defaults = {
    ageHours: '48',
    bilirubinLevel: '11'
  };

  const els = {
    ageHours: document.getElementById('ageHours'),
    bilirubinLevel: document.getElementById('bilirubinLevel'),
    calcBtn: document.getElementById('calcBtn'),
    resetBtn: document.getElementById('resetBtn'),
    errorBox: document.getElementById('errorBox'),
    riskLevel: document.getElementById('riskLevel'),
    riskInterpretation: document.getElementById('riskInterpretation'),
    ageSummary: document.getElementById('ageSummary'),
    bilirubinSummary: document.getElementById('bilirubinSummary')
  };

  const debounce = (fn, delay = 100) => {
    let timer;
    return (...args) => {
      clearTimeout(timer);
      timer = setTimeout(() => fn(...args), delay);
    };
  };

  const round1 = (value) => Math.round((value + Number.EPSILON) * 10) / 10;

  function parseInputs() {
    return {
      ageHours: Number(els.ageHours.value),
      bilirubinLevel: Number(els.bilirubinLevel.value)
    };
  }

  function validate(inputs) {
    const errors = [];
    if (!Number.isFinite(inputs.ageHours) || inputs.ageHours <= 0) {
      errors.push('Age in hours must be a number greater than 0.');
    }
    if (!Number.isFinite(inputs.bilirubinLevel) || inputs.bilirubinLevel < 0) {
      errors.push('Bilirubin level must be 0 or greater.');
    }
    return { ok: errors.length === 0, errors };
  }

  function compute(inputs) {
    let riskLabel = 'Low Risk';
    if (inputs.bilirubinLevel > 15 || inputs.ageHours < 24) {
      riskLabel = 'High Risk';
    } else if (inputs.bilirubinLevel > 10) {
      riskLabel = 'Moderate Risk';
    }
    let interpretation = 'Continue routine monitoring per AAP guidance.';
    if (riskLabel === 'High Risk') {
      interpretation = 'High Risk: follow the AAP hyperbilirubinemia pathway and consult a neonatologist.';
    } else if (riskLabel === 'Moderate Risk') {
      interpretation = 'Moderate Risk: monitor levels closely and consider follow-up testing.';
    }
    return { riskLabel, interpretation };
  }

  function formatResult(outputs, inputs) {
    const formatValue = (value) => {
      if (!Number.isFinite(value)) return '—';
      return Number.isInteger(value) ? String(value) : round1(value).toFixed(1);
    };
    return {
      riskLabel: outputs.riskLabel,
      interpretation: outputs.interpretation,
      ageSummary: `${formatValue(inputs.ageHours)} hrs`,
      bilirubinSummary: `${formatValue(inputs.bilirubinLevel)} mg/dL`
    };
  }

  function renderError(errors) {
    if (!errors || errors.length === 0) {
      els.errorBox.style.display = 'none';
      els.errorBox.textContent = '';
      return;
    }
    els.errorBox.style.display = 'block';
    els.errorBox.textContent = errors.join(' ');
  }

  function renderResults(formatted) {
    if (!formatted) return;
    els.riskLevel.textContent = formatted.riskLabel;
    els.riskInterpretation.textContent = formatted.interpretation;
    els.ageSummary.textContent = formatted.ageSummary;
    els.bilirubinSummary.textContent = formatted.bilirubinSummary;
  }

  function render(formatted, errors = []) {
    renderError(errors);
    if (formatted && errors.length === 0) {
      renderResults(formatted);
    }
  }

  function update() {
    const inputs = parseInputs();
    const validation = validate(inputs);
    if (!validation.ok) {
      render(null, validation.errors);
      return;
    }
    const computed = compute(inputs);
    const formatted = formatResult(computed, inputs);
    render(formatted, []);
  }

  els.calcBtn.addEventListener('click', update);
  els.resetBtn.addEventListener('click', () => {
    els.ageHours.value = defaults.ageHours;
    els.bilirubinLevel.value = defaults.bilirubinLevel;
    update();
  });

  const debouncedUpdate = debounce(update, 100);
  document.querySelectorAll('#inputsCard input').forEach((input) => {
    input.addEventListener('input', debouncedUpdate);
    input.addEventListener('change', debouncedUpdate);
  });

  update();
})();
</script>
