        <p>
          This tool assumes a regular decagon (n=10) and uses exact trigonometric identities to derive side length, apothem,

        </p>

            <p>\[A = \frac{n}{4} s^2 \cot\left(\frac{\pi}{n}\right) = \frac{5}{2} s^2 \cot\left(\frac{\pi}{10}\right) = \frac{1}{2} P a\]</p>

    (function() {
      'use strict';

      const debounce = (fn, delay = 100) => {
        let timer;
        return (...args) => {
          clearTimeout(timer);
          timer = setTimeout(() => fn(...args), delay);
        };
      };

      const roundTo = (value, decimals) => {
        const safe = Number.isFinite(value) ? value : 0;
        const factor = Math.pow(10, decimals);
        return Math.round((safe + Number.EPSILON) * factor) / factor;
      };

      const clampPrecision = (value) => {
        if (!Number.isFinite(value)) return 4;
        return Math.max(0, Math.min(10, Math.round(value)));
      };

        angles: document.getElementById('anglesOutput'),
        errorBox: document.getElementById('errorBox')
      };

      const defaults = {
        known: 'side',
        unit: 'cm',
        value: '12',
        precision: '4'
      };

      const consts = {
        n: 10,
        PI: Math.PI
      };

      const trig = (() => {
        const alpha = consts.PI / consts.n;
        return {
          tanAlpha: Math.tan(alpha),
          sinAlpha: Math.sin(alpha),
          cotAlpha: 1 / Math.tan(alpha)
        };
      })();

      let latestOutputs = null;
      const highlightDefault = 'Highlighted quantity';

      function parseInputs() {
        return {
          known: els.known.value,
          unit: els.unit.value,
          value: parseFloat(els.value.value),
          precision: parseFloat(els.precision.value)
        };
      }

      function validate(inputs) {
        const errors = [];
        const validFields = ['side', 'apothem', 'radius', 'perimeter', 'area'];
        if (!validFields.includes(inputs.known)) {
          errors.push('Please select a known quantity.');
        }
        if (!Number.isFinite(inputs.value) || inputs.value <= 0) {
          errors.push('Enter a number greater than zero.');
        }
        if (!inputs.unit) {
          errors.push('Choose a linear unit.');
        }
        inputs.precision = clampPrecision(inputs.precision);
        return { ok: errors.length === 0, errors, precision: inputs.precision };
      }

      function compute(inputs) {
        const { value, known } = inputs;
        const { n } = consts;
        const { sinAlpha, tanAlpha, cotAlpha } = trig;
        const results = {
          s: null,
          a: null,
          R: null,
          P: null,
          A: null
        };

        const builders = {
          side: () => value,
          apothem: () => 2 * value * tanAlpha,
          radius: () => 2 * value * sinAlpha,
          perimeter: () => value / n,
          area: () => Math.sqrt((4 * value) / (n * cotAlpha))
        };

        const sideLength = builders[known] ? builders[known]() : null;
        if (!Number.isFinite(sideLength) || sideLength <= 0) {
          return null;
        }

        results.s = sideLength;
        results.P = n * sideLength;
        results.a = sideLength / (2 * tanAlpha);
        results.R = sideLength / (2 * sinAlpha);
        results.A = (n / 4) * sideLength * sideLength * cotAlpha;

        return results;
      }

        if (!outputs) return null;
        const precision = inputs.precision;
        const linearUnit = inputs.unit || 'units';
        const areaUnit = `${linearUnit}²`;

        const fmtLinear = (value) => `${roundTo(value, precision).toFixed(precision)} ${linearUnit}`;
        const fmtArea = (value) => `${roundTo(value, precision).toFixed(precision)} ${areaUnit}`;

        const highlightMap = {
          side: fmtLinear(outputs.s),
          apothem: fmtLinear(outputs.a),
          radius: fmtLinear(outputs.R),
          perimeter: fmtLinear(outputs.P),
          area: fmtArea(outputs.A)
        };

        return {
          highlight: highlightMap[inputs.known] || fmtLinear(outputs.s),
          display: {
            s: fmtLinear(outputs.s),
            a: fmtLinear(outputs.a),
            R: fmtLinear(outputs.R),
            P: fmtLinear(outputs.P),
            A: fmtArea(outputs.A)
          }
        };
      }

      function renderError(errors) {
        if (!errors || errors.length === 0) {
          els.errorBox.style.display = 'none';
          els.errorBox.textContent = '';
          return;
        }
        els.errorBox.style.display = 'block';
        els.errorBox.textContent = errors.join(' ');
      }

      function render(formatted) {
        if (!formatted) return;
        els.highlightValue.textContent = formatted.highlight;
        els.side.textContent = formatted.display.s;
        els.apothem.textContent = formatted.display.a;
        els.radius.textContent = formatted.display.R;
        els.perimeter.textContent = formatted.display.P;
        els.area.textContent = formatted.display.A;
        els.angles.textContent = 'Interior 144°, Central 36°, Exterior 36°';
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          renderError(validation.errors);
          latestOutputs = null;
          return;
        }
        renderError([]);
        const results = compute(inputs);
        if (!results) {
          renderError(['Unable to compute decagon values with the current inputs.']);
          latestOutputs = null;
          return;
        }
        inputs.precision = validation.precision;
        latestOutputs = { inputs, results };
        const formatted = format(results, inputs);
        render(formatted);
      }

      function reset() {
        els.known.value = defaults.known;
        els.unit.value = defaults.unit;
        els.value.value = defaults.value;
        els.precision.value = defaults.precision;
        els.highlightLabel.textContent = highlightDefault;
        update();
      }

      function copyResults() {
        if (!latestOutputs) return;
        const formatted = format(latestOutputs.results, latestOutputs.inputs);
        if (!formatted) return;
        const lines = [
          `Decagon results (${new Date().toISOString()}):`,
          `• Side (s): ${formatted.display.s}`,
          `• Apothem (a): ${formatted.display.a}`,
          `• Circumradius (R): ${formatted.display.R}`,
          `• Perimeter (P): ${formatted.display.P}`,
          `• Area (A): ${formatted.display.A}`,
          `• Angles: Interior 144°, Central 36°, Exterior 36°`
        ];
        navigator.clipboard.writeText(lines.join('\n')).then(() => {
          els.highlightLabel.textContent = 'Results copied';
          setTimeout(() => { els.highlightLabel.textContent = highlightDefault; }, 1200);
        }).catch(() => {
          els.highlightLabel.textContent = 'Copy failed';
          setTimeout(() => { els.highlightLabel.textContent = highlightDefault; }, 1200);
        });
      }

      const debouncedUpdate = debounce(update, 100);

      document.querySelectorAll('#inputsCard input, #inputsCard select, #inputsCard textarea')
        .forEach((el) => {
          el.addEventListener('input', debouncedUpdate);
          el.addEventListener('change', debouncedUpdate);
        });

      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', reset);
      els.copyBtn.addEventListener('click', copyResults);

      update();
    })();
  </script>
