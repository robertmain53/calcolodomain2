          <h2 class="text-2xl font-semibold">Methodology</h2>
          <p class="text-gray-700">The calculator first computes the out-the-door price by applying tax to the price after any trade-in adjustment, then optionally adds financed fees. It subtracts upfront equity (down payment + trade-in equity) and simulates an amortization schedule period-by-period using the standard loan formula to accumulate interest, principal, and balance until the loan is paid off. Ownership cost adds any optional monthly insurance or maintenance to the periodic payment.</p>

      <section class="card meta-section space-y-4 mt-10" id="pageMeta" aria-label="Verification, changelog, formulas, citations">
        <details id="formulaDetails">
          <summary class="font-bold text-gray-800">Formulas</summary>
          <div class="mt-3 p-3 bg-white border border-gray-200 rounded-lg text-sm text-gray-700">
            <p><strong>Out-the-door price:</strong> Tax is applied to the price after any trade-in adjustment: <span class="font-mono">TaxableBase = max(0, Price - (Apply trade-in ? Trade-in : 0))</span>. <span class="font-mono">OTD = Price + TaxableBase × (SalesTax/100) + Fees</span>.</p>
            <p class="mt-2"><strong>Amount financed:</strong> <span class="font-mono">Financed = OTD - DownPayment - max(0, TradeIn - TradePayoff) + (FinanceFees ? Fees : 0)</span>, floored at zero.</p>
            <p class="mt-2"><strong>Periodic payment:</strong> <span class="font-mono">Payment = P × r / (1 - (1 + r)<sup>-n</sup>)</span>, where P = amount financed, r = APR/100 ÷ periodsPerYear, and n = round(Term × periodsPerYear). Zero-rate loans divide principal by n.</p>
            <p class="mt-2"><strong>Variables:</strong></p>
            <ul class="list-disc list-inside text-gray-600 mt-1">
              <li>P: Amount financed after down payment &amp; equity.</li>
              <li>r: Periodic interest rate (APR / 100 ÷ payments per year).</li>
              <li>n: Total payments (term in chosen unit × payments per year).</li>
              <li>Fees: Title, dealer, and optional financed fees.</li>
              <li>Ownership cost: Payment + monthly insurance/maintenance.</li>
            </ul>
          </div>
        </details>

(function() {
  const round2 = (x) => Number.isFinite(x) ? Math.round((x + Number.EPSILON) * 100) / 100 : 0;
  const fmtCurrency = (value) => {
    try {
      return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(Number.isFinite(value) ? value : 0);
    } catch {
      return '$' + round2(value).toFixed(2);
    }
  };

  const refs = {
    boatPrice: document.getElementById('boatPrice'),
    salesTax: document.getElementById('salesTax'),
    titleFees: document.getElementById('titleFees'),
    dealerFees: document.getElementById('dealerFees'),
    financeFees: document.getElementById('financeFees'),
    applyTradeToTax: document.getElementById('applyTradeToTax'),
    downPayment: document.getElementById('downPayment'),
    tradeIn: document.getElementById('tradeIn'),
    tradePayoff: document.getElementById('tradePayoff'),
    apr: document.getElementById('apr'),
    term: document.getElementById('term'),
    termUnit: document.getElementById('termUnit'),
    frequency: document.getElementById('frequency'),
    startMonth: document.getElementById('startMonth'),
    insMaint: document.getElementById('insMaint'),
    outPayment: document.getElementById('outPayment'),
    outFinanced: document.getElementById('outFinanced'),
    outInterest: document.getElementById('outInterest'),
    outTotalCost: document.getElementById('outTotalCost'),
    outOtd: document.getElementById('outOtd'),
    outOwnership: document.getElementById('outOwnership'),
    errorBox: document.getElementById('errorBox'),
    payoffTime: document.getElementById('payoffTime'),
    scheduleWrap: document.getElementById('scheduleWrap'),
    scheduleBody: document.getElementById('scheduleBody'),
    toggleSchedule: document.getElementById('toggleSchedule'),
    downloadCsv: document.getElementById('downloadCsv'),
    calcBtn: document.getElementById('calcBtn'),
    resetBtn: document.getElementById('resetBtn')
  };

  const defaults = {
    boatPrice: 45000,
    salesTax: 7,
    titleFees: 300,
    dealerFees: 500,
    financeFees: true,
    applyTradeToTax: true,
    downPayment: 5000,
    tradeIn: 0,
    tradePayoff: 0,
    apr: 7.49,
    term: 10,
    termUnit: 'years',
    frequency: 'monthly',
    startMonth: '',
    insMaint: 120
  };

  const state = {
    schedule: []
  };

  const debounce = (fn, delay = 100) => {
    let timer;
    return (...args) => {
      clearTimeout(timer);
      timer = setTimeout(() => fn(...args), delay);
    };
  };

  function periodsPerYear(freq) {
    return freq === 'weekly' ? 52 : freq === 'biweekly' ? 26 : 12;
  }

  function parseInputs() {
    return {
      boatPrice: parseFloat(refs.boatPrice.value),
      salesTax: parseFloat(refs.salesTax.value),
      titleFees: parseFloat(refs.titleFees.value),
      dealerFees: parseFloat(refs.dealerFees.value),
      financeFees: refs.financeFees.checked,
      applyTradeToTax: refs.applyTradeToTax.checked,
      downPayment: parseFloat(refs.downPayment.value),
      tradeIn: parseFloat(refs.tradeIn.value),
      tradePayoff: parseFloat(refs.tradePayoff.value),
      apr: parseFloat(refs.apr.value),
      term: parseFloat(refs.term.value),
      termUnit: refs.termUnit.value,
      frequency: refs.frequency.value,
      startMonth: refs.startMonth.value,
      insMaint: parseFloat(refs.insMaint.value)
    };
  }

  function validate(inputs) {
    const errors = [];
    if (!Number.isFinite(inputs.boatPrice) || inputs.boatPrice < 0) {
      errors.push('Boat price must be 0 or greater.');
    }
    if (!Number.isFinite(inputs.salesTax) || inputs.salesTax < 0 || inputs.salesTax > 25) {
      errors.push('Sales tax must be between 0% and 25%.');
    }
    ['titleFees', 'dealerFees', 'downPayment', 'tradeIn', 'tradePayoff', 'insMaint'].forEach((field) => {
      if (!Number.isFinite(inputs[field]) || inputs[field] < 0) {
        errors.push(`${field} must be 0 or greater.`);
      }
    });
    if (!Number.isFinite(inputs.apr) || inputs.apr < 0 || inputs.apr > 99) {
      errors.push('APR must be between 0% and 99%.');
    }
    if (!Number.isFinite(inputs.term) || inputs.term < 0.25) {
      errors.push('Term must be at least 0.25 units.');
    }
    const freqSet = new Set(['monthly', 'biweekly', 'weekly']);
    if (!freqSet.has(inputs.frequency)) {
      errors.push('Payment frequency is invalid.');
    }
    const unitSet = new Set(['years', 'months']);
    if (!unitSet.has(inputs.termUnit)) {
      errors.push('Term unit is invalid.');
    }
    return { ok: errors.length === 0, errors };
  }

  function compute(inputs) {
    const price = inputs.boatPrice;
    const taxRate = inputs.salesTax / 100;
    const fees = inputs.titleFees + inputs.dealerFees;
    const tradeAdj = inputs.applyTradeToTax ? inputs.tradeIn : 0;
    const taxableBase = Math.max(0, price - tradeAdj);
    const tax = taxableBase * taxRate;
    const otd = price + tax + fees;
    const equity = Math.max(0, inputs.tradeIn - inputs.tradePayoff);
    let financed = otd - inputs.downPayment - equity + (inputs.financeFees ? fees : 0);
    financed = Math.max(0, financed);

    const ppy = periodsPerYear(inputs.frequency);
    let periods = inputs.term;
    if (inputs.termUnit === 'years') {
      periods = periods * ppy;
    } else if (inputs.termUnit === 'months') {
      periods = (periods / 12) * ppy;
    }
    periods = Math.max(1, Math.round(periods));

    const aprDecimal = inputs.apr / 100;
    const r = aprDecimal / ppy;

    if (price < 0 || !Number.isFinite(price) || !Number.isFinite(r) || periods <= 0) {
      return null;
    }

    let payment = 0;
    if (r === 0) {
      payment = financed / periods;
    } else {
      const denominator = 1 - Math.pow(1 + r, -periods);
      if (denominator === 0) return null;
      payment = financed * r / denominator;
    }

    if (!Number.isFinite(payment)) return null;

    const schedule = [];
    let balance = financed;
    const today = inputs.startMonth ? new Date(inputs.startMonth + '-01') : new Date();
    const daysPer = Math.round(365 / ppy) || 30;

    for (let i = 1; i <= periods; i++) {
      if (balance <= 0) {
        break;
      }
      const interest = r === 0 ? 0 : balance * r;
      const principal = Math.min(payment - interest, balance);
      balance = Math.max(0, balance - principal);
      const date = new Date(today);
      date.setDate(date.getDate() + i * daysPer);
      const periodPayment = r === 0 ? payment : principal + interest;
      schedule.push({
        num: i,
        date,
        payment: round2(periodPayment),
        interest: round2(interest),
        principal: round2(principal),
        balance: round2(balance)
      });
    }

    const totalInterest = round2(schedule.reduce((sum, row) => sum + row.interest, 0));
    const totalCost = round2(financed + totalInterest);
    const ownership = round2(payment * (inputs.frequency === 'monthly' ? 1 : (12 / ppy)) + inputs.insMaint);

    return {
      payment,
      financed: round2(financed),
      totalInterest,
      totalCost,
      otd: round2(otd),
      schedule,
      ownership,
      periods
    };
  }

  function formatResults(out, inputs) {
    return {
      payment: fmtCurrency(out.payment),
      financed: fmtCurrency(out.financed),
      totalInterest: fmtCurrency(out.totalInterest),
      totalCost: fmtCurrency(out.totalCost),
      otd: fmtCurrency(out.otd),
      ownership: fmtCurrency(out.ownership),
      payoff: `${out.schedule.length} payments (~${payoffLabel(out.schedule.length, periodsPerYear(inputs.frequency))})`
    };
  }

  function payoffLabel(count, freq) {
    const years = count / freq;
    const wholeYears = Math.floor(years);
    const remMonths = Math.round((years - wholeYears) * 12);
    if (wholeYears <= 0) return `${Math.max(1, remMonths)} month(s)`;
    if (remMonths <= 0) return `${wholeYears} year(s)`;
    return `${wholeYears} yr ${remMonths} mo`;
  }

  function renderSchedule() {
    refs.scheduleBody.innerHTML = '';
    const rows = state.schedule.slice(0, 360);
    const fragment = document.createDocumentFragment();
    rows.forEach(row => {
      const tr = document.createElement('tr');
      tr.innerHTML = `
        <td>${row.num}</td>
        <td>${new Intl.DateTimeFormat('en-US', { year: 'numeric', month: 'short' }).format(row.date)}</td>
        <td>${fmtCurrency(row.payment)}</td>
        <td>${fmtCurrency(row.interest)}</td>
        <td>${fmtCurrency(row.principal)}</td>
        <td>${fmtCurrency(row.balance)}</td>
      `;
      fragment.appendChild(tr);
    });
    refs.scheduleBody.appendChild(fragment);
  }

  function render(formatted, errors = []) {
    if (errors.length) {
      refs.errorBox.style.display = 'block';
      refs.errorBox.textContent = errors.join(' ');
    } else {
      refs.errorBox.style.display = 'none';
      refs.errorBox.textContent = '';
    }

    if (!formatted) {
      refs.outPayment.textContent = '$0.00';
      refs.outFinanced.textContent = '$0.00';
      refs.outInterest.textContent = '$0.00';
      refs.outTotalCost.textContent = '$0.00';
      refs.outOtd.textContent = '$0.00';
      refs.outOwnership.textContent = '$0.00';
      refs.payoffTime.textContent = '—';
      refs.downloadCsv.disabled = true;
      state.schedule = [];
      refs.scheduleBody.innerHTML = '';
      return;
    }

    refs.outPayment.textContent = formatted.payment;
    refs.outFinanced.textContent = formatted.financed;
    refs.outInterest.textContent = formatted.totalInterest;
    refs.outTotalCost.textContent = formatted.totalCost;
    refs.outOtd.textContent = formatted.otd;
    refs.outOwnership.textContent = formatted.ownership;
    refs.payoffTime.textContent = formatted.payoff;
    refs.downloadCsv.disabled = false;
    if (refs.scheduleWrap.style.display === 'block') {
      renderSchedule();
    }
  }

  function update() {
    const inputs = parseInputs();
    const validation = validate(inputs);
    if (!validation.ok) {
      refs.downloadCsv.disabled = true;
      render(null, validation.errors);
      return;
    }
    const output = compute(inputs);
    if (!output) {
      refs.downloadCsv.disabled = true;
      render(null, ['Calculation failed for the current inputs. Please revise values.']);
      return;
    }
    state.schedule = output.schedule;
    const formatted = formatResults(output, inputs);
    render(formatted);
  }

  function generateCsv() {
    if (!state.schedule.length) return '';
    const header = ['#','Date','Total Payment','Interest','Principal','Balance'];
    const rows = state.schedule.map(row => {
      const date = new Intl.DateTimeFormat('en-US', { year: 'numeric', month: 'short' }).format(row.date);
      return [row.num, date, row.payment, row.interest, row.principal, row.balance].join(',');
    });
    return `${header.join(',')}\n${rows.join('\n')}`;
  }

  refs.calcBtn.addEventListener('click', update);
  refs.resetBtn.addEventListener('click', () => {
    Object.keys(defaults).forEach(key => {
      if (key === 'financeFees' || key === 'applyTradeToTax') {
        refs[key].checked = defaults[key];
      } else {
        refs[key].value = defaults[key];
      }
    });
    refs.scheduleWrap.style.display = 'none';
    refs.toggleSchedule.textContent = 'View amortization schedule';
    update();
  });

  refs.toggleSchedule.addEventListener('click', () => {
    const isHidden = refs.scheduleWrap.style.display !== 'block';
    refs.scheduleWrap.style.display = isHidden ? 'block' : 'none';
    refs.toggleSchedule.textContent = isHidden ? 'Hide amortization schedule' : 'View amortization schedule';
    if (isHidden) renderSchedule();
  });

  refs.downloadCsv.addEventListener('click', () => {
    if (!state.schedule.length) return;
    const csv = generateCsv();
    const blob = new Blob([csv], { type: 'text/csv' });
    const url = URL.createObjectURL(blob);
    const anchor = document.createElement('a');
    anchor.href = url;
    anchor.download = 'boat-loan-schedule.csv';
    anchor.click();
    URL.revokeObjectURL(url);
  });

  const debouncedUpdate = debounce(update, 100);
  document.querySelectorAll('#inputsCard input, #inputsCard select')

      el.addEventListener('input', debouncedUpdate);
      el.addEventListener('change', debouncedUpdate);
    });

  update();
})();
