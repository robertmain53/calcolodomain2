        <p>ABSI = 85 / (24^(2/3) * 30^(1/2)) = 0.0800</p>

  <script>
    (() => {
      'use strict';

      const els = {
        waist: document.getElementById('waistCircumference'),
        bmi: document.getElementById('bmi'),
        age: document.getElementById('age'),
        absiValue: document.getElementById('absiValue'),
        errorBox: document.getElementById('errorBox'),
        calcBtn: document.getElementById('calcBtn'),
        resetBtn: document.getElementById('resetBtn')
      };

      const round4 = (x) => Math.round((x + Number.EPSILON) * 10000) / 10000;

      function parseInputs() {
        return {
          waist: parseFloat(els.waist.value),
          bmi: parseFloat(els.bmi.value),
          age: parseFloat(els.age.value)
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!Number.isFinite(inputs.waist) || inputs.waist <= 0) errors.push('Please enter a valid waist circumference.');
        if (!Number.isFinite(inputs.bmi) || inputs.bmi <= 0) errors.push('Please enter a valid BMI.');
        if (!Number.isFinite(inputs.age) || inputs.age <= 0) errors.push('Please enter a valid age.');
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const absi = inputs.waist / (Math.pow(inputs.bmi, 2 / 3) * Math.pow(inputs.age, 1 / 2));
        return { absi };
      }

      function format(outputs) {
        const safe = Number.isFinite(outputs.absi) ? outputs.absi : 0;
        return { absiText: round4(safe).toFixed(4) };
      }

      function render(formatted, errors) {
        if (errors && errors.length) {
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
        } else {
          els.errorBox.style.display = 'none';
          els.errorBox.textContent = '';
        }
        if (!formatted) return;
        els.absiValue.textContent = formatted.absiText;
      }

      function update() {
        const inputs = parseInputs();
        const v = validate(inputs);
        if (!v.ok) {
          render(null, v.errors);
          return;
        }
        const outputs = compute(inputs);
        const formatted = format(outputs);
        render(formatted, []);
      }

      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', () => {
        document.querySelectorAll('input').forEach(i => { i.value = i.defaultValue; });
        document.querySelectorAll('select').forEach(s => { s.value = s.querySelector('option[selected]')?.value || s.value; });
        update();
      });

      let t = null;
      document.querySelectorAll('input, select').forEach(el => {
        el.addEventListener('input', () => {
          clearTimeout(t);
          t = setTimeout(update, 100);
        });
        el.addEventListener('change', () => {
          clearTimeout(t);
          t = setTimeout(update, 100);
        });
      });

      update();
    })();
  </script>
