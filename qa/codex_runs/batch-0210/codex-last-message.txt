(function() {
  'use strict';

  const debounce = (fn, delay = 100) => {
    let timeout;
    return (...args) => {
      clearTimeout(timeout);
      timeout = setTimeout(() => fn(...args), delay);
    };
  };

  const roundTo = (value, decimals = 6) => {
    const factor = Math.pow(10, decimals);
    return Math.round((value + Number.EPSILON) * factor) / factor;
  };

  const formatValue = (value) => {
    if (!Number.isFinite(value) || value === 0) {
      return '0';
    }
    const abs = Math.abs(value);
    if (abs >= 1e9 || abs < 1e-3) {
      return value.toExponential(6);
    }
    const formatted = roundTo(value, 6).toFixed(6).replace(/\.0+$/, '').replace(/(\.[0-9]*?)0+$/, '$1').replace(/\.$/, '');
    return formatted;
  };

  const els = {
    direction: document.getElementById('direction'),
    valueInput: document.getElementById('valueInput'),
    resultValue: document.getElementById('resultValue'),
    resultLabel: document.getElementById('resultLabel'),
    resultUnit: document.getElementById('resultUnit'),
    resultInputInfo: document.getElementById('resultInputInfo'),
    resultRate: document.getElementById('resultRate'),
    errorBox: document.getElementById('errorBox'),
    calcBtn: document.getElementById('calcBtn'),
    resetBtn: document.getElementById('resetBtn')
  };

  const FACTOR = 1055.05585262;
  const RECIPROCAL = 1 / FACTOR;

  function parseInputs() {
    return {
      direction: els.direction.value,
      value: parseFloat(els.valueInput.value)
    };
  }

  function validate(inputs) {
    const errors = [];
    if (!Number.isFinite(inputs.value)) {
      errors.push('Enter a valid number.');
    }
    return { ok: errors.length === 0, errors };
  }

  function compute(inputs) {
    const normalized = Number.isFinite(inputs.value) ? inputs.value : 0;
    const isForward = inputs.direction === 'btu-to-j';
    const resultValue = isForward ? normalized * FACTOR : normalized / FACTOR;
    const directionLabel = isForward ? 'BTU → Joules' : 'Joules → BTU';
    const inputLabel = isForward ? `${formatValue(normalized)} BTU` : `${formatValue(normalized)} J`;
    const outputUnit = isForward ? 'J' : 'BTU';
    const rateLabel = isForward ? `1 BTU = ${formatValue(FACTOR)} J` : `1 J = ${formatValue(RECIPROCAL)} BTU`;
    return { resultValue, inputLabel, outputUnit, rateLabel, directionLabel };
  }

  function format(outputs, inputs) {
    return {
      converted: formatValue(outputs.resultValue),
      inputSnapshot: outputs.inputLabel,
      rate: outputs.rateLabel,
      unit: outputs.outputUnit,
      direction: outputs.directionLabel
    };
  }

  function render(formatted, errors) {
    if (errors && errors.length > 0) {
      els.errorBox.style.display = 'block';
      els.errorBox.textContent = errors.join(' ');
      return;
    }
    els.errorBox.style.display = 'none';
    els.errorBox.textContent = '';
    els.resultValue.textContent = `${formatted.converted} ${formatted.unit}`;
    els.resultInputInfo.textContent = formatted.inputSnapshot;
    els.resultRate.textContent = formatted.rate;
    els.resultUnit.textContent = formatted.unit;
    els.resultLabel.textContent = formatted.direction;
  }

  function update() {
    const inputs = parseInputs();
    const validation = validate(inputs);
    if (!validation.ok) {
      render({}, validation.errors);
      return;
    }
    const outputs = compute(inputs);
    const formatted = format(outputs, inputs);
    render(formatted, []);
  }

  const debouncedUpdate = debounce(update, 100);

  document.querySelectorAll('#inputsCard input, #inputsCard select, #inputsCard textarea').forEach((el) => {
    el.addEventListener('input', debouncedUpdate);
    el.addEventListener('change', debouncedUpdate);
  });

  els.calcBtn.addEventListener('click', update);
  els.resetBtn.addEventListener('click', () => {
    els.direction.value = 'btu-to-j';
    els.valueInput.value = '1';
    update();
  });

  update();
})();
