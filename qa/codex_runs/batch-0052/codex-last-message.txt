        <p>Flow Rate is computed using the product of the three primary inputs:</p>
        <pre style="margin:8px 0; padding:12px; background:#f1f5f9; border-radius:8px;">
Flow Rate = Area Size × Permeability × Water Table Level
        </pre>

        <h3>Glossary of Variables</h3>

          <li><strong>Water Table Level:</strong> The depth of the groundwater relative to the surface.</li>

    (function() {

      'use strict';

      const debounce = (fn, delay = 100) => {
        let timer;
        return (...args) => {
          clearTimeout(timer);
          timer = setTimeout(() => fn(...args), delay);
        };
      };

      const round2 = (value) => Math.round(((Number.isFinite(value) ? value : 0) + Number.EPSILON) * 100) / 100;

      const fmtNumber = (value) => {
        if (!Number.isFinite(value)) return '—';
        return new Intl.NumberFormat('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 }).format(value);
      };

      const formatFlow = (value) => {
        if (!Number.isFinite(value)) return '—';
        return `${fmtNumber(value)} m³/day`;
      };

      const defaults = {
        areaSize: 100,
        permeability: 0.5,
        waterTable: 2
      };

      const els = {
        areaSize: document.getElementById('areaSize'),
        permeability: document.getElementById('permeability'),
        waterTable: document.getElementById('waterTable'),

        totalFlowRate: document.getElementById('totalFlowRate'),
        perAreaRate: document.getElementById('perAreaRate'),
        permeabilityEffect: document.getElementById('permeabilityEffect'),
        waterTableEffect: document.getElementById('waterTableEffect'),
        confidenceLabel: document.getElementById('confidenceLabel'),

        errorBox: document.getElementById('errorBox'),
        calcBtn: document.getElementById('calcBtn'),
        resetBtn: document.getElementById('resetBtn')
      };

      function parseInputs() {
        return {
          areaSize: parseFloat(els.areaSize.value),
          permeability: parseFloat(els.permeability.value),
          waterTable: parseFloat(els.waterTable.value)
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!Number.isFinite(inputs.areaSize) || inputs.areaSize <= 0) {
          errors.push('Area size must be greater than 0.');
        }
        if (!Number.isFinite(inputs.permeability) || inputs.permeability <= 0) {
          errors.push('Permeability must be greater than 0.');
        }
        if (!Number.isFinite(inputs.waterTable) || inputs.waterTable <= 0) {
          errors.push('Water table level must be greater than 0.');
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const flowRate = inputs.areaSize * inputs.permeability * inputs.waterTable;
        if (!Number.isFinite(flowRate)) return null;

        const perAreaRate = inputs.areaSize > 0 ? round2(flowRate / inputs.areaSize) : null;
        const permeabilityEffect = inputs.permeability > 0 ? round2(flowRate / inputs.permeability) : null;
        const waterTableEffect = inputs.waterTable > 0 ? round2(flowRate / inputs.waterTable) : null;

        return {
          flowRate: round2(flowRate),
          perAreaRate,
          permeabilityEffect,
          waterTableEffect
        };
      }

      function format(outputs) {
        if (!outputs) return null;
        return {
          flowRate: formatFlow(outputs.flowRate),
          perAreaRate: fmtNumber(outputs.perAreaRate) + ' m³/day/m²',
          permeabilityEffect: fmtNumber(outputs.permeabilityEffect) + ' m³/day/(m/day)',
          waterTableEffect: fmtNumber(outputs.waterTableEffect) + ' m³/day/m',
          confidenceLabel: 'Estimate based on linear model'
        };
      }

      function renderError(errors) {
        if (!errors || errors.length === 0) {
          els.errorBox.style.display = 'none';
          els.errorBox.textContent = '';
          return;
        }
        els.errorBox.style.display = 'block';
        els.errorBox.textContent = errors.join(' ');
      }

      function render(formatted, errors) {
        renderError(errors);
        if (!formatted) {
          els.totalFlowRate.textContent = '—';
          els.perAreaRate.textContent = '—';
          els.permeabilityEffect.textContent = '—';
          els.waterTableEffect.textContent = '—';
          els.confidenceLabel.textContent = '—';
          return;
        }

        els.totalFlowRate.textContent = formatted.flowRate;
        els.perAreaRate.textContent = formatted.perAreaRate;
        els.permeabilityEffect.textContent = formatted.permeabilityEffect;
        els.waterTableEffect.textContent = formatted.waterTableEffect;
        els.confidenceLabel.textContent = formatted.confidenceLabel;
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          render(null, validation.errors);
          return;
        }

        const outputs = compute(inputs);
        if (!outputs) {
          render(null, ['Unable to compute with the current inputs.']);
          return;
        }

        const formatted = format(outputs);
        render(formatted, []);
      }

      function reset() {
        els.areaSize.value = defaults.areaSize;
        els.permeability.value = defaults.permeability;
        els.waterTable.value = defaults.waterTable;
        update();
      }

      const debouncedUpdate = debounce(update, 100);

      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', reset);

      document.querySelectorAll('#inputsCard input').forEach((el) => {
        el.addEventListener('input', debouncedUpdate);
        el.addEventListener('change', debouncedUpdate);
      });

      update();
    })();
  </script>
