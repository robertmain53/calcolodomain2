      document.addEventListener('DOMContentLoaded', function () {
        'use strict';

        const typeDefs = {
          qrcode: { is1D: false, help: 'Enter any text, URL, or data to encode.', label: 'QR Code', validation: /.+/ },
          datamatrix: { is1D: false, help: 'Enter any text or data. Ideal for small industrial parts.', label: 'Data Matrix', validation: /.+/ },

          code128: { is1D: true, help: 'Enter alphanumeric data for logistics and inventory.', label: 'Code 128', validation: /.+/ }
        };
        const DEFAULTS = {
          barcodeType: 'qrcode',
          data: '',
          barWidth: '2',
          barHeight: '100',
          showText: true,
          moduleSize: '5',
          eccLevel: 'M'
        };
        const allowedEcc = new Set(['L', 'M', 'Q', 'H']);

        const els = {
          barcodeType: document.getElementById('barcodeType'),
          data: document.getElementById('data'),
          dataHelp: document.getElementById('data-help'),

          barWidth: document.getElementById('barWidth'),
          barHeight: document.getElementById('barHeight'),
          showText: document.getElementById('showText'),
          moduleSize: document.getElementById('moduleSize'),
          eccLevel: document.getElementById('eccLevel'),
          calcBtn: document.getElementById('calcBtn'),
          resetBtn: document.getElementById('resetBtn'),
          previewLabel: document.getElementById('previewLabel'),
          resultsPlaceholder: document.getElementById('results-placeholder'),
          canvas: document.getElementById('barcode-canvas'),
          downloadContainer: document.getElementById('download-container'),
          downloadPNG: document.getElementById('downloadPNG'),
          generatorError: document.getElementById('generator-error'),
          errorBox: document.getElementById('errorBox')
        };

        function debounce(fn, delay = 100) {
          let timer;
          return function () {
            const context = this;
            const args = arguments;
            clearTimeout(timer);
            timer = setTimeout(() => fn.apply(context, args), delay);
          };
        }

        function sanitizeSegment(value) {
          return (value || 'barcode').replace(/[^A-Za-z0-9_-]/g, '').slice(0, 10) || 'barcode';
        }

        function updateFormState() {
          const selected = typeDefs[els.barcodeType.value];
          if (!selected) return;
          els.options1D.classList.toggle('hidden', !selected.is1D);
          els.options2D.classList.toggle('hidden', selected.is1D);
          els.eccLevelContainer.classList.toggle('hidden', els.barcodeType.value !== 'qrcode');
          els.dataHelp.textContent = selected.help;
          els.data.placeholder = selected.help;
          clearValidation();
        }

        function clearValidation() {
          els.dataError.textContent = '';
          els.dataError.classList.remove('visible');
          els.data.removeAttribute('aria-invalid');
        }

        function showDataError(message) {
          els.dataError.textContent = message;
          els.dataError.classList.add('visible');
          els.data.setAttribute('aria-invalid', 'true');
        }

        function parseInputs() {
          return {
            barcodeType: els.barcodeType.value,
            data: els.data.value.trim(),
            barWidth: Number.parseFloat(els.barWidth.value),
            barHeight: Number.parseFloat(els.barHeight.value),
            showText: els.showText.checked,
            moduleSize: Number.parseFloat(els.moduleSize.value),
            eccLevel: els.eccLevel.value
          };
        }

        function validate(inputs) {
          const errors = [];
          clearValidation();
          const def = typeDefs[inputs.barcodeType];
          if (!def) {
            errors.push('Select a valid barcode type.');
          }
          if (!inputs.data) {
            errors.push('Data cannot be empty.');
            showDataError('Data cannot be empty.');
          } else if (def && def.validation && !def.validation.test(inputs.data)) {
            errors.push('The data format is not valid for the selected barcode.');
            showDataError(def.help);
          }
          if (def && def.is1D) {
            if (!Number.isFinite(inputs.barWidth) || inputs.barWidth <= 0) errors.push('Bar width must be greater than 0.');
            if (!Number.isFinite(inputs.barHeight) || inputs.barHeight <= 0) errors.push('Bar height must be greater than 0.');
          } else {
            if (!Number.isFinite(inputs.moduleSize) || inputs.moduleSize <= 0) errors.push('Module size must be greater than 0.');
            if (inputs.barcodeType === 'qrcode' && !allowedEcc.has(inputs.eccLevel)) errors.push('Error correction level is invalid.');
          }
          return { ok: errors.length === 0, errors };
        }

        function compute(inputs) {
          const def = typeDefs[inputs.barcodeType];
          if (!def) return null;
          const options = {
            bcid: inputs.barcodeType,
            text: inputs.data,
            scale: 3,
            padding: 10
          };
          if (def.is1D) {
            options.width = Math.max(1, inputs.barWidth);
            options.height = Math.max(10, inputs.barHeight);
            options.includetext = inputs.showText;
            options.textxalign = 'center';
          } else {
            options.scale = Math.max(2, inputs.moduleSize);
            if (inputs.barcodeType === 'qrcode') {
              options.eclevel = inputs.eccLevel;
            }
          }
          return { options, typeLabel: def.label || inputs.barcodeType.toUpperCase() };
        }

        function format(outputs, inputs) {
          return {
            options: outputs.options,
            downloadName: `${inputs.barcodeType}-${sanitizeSegment(inputs.data)}.png`,
            previewLabel: `${outputs.typeLabel} preview`
          };
        }

        function render(formatted, errors) {
          const hasErrors = errors && errors.length > 0;
          if (hasErrors) {
            els.errorBox.style.display = 'block';
            els.errorBox.textContent = errors.join(' ');
          } else {
            els.errorBox.style.display = 'none';
            els.errorBox.textContent = '';
          }
          els.generatorError.style.display = 'none';
          els.generatorError.textContent = '';
          if (!formatted || !formatted.options) {
            els.resultsPlaceholder.textContent = hasErrors
              ? 'Fix the highlighted values and regenerate to preview your barcode.'
              : 'Complete the form and click Generate to render a barcode preview.';
            els.resultsPlaceholder.classList.remove('hidden');
            els.canvas.classList.add('hidden');
            els.downloadContainer.classList.add('hidden');
            els.downloadPNG.setAttribute('aria-disabled', 'true');
            return;
          }

          try {
            bwipjs.toCanvas(els.canvas, formatted.options);
            els.previewLabel.textContent = formatted.previewLabel;
            els.resultsPlaceholder.classList.add('hidden');
            els.canvas.classList.remove('hidden');
            els.downloadContainer.classList.remove('hidden');
            els.downloadPNG.href = els.canvas.toDataURL('image/png');
            els.downloadPNG.download = formatted.downloadName;
            els.downloadPNG.setAttribute('aria-disabled', 'false');
          } catch (err) {
            const message = err.message && err.message.includes('Check digit')
              ? 'Invalid check digit. Please check the numeric string.'
              : err.message || 'Unable to render the barcode at this time.';
            els.generatorError.textContent = `Error: ${message}`;
            els.generatorError.style.display = 'block';
            els.resultsPlaceholder.classList.remove('hidden');
            els.canvas.classList.add('hidden');
            els.downloadContainer.classList.add('hidden');
            els.downloadPNG.setAttribute('aria-disabled', 'true');
            console.error(err);
          }
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render(null, validation.errors);
            return;
          }
          const outputs = compute(inputs);
          if (!outputs) {
            render(null, ['Unable to prepare barcode options.']);
            return;
          }
          const formatted = format(outputs, inputs);
          render(formatted, []);
        }

        function resetForm() {
          els.barcodeType.value = DEFAULTS.barcodeType;
          els.data.value = DEFAULTS.data;
          els.barWidth.value = DEFAULTS.barWidth;
          els.barHeight.value = DEFAULTS.barHeight;
          els.showText.checked = DEFAULTS.showText;

          els.eccLevel.value = DEFAULTS.eccLevel;

          resetForm();
          update();
        });

          updateFormState();
          debouncedUpdate();
        });

        resetForm();
        update();
