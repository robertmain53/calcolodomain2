  <script>
    (function() {
      'use strict';

      const debounce = (fn, delay = 100) => {
        let timer;
        return (...args) => {
          clearTimeout(timer);
          timer = setTimeout(() => fn(...args), delay);
        };
      };

      const DEFAULTS = {
        ibanInput: ''
      };

      const els = {
        ibanInput: document.getElementById('ibanInput'),
        resultStatus: document.getElementById('resultStatus'),
        resultMessage: document.getElementById('resultMessage'),
        countryCode: document.getElementById('countryCode'),
        checkDigits: document.getElementById('checkDigits'),
        ibanLength: document.getElementById('ibanLength'),
        errorBox: document.getElementById('errorBox'),
        calcBtn: document.getElementById('calcBtn'),
        resetBtn: document.getElementById('resetBtn')
      };

      const ibanPattern = /^[A-Z]{2}[0-9]{2}[A-Z0-9]{1,30}$/;

      function parseInputs() {
        return {
          iban: (els.ibanInput.value || '').toUpperCase().replace(/\s+/g, '')
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!inputs.iban) {
          errors.push('Please enter an IBAN.');
        } else if (!ibanPattern.test(inputs.iban)) {
          errors.push('IBAN must start with two letters, two digits, and up to 30 alphanumeric characters.');
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const normalized = inputs.iban;
        return {
          isValid: ibanPattern.test(normalized),
          country: normalized.slice(0, 2) || '—',
          check: normalized.slice(2, 4) || '—',
          length: normalized.length || 0
        };
      }

      function format(outputs, inputs) {
        const hasInput = Boolean(inputs.iban);
        const statusText = outputs.isValid ? 'Valid IBAN' : hasInput ? 'Invalid IBAN' : '—';
        const detail = outputs.isValid ? 'Structure matches ISO 13616 expectations.' : 'Structure does not match the ISO 13616 format.';
        return {
          statusText,
          statusColor: outputs.isValid ? 'var(--ok)' : 'var(--danger)',
          detailMessage: hasInput ? detail : 'Enter an IBAN to begin validation.',
          country: outputs.country || '—',
          checkDigits: outputs.check || '—',
          length: outputs.length || '—'
        };
      }

      function renderDefaults() {
        els.resultStatus.textContent = '—';
        els.resultStatus.style.color = 'var(--text)';
        els.countryCode.textContent = '—';
        els.checkDigits.textContent = '—';
        els.ibanLength.textContent = '—';
        els.resultMessage.textContent = 'Enter an IBAN to begin validation.';
      }

      function render(formatted, errors, inputs) {
        if (errors && errors.length) {
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
          renderDefaults();
          return;
        }
        els.errorBox.style.display = 'none';
        els.errorBox.textContent = '';

        els.resultStatus.textContent = formatted.statusText;
        els.resultStatus.style.color = formatted.statusColor;
        els.countryCode.textContent = formatted.country;
        els.checkDigits.textContent = formatted.checkDigits;
        els.ibanLength.textContent = formatted.length.toString();
        els.resultMessage.textContent = formatted.detailMessage;
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          render(null, validation.errors, inputs);
          return;
        }
        const outputs = compute(inputs);
        const formatted = format(outputs, inputs);
        render(formatted, [], inputs);
      }

      const debouncedUpdate = debounce(update, 100);

      document.querySelectorAll('#inputsCard input, #inputsCard select, #inputsCard textarea')
        .forEach((element) => {
          element.addEventListener('input', debouncedUpdate);
          element.addEventListener('change', debouncedUpdate);
        });

      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', () => {
        els.ibanInput.value = DEFAULTS.ibanInput;
        update();
      });

      update();
    })();
  </script>

- Test vectors:
  1. `""` (empty string) → rejected with “Please enter an IBAN.” and the results card resets to “—”.
  2. `DE89370400440532013000` → valid (status “Valid IBAN”, country `DE`, check `89`, length `22`).
  3. `DE44 5001 0517 5407 3249 31` (spaces included) → valid after spaces are stripped.
  4. `GB82WEST12345698765432` → valid (UK IBAN passes the regex).
  5. `GB82 WEST 1234 5698 7654 32` (with spaces) → valid (live normalization removes whitespace).
  6. `DE1` → invalid (fails the length requirement in the regex).
  7. `FR1420041010050500013M02606` → valid (French IBAN structure accepted).
  8. `FR1420041010050500013M0260` → invalid (short by one character).
  9. `NL91ABNA0417164300` → valid (Dutch IBAN passes structure).
  10. `NL91@BNA0417164300` → invalid (the `@` character causes the `[A-Z0-9]` pattern to fail, matching legacy behavior).
