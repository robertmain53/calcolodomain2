        <h2>How to use</h2>
        <p>Divide the total elapsed minutes spent on a batch by the number of units finished to get an average cycle time per unit. Update either field and click Calculate to refresh the results; the inputs also respond to typing thanks to the debounced watcher.</p>
        <h2>Methodology</h2>
        <p>This calculator implements the standard cycle-time equation. Throughput is the reciprocal projected to an hourly cadence, and the pace in seconds per unit helps benchmark against faster processes. All formatting uses rounding to the centisecond so the UI never shows NaN or infinite values.</p>
        <h3>Glossary</h3>
        <ul>
          <li><strong>Total Time:</strong> Duration from start to completion of the observed batch, in minutes.</li>
          <li><strong>Total Units Produced:</strong> Count of finished units during the period.</li>
          <li><strong>Cycle Time:</strong> Average minutes required to produce one unit.</li>
          <li><strong>Throughput:</strong> Average units completed per hour.</li>
          <li><strong>Seconds per Unit:</strong> Cycle time converted to seconds for finer-grain pacing.</li>
        </ul>
        <h3>Example</h3>
        <p>Applying a 120-minute run that yields 500 units yields a cycle time of 0.24 minutes per unit, throughput of 250 units/hour, and 14.4 seconds per unit.</p>
        <div class="formula-box">
          \[
          \text{Cycle Time} = \frac{\text{Total Time}}{\text{Total Units Produced}}
          \]
        </div>
        <h3>Frequently Asked Questions</h3>
        <h4>What is cycle time?</h4>
        <p>Cycle time is the average elapsed time to produce a single unit from the production process.</p>
        <h4>Why is cycle time important?</h4>
        <p>It highlights inefficiencies and bottlenecks so you can target improvements.</p>
        <h4>How can I reduce cycle time?</h4>
        <p>Streamline work steps, reduce waiting, and improve setup practices to shorten cycle time.</p>
        <h4>What factors affect cycle time?</h4>
        <p>Equipment uptime, staffing, material availability, and task complexity impact cycle time.</p>
        <h4>Is cycle time the same as lead time?</h4>
        <p>No, lead time spans the entire customer cycle (order to delivery), while cycle time measures production only.</p>

  <script>
    (function() {
      'use strict';

      const debounce = (fn, delay = 100) => {
        let timeout;
        return (...args) => {
          clearTimeout(timeout);
          timeout = setTimeout(() => fn(...args), delay);
        };
      };

      const round2 = (value) => Math.round((value + Number.EPSILON) * 100) / 100;
      const fmtNumber = (value) => {
        const safe = Number.isFinite(value) ? value : 0;
        return round2(safe).toFixed(2);
      };

      const defaults = {
        totalTime: 120,
        totalUnits: 500
      };

      const els = {
        totalTime: document.getElementById('totalTime'),
        totalUnits: document.getElementById('totalUnits'),
        cycleTimeResult: document.getElementById('cycleTimeResult'),
        throughputResult: document.getElementById('throughputResult'),
        paceResult: document.getElementById('paceResult'),
        errorBox: document.getElementById('errorBox'),
        calcBtn: document.getElementById('calcBtn'),
        resetBtn: document.getElementById('resetBtn')
      };

      function parseInputs() {
        return {
          totalTime: parseFloat(els.totalTime.value) || 0,
          totalUnits: parseFloat(els.totalUnits.value) || 0
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!Number.isFinite(inputs.totalTime) || inputs.totalTime <= 0) {
          errors.push('Total time must be greater than 0.');
        }
        if (!Number.isFinite(inputs.totalUnits) || inputs.totalUnits <= 0) {
          errors.push('Units produced must be greater than 0.');
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const { totalTime, totalUnits } = inputs;
        if (totalUnits <= 0) return null;
        const cycleTime = totalTime / totalUnits;
        const throughputPerHour = cycleTime > 0 ? (60 / cycleTime) : 0;
        const secondsPerUnit = cycleTime * 60;
        return { cycleTime, throughputPerHour, secondsPerUnit };
      }

      function format(outputs) {
        return {
          cycleTime: `${fmtNumber(outputs.cycleTime)} min`,
          throughput: `${fmtNumber(outputs.throughputPerHour)} units/hr`,
          pace: `${fmtNumber(outputs.secondsPerUnit)} sec/unit`
        };
      }

      function render(formatted, errors) {
        if (!errors || errors.length === 0) {
          els.errorBox.style.display = 'none';
          els.errorBox.textContent = '';
        } else {
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
        }

        if (formatted) {
          els.cycleTimeResult.textContent = formatted.cycleTime;
          els.throughputResult.textContent = formatted.throughput;
          els.paceResult.textContent = formatted.pace;
        } else {
          els.cycleTimeResult.textContent = '—';
          els.throughputResult.textContent = '—';
          els.paceResult.textContent = '—';
        }
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          render(null, validation.errors);
          return;
        }
        const outputs = compute(inputs);
        if (!outputs) {
          render(null, ['Calculation failed.']);
          return;
        }
        render(format(outputs), []);
      }

      const debouncedUpdate = debounce(update, 100);

      els.calcBtn.addEventListener('click', update);
      els.resetBtn.addEventListener('click', () => {
        els.totalTime.value = defaults.totalTime;
        els.totalUnits.value = defaults.totalUnits;
        update();
      });

      document.querySelectorAll('#inputsCard input').forEach((element) => {
        element.addEventListener('input', debouncedUpdate);
        element.addEventListener('change', debouncedUpdate);
      });

      update();
    })();
  </script>
