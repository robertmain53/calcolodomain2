        <p class="text-gray-600">We sort the numeric entries, then split the list into halves to compute Q1 (median of the lower half) and Q3 (median of the upper half). The median follows the same median rule and the min/max are the first and last sorted elements.</p>
        <ul class="list-disc pl-5 mt-3 text-gray-700">
          <li>The interquartile range (IQR) equals Q3 minus Q1 and highlights the middle 50% of the dataset.</li>

          <li>Sorted data appears below when you toggle the data table, helping verify inputs and potential outliers.</li>

              <p><strong>What is a box and whisker plot?</strong> A box and whisker plot is a graphical representation of data that shows the distribution and identifies outliers.</p>
              <p><strong>How do I interpret a box plot?</strong> The box represents the interquartile range and the line inside shows the median; whiskers display spread beyond Q1 and Q3.</p>
              <p><strong>What do the whiskers represent?</strong> They indicate variability outside the quartiles, showing the overall range of the dataset.</p>

        }

        function validate(inputs) {
          const errors = [];
          const invalid = inputs.entries.filter(entry => Number.isNaN(entry.value));
          const numbers = inputs.entries
            .filter(entry => Number.isFinite(entry.value))
            .map(entry => entry.value);
          if (!inputs.raw) {
            errors.push('Please enter at least one number.');
          }
          if (invalid.length) {
            errors.push('Remove the non-numeric entries: ' + invalid.map(entry => entry.token).join(', ') + '.');
          }
          if (!numbers.length) {
            errors.push('Provide at least one valid numeric data point.');
          }
          return { ok: errors.length === 0, errors, numbers };
        }

        function getMedian(values) {
          const len = values.length;
          if (!len) {
            return 0;
          }
          const half = Math.floor(len / 2);
          if (len % 2 === 0) {
            return (values[half - 1] + values[half]) / 2;
          }
          return values[half];
        }

        function compute(numbers) {
          const sorted = [...numbers].sort((a, b) => a - b);
          const min = sorted[0];
          const max = sorted[sorted.length - 1];
          const median = getMedian(sorted);
          const halfway = Math.floor(sorted.length / 2);
          const lower = sorted.slice(0, halfway);
          const upper = sorted.length % 2 === 0 ? sorted.slice(halfway) : sorted.slice(halfway + 1);
          const q1 = getMedian(lower);
          const q3 = getMedian(upper);
          const iqr = q3 - q1;
          return { sorted, min, q1, median, q3, max, iqr };
        }

        function formatResults(outputs) {
          const safe = value => (Number.isFinite(value) ? formatter.format(value) : '—');
          return {
            min: safe(outputs.min),
            q1: safe(outputs.q1),
            median: safe(outputs.median),
            q3: safe(outputs.q3),
            max: safe(outputs.max),
            iqr: safe(outputs.iqr)
          };
        }

        function render(formatted, outputs, errors) {
          if (errors && errors.length) {
            errorBox.textContent = errors.join(' ');
            errorBox.style.display = 'block';
            distributionSummary.textContent = 'Correct the data to see updated results.';
            downloadCsvBtn.disabled = true;
            latestOutputs = null;
            scheduleBody.innerHTML = '';
            resultFields.min.textContent = '—';
            resultFields.q1.textContent = '—';
            resultFields.median.textContent = '—';
            resultFields.q3.textContent = '—';
            resultFields.max.textContent = '—';
            resultFields.iqr.textContent = '—';
            resultFields.iqrLabel.textContent = 'Interquartile range (Q3 − Q1)';
            return;
          }
          errorBox.style.display = 'none';
          errorBox.textContent = '';
          downloadCsvBtn.disabled = false;
          latestOutputs = outputs;
          resultFields.min.textContent = formatted.min;
          resultFields.q1.textContent = formatted.q1;
          resultFields.median.textContent = formatted.median;
          resultFields.q3.textContent = formatted.q3;
          resultFields.max.textContent = formatted.max;
          resultFields.iqr.textContent = formatted.iqr;
          resultFields.iqrLabel.textContent = 'Interquartile range (Q3 − Q1)';
          distributionSummary.textContent = `Interquartile range is ${formatted.iqr} across ${outputs.sorted.length} value${outputs.sorted.length === 1 ? '' : 's'}.`;
          scheduleBody.innerHTML = outputs.sorted
            .map((value, index) => `<tr><td>${index + 1}</td><td>${formatter.format(value)}</td></tr>`)
            .join('');
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render({}, {}, validation.errors);
            return;
          }
          const outputs = compute(validation.numbers);
          const formatted = formatResults(outputs);
          render(formatted, outputs, []);
        }

        function scheduleToggle() {
          scheduleVisible = !scheduleVisible;
          scheduleWrap.style.display = scheduleVisible ? 'block' : 'none';
          toggleScheduleBtn.textContent = scheduleVisible ? 'Hide data table' : 'View data table';
        }

        function resetState() {
          dataInput.value = DEFAULT_INPUT;
          scheduleVisible = false;
          scheduleWrap.style.display = 'none';
          toggleScheduleBtn.textContent = 'View data table';
          update();
        }

        function downloadCsv() {
          if (!latestOutputs) {
            return;
          }
          const rows = [
            ['Statistic', 'Value'],
            ['Min', formatter.format(latestOutputs.min)],
            ['Q1', formatter.format(latestOutputs.q1)],
            ['Median', formatter.format(latestOutputs.median)],
            ['Q3', formatter.format(latestOutputs.q3)],
            ['Max', formatter.format(latestOutputs.max)],
            ['IQR', formatter.format(latestOutputs.iqr)],
            ['', ''],
            ['Sorted Data', '']
          ];
          latestOutputs.sorted.forEach((value, index) => {
            rows.push([`Value ${index + 1}`, formatter.format(value)]);
          });
          const csvContent = rows
            .map(row => row.map(cell => `"${cell}"`).join(','))
            .join('\n');
          const blob = new Blob([csvContent], { type: 'text/csv' });
          const url = URL.createObjectURL(blob);
          const link = document.createElement('a');
          link.href = url;
          link.download = 'box-and-whisker-data.csv';
          document.body.appendChild(link);
          link.click();
          document.body.removeChild(link);
          URL.revokeObjectURL(url);
        }

        function handleDebounce() {
          clearTimeout(debounceTimer);
          debounceTimer = setTimeout(update, 100);
        }

        calcBtn.addEventListener('click', update);
        resetBtn.addEventListener('click', resetState);
        dataInput.addEventListener('input', handleDebounce);
        dataInput.addEventListener('change', handleDebounce);
        toggleScheduleBtn.addEventListener('click', scheduleToggle);
        downloadCsvBtn.addEventListener('click', downloadCsv);

        resetState();
      })();
    </script>
</body></html>
