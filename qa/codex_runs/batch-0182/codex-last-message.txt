<script>
(function() {
  'use strict';

  const round1 = (value) => Math.round((value + Number.EPSILON) * 10) / 10;
  const defaults = {
    weight: '75',
    calories: '2500',
    protein: '30',
    fat: '25'
  };

  const els = {
    weight: document.getElementById('weight'),
    calories: document.getElementById('calories'),
    protein: document.getElementById('protein'),
    fat: document.getElementById('fat'),
    caloriesResult: document.getElementById('caloriesResult'),
    proteinResult: document.getElementById('proteinResult'),
    fatResult: document.getElementById('fatResult'),
    carbsResult: document.getElementById('carbsResult'),
    errorBox: document.getElementById('errorBox'),
    calcBtn: document.getElementById('calcBtn'),
    resetBtn: document.getElementById('resetBtn')
  };

  function parseInputs() {
    return {
      weight: Number(els.weight.value) || 0,
      calories: Number(els.calories.value) || 0,
      protein: Number(els.protein.value) || 0,
      fat: Number(els.fat.value) || 0
    };
  }

  function validate(inputs) {
    const errors = [];
    if (!Number.isFinite(inputs.weight) || inputs.weight <= 0) {
      errors.push('Weight must be greater than 0.');
    }
    if (!Number.isFinite(inputs.calories) || inputs.calories <= 0) {
      errors.push('Calories must be greater than 0.');
    }
    if (!Number.isFinite(inputs.protein) || inputs.protein < 0 || inputs.protein > 100) {
      errors.push('Protein percentage must be between 0 and 100.');
    }
    if (!Number.isFinite(inputs.fat) || inputs.fat < 0 || inputs.fat > 100) {
      errors.push('Fat percentage must be between 0 and 100.');
    }
    if (inputs.protein + inputs.fat >= 100) {
      errors.push('Protein and fat percentages must total less than 100.');
    }
    return { ok: errors.length === 0, errors };
  }

  function compute(inputs) {
    const proteinCals = inputs.calories * (inputs.protein / 100);
    const fatCals = inputs.calories * (inputs.fat / 100);
    const carbsCals = inputs.calories - (proteinCals + fatCals);
    if (!Number.isFinite(proteinCals) || !Number.isFinite(fatCals) || !Number.isFinite(carbsCals)) {
      return null;
    }
    const proteinGrams = proteinCals / 4;
    const fatGrams = fatCals / 9;
    const carbsGrams = Math.max(0, carbsCals / 4);
    return {
      calories: inputs.calories,
      protein: Math.max(0, proteinGrams),
      fat: Math.max(0, fatGrams),
      carbs: Math.max(0, carbsGrams)
    };
  }

  function format(outputs) {
    return {
      calories: `${round1(outputs.calories).toFixed(0)} kcal`,
      protein: `${round1(outputs.protein).toFixed(1)} g`,
      fat: `${round1(outputs.fat).toFixed(1)} g`,
      carbs: `${round1(outputs.carbs).toFixed(1)} g`
    };
  }

  function render(formatted, errors) {
    if (errors && errors.length) {
      els.errorBox.textContent = errors.join(' ');
      els.errorBox.style.display = 'block';
      els.caloriesResult.textContent = '—';
      els.proteinResult.textContent = '—';
      els.fatResult.textContent = '—';
      els.carbsResult.textContent = '—';
      return;
    }
    els.errorBox.style.display = 'none';
    els.errorBox.textContent = '';
    if (formatted) {
      els.caloriesResult.textContent = formatted.calories;
      els.proteinResult.textContent = formatted.protein;
      els.fatResult.textContent = formatted.fat;
      els.carbsResult.textContent = formatted.carbs;
    }
  }

  function update() {
    const inputs = parseInputs();
    const validation = validate(inputs);
    if (!validation.ok) {
      render(null, validation.errors);
      return;
    }
    const outputs = compute(inputs);
    if (!outputs) {
      render(null, ['Unable to compute results for the provided inputs.']);
      return;
    }
    render(format(outputs), []);
  }

  function reset() {
    els.weight.value = defaults.weight;
    els.calories.value = defaults.calories;
    els.protein.value = defaults.protein;
    els.fat.value = defaults.fat;
    update();
  }

  const debounce = (fn, delay = 100) => {
    let timeout;
    return (...args) => {
      clearTimeout(timeout);
      timeout = setTimeout(() => fn(...args), delay);
    };
  };

  const debouncedUpdate = debounce(update, 100);
  document.querySelectorAll('#inputsCard input')
    .forEach((el) => {
      el.addEventListener('input', debouncedUpdate);
      el.addEventListener('change', debouncedUpdate);
    });

  els.calcBtn.addEventListener('click', update);
  els.resetBtn.addEventListener('click', reset);

  reset();
})();
