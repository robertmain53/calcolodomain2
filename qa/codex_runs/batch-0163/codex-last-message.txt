          <section class="card content-card space-y-6" id="howToUse" aria-label="How to use and methodology">
            <h2 class="text-2xl font-semibold">How to Use This Calculator</h2>
            <p>Choose the current cervical dilation, effacement, fetal station, consistency, and position. The tool adds the numeric score assigned to each domain to produce the Bishop Score. Higher totals point toward a more favorable cervix for induction.</p>
            <p class="text-sm text-gray-500">Click Calculate after you adjust any field, or simply wait for the automatic recompute triggered by the select controls.</p>

            <h2 class="text-2xl font-semibold mt-6">Methodology</h2>
            <div class="bg-gray-50 border border-gray-200 rounded-lg p-4">
              <p>The calculator mirrors the clinical Bishop scoring system. Each cervical observation carries a weighted value (0–3 for dilation, 0–3 for effacement, 0–3 for station, and 0–2 each for consistency and position). We sum the values with no smoothing to deliver the same practical result obstetric teams use when gauging induction readiness.</p>
            </div>
            <ul class="list-disc pl-5 mt-4 space-y-2 text-gray-700">
              <li>Scores below 6 usually mean the cervix is unfavorable for induction.</li>
              <li>Scores between 6 and 8 sit in a gray zone that may benefit from cervical ripening.</li>
              <li>A score of 9 or more is considered favorable and usually predicts easier induction.</li>
            </ul>
            <div class="mt-4 bg-gray-50 border border-gray-200 rounded-lg p-4 text-sm text-gray-700">
              Results are estimates for reference only. Clinical judgment, ultrasound findings, and provider protocols should guide actual labor and delivery decisions.
            </div>

            <section>
              <h3 class="text-xl font-semibold">Glossary</h3>
              <ul class="list-disc pl-5 mt-3 space-y-2 text-gray-700">
                <li><strong>Dilation:</strong> Diameter of the cervical opening measured in centimeters.</li>
                <li><strong>Effacement:</strong> Percentage of cervical thinning relative to the normal 100% state.</li>
                <li><strong>Station:</strong> Fetal head position relative to the maternal pelvic ischial spines.</li>
                <li><strong>Consistency:</strong> Firmness of the cervix (firm, medium, soft).</li>
                <li><strong>Position:</strong> Cervical alignment toward the pelvic axis (posterior, mid, anterior).</li>
              </ul>
            </section>

            <section>
              <h3 class="text-xl font-semibold mt-6">Frequently Asked Questions</h3>
              <div class="faq">
                <h3>What is the Bishop Score?</h3>
                <p>The Bishop Score is a system used to assess the readiness of the cervix for labor.</p>
                <h3>Why is the Bishop Score important?</h3>
                <p>It helps in determining the likelihood of a successful induction and the readiness for labor.</p>
                <h3>How is the Bishop Score calculated?</h3>
                <p>It is calculated by evaluating cervical dilation, effacement, fetal station, consistency, and position.</p>
                <h3>What does a high Bishop Score indicate?</h3>
                <p>A high Bishop Score suggests that the cervix is more favorable for labor induction.</p>
                <h3>Can the Bishop Score predict labor?</h3>
                <p>While it provides valuable information, it should not be treated as an absolute predictor of labor onset.</p>
              </div>

      (function () {
        'use strict';

        const debounce = (fn, delay = 100) => {
          let timeout;
          return function (...args) {
            clearTimeout(timeout);
            timeout = setTimeout(() => fn(...args), delay);
          };
        };

        const classification = (score) => {
          if (score >= 9) {
            return { label: 'Favorable for induction', summary: 'Favorable' };
          }
          if (score >= 6) {
            return { label: 'Borderline (gray zone)', summary: 'Borderline' };
          }
          return { label: 'Unfavorable for induction readiness', summary: 'Unfavorable' };
        };

        const els = {
          dilation: document.getElementById('dilation'),
          effacement: document.getElementById('effacement'),
          station: document.getElementById('station'),
          consistency: document.getElementById('consistency'),
          position: document.getElementById('position'),
          calculateBtn: document.getElementById('calcBtn'),
          resetBtn: document.getElementById('resetBtn'),
          scoreDisplay: document.getElementById('bishopScore'),
          scoreInterpretation: document.getElementById('scoreInterpretation'),
          scoreInterpretationSummary: document.getElementById('scoreInterpretationSummary'),
          scoreDetails: document.getElementById('scoreDetails'),
          errorBox: document.getElementById('errorBox')
        };

        const allowed = {
          dilation: new Set([0, 1, 2, 3, 4, 5, 6]),
          effacement: new Set([0, 1, 2, 3, 4, 5]),
          station: new Set([0, 1, 2, 3, 4, 5]),
          consistency: new Set([0, 1, 2]),
          position: new Set([0, 1, 2])
        };

        const defaults = {
          dilation: 3,
          effacement: 3,
          station: 3,
          consistency: 2,
          position: 2
        };

        function parseInputs() {
          return {
            dilation: Number.parseInt(els.dilation.value, 10),
            effacement: Number.parseInt(els.effacement.value, 10),
            station: Number.parseInt(els.station.value, 10),
            consistency: Number.parseInt(els.consistency.value, 10),
            position: Number.parseInt(els.position.value, 10)
          };
        }

        function validate(inputs) {
          const errors = [];
          if (!allowed.dilation.has(inputs.dilation)) errors.push('Select a valid dilation value.');
          if (!allowed.effacement.has(inputs.effacement)) errors.push('Select a valid effacement value.');
          if (!allowed.station.has(inputs.station)) errors.push('Select a valid station value.');
          if (!allowed.consistency.has(inputs.consistency)) errors.push('Select a valid consistency value.');
          if (!allowed.position.has(inputs.position)) errors.push('Select a valid position value.');
          return { ok: errors.length === 0, errors };
        }

        function compute(inputs) {
          const total = inputs.dilation + inputs.effacement + inputs.station + inputs.consistency + inputs.position;
          return { total, breakdown: inputs };
        }

        function format(outputs) {
          const verdict = classification(outputs.total);
          const detailParts = [
            `Dilation ${outputs.breakdown.dilation}`,
            `Effacement ${outputs.breakdown.effacement}`,
            `Station ${outputs.breakdown.station}`,
            `Consistency ${outputs.breakdown.consistency}`,
            `Position ${outputs.breakdown.position}`
          ];
          return {
            scoreText: outputs.total.toString(),
            interpretation: verdict.label,
            interpretationSummary: verdict.summary,
            detailText: detailParts.join(', ')
          };
        }

        function renderError(errors) {
          if (!errors || errors.length === 0) {
            els.errorBox.style.display = 'none';
            els.errorBox.textContent = '';
            return;
          }
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
        }

        function render(formatted, errors) {
          renderError(errors);
          if (!formatted) {
            els.scoreDisplay.textContent = '—';
            els.scoreInterpretation.textContent = 'Awaiting valid inputs.';
            els.scoreInterpretationSummary.textContent = '—';
            els.scoreDetails.textContent = 'Choose values to see the breakdown.';
            return;
          }
          els.scoreDisplay.textContent = formatted.scoreText;
          els.scoreInterpretation.textContent = formatted.interpretation;
          els.scoreInterpretationSummary.textContent = formatted.interpretationSummary;
          els.scoreDetails.textContent = formatted.detailText;
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render(null, validation.errors);
            return;
          }
          const outputs = compute(inputs);
          const formatted = format(outputs);
          render(formatted, []);
        }

        function resetToDefaults() {
          Object.entries(defaults).forEach(([key, value]) => {
            if (els[key]) {
              els[key].value = String(value);
            }
          });
          update();
        }

        const debouncedUpdate = debounce(update, 100);
        ['change', 'input'].forEach((event) => {
          Object.keys(allowed).forEach((key) => {
            const el = els[key];
            if (el) el.addEventListener(event, debouncedUpdate);
          });
        });

        els.calculateBtn.addEventListener('click', update);
        els.resetBtn.addEventListener('click', resetToDefaults);

        resetToDefaults();
      })();
