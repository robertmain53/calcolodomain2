        <h2 class="text-2xl font-semibold">How to Use This Calculator</h2>
        <p>Type your existing bra size (band followed by cup), select the standards you are translating from and to, and then press Convert. The converter locks the band to a reference metric, keeps the cup sequence intact, and explains any adjustments.</p>

        <div class="bg-gray-50 border border-gray-200 rounded-lg">
          <h3 class="text-xl font-semibold">Methodology</h3>
          <p class="mt-2 text-sm text-gray-700">We map US/UK band sizes to their European centimeter equivalents (28 → 60 cm, 30 → 65 cm, etc.) and use a shared cup order to preserve relative volume. When a target standard lacks a cup label, we clamp to the closest available option and note the adjustment.</p>
        </div>

        <section>
          <h3 class="text-lg font-semibold">Data Source &amp; Methodology</h3>
          <p>All calculations rely on the International Bra Size Association's published conversions. See <a href="https://www.brasize.org" class="text-blue-600 hover:underline" target="_blank" rel="noopener">brasize.org</a> for the raw tables that inspired this converter.</p>
        </section>

        <section>
          <h3 class="text-lg font-semibold">The Formula Explained</h3>
          <p>Band anchors are even-numbered to align with brand conventions. Each cup is assigned a position in a master sequence (AAA → K). The converter transfers the cup index directly, then selects the matching label from the destination standard.</p>
        </section>

        <section>
          <h3 class="text-lg font-semibold">Glossary of Terms</h3>
          <ul class="text-gray-700 space-y-2">
            <li><strong>Current Size:</strong> The band + cup you enter, interpreted per the selected source standard.</li>
            <li><strong>Converted Size:</strong> The closest equivalent in the target standard after mapping the band anchor and cup order.</li>
            <li><strong>Anchor Band:</strong> The shared metric (cm) we use to translate numeric band widths.</li>
            <li><strong>Cup Shift:</strong> Notes when the target standard lacks certain cup labels so you know the adjustment.</li>
          </ul>
        </section>

        <section>
          <h3 class="text-lg font-semibold">How It Works: A Step-by-Step Example</h3>
          <p>Enter 34B as US, choose EU on the right, and click Convert. The converter anchors band 34 to 75 cm and keeps the cup at the same place in the master sequence, delivering EU 75B as the equivalent.</p>

          EU: ['AA','A','B','C','D','E','F','G','H','I','J','K']
        };

        const cupAliases = {
          AAA: 'AAA', AA: 'AA', A: 'A', B: 'B', C: 'C', D: 'D', DD: 'DD', DDD: 'DDD',
          E: 'E', EE: 'E', F: 'F', FF: 'FF', G: 'G', GG: 'GG', H: 'H', HH: 'HH', I: 'I', J: 'J', K: 'K'
        };

        const debounce = (fn, delay = 100) => {
          let timer;
          return (...args) => {
            clearTimeout(timer);
            timer = setTimeout(() => fn(...args), delay);
          };
        };

        const els = {
          currentSize: document.getElementById('currentSize'),
          fromStandard: document.getElementById('fromStandard'),
          toStandard: document.getElementById('toStandard'),
          convertedSize: document.getElementById('convertedSize'),
          convertedNarrative: document.getElementById('convertedNarrative'),
          bandAnchorValue: document.getElementById('bandAnchorValue'),
          cupShiftValue: document.getElementById('cupShiftValue'),
          originalSizeValue: document.getElementById('originalSizeValue'),
          targetStandardValue: document.getElementById('targetStandardValue'),
          errorBox: document.getElementById('errorBox'),
          calcBtn: document.getElementById('calcBtn'),
          resetBtn: document.getElementById('resetBtn')
        };

        const defaults = {
          currentSize: '34B',
          fromStandard: 'US',
          toStandard: 'EU'
        };

        function clampNearest(value, options) {
          if (!Number.isFinite(value)) return NaN;
          let closest = options[0];
          for (const option of options) {
            if (Math.abs(option - value) < Math.abs(closest - value)) {
              closest = option;
            }
          }
          return closest;
        }

        function sanitizeBand(standard, band) {
          const available = standard === 'EU' ? bandPairs.map(pair => pair.eu) : bandPairs.map(pair => pair.us);
          return clampNearest(band, available);
        }

        function measurementForBand(standard, band) {
          const sanitized = sanitizeBand(standard, band);
          if (!Number.isFinite(sanitized)) return NaN;
          if (standard === 'EU') return sanitized;
          const mapping = bandPairs.find(pair => pair.us === sanitized);
          return mapping ? mapping.eu : sanitized;
        }

        function bandForMeasurement(standard, measurement) {
          if (!Number.isFinite(measurement)) return NaN;
          if (standard === 'EU') return sanitizeBand('EU', measurement);
          let closest = bandPairs[0];
          for (const pair of bandPairs) {
            if (Math.abs(pair.eu - measurement) < Math.abs(closest.eu - measurement)) {
              closest = pair;
            }
          }
          return closest.us;
        }

        function canonicalCup(raw) {
          if (!raw) return '';
          const cleaned = raw.toUpperCase().replace(/[^A-Z]/g, '');
          return cupAliases[cleaned] || '';
        }

        function cupIndexForLabel(label) {
          return cupMasterOrder.indexOf(label);
        }

        function cupLabelForStandard(index, standard) {
          const list = cupOrders[standard] || cupMasterOrder;
          const bounded = Math.min(Math.max(0, index), list.length - 1);
          return list[bounded];
        }

        function parseInputs() {
          const raw = els.currentSize.value || '';
          const sanitized = raw.trim().toUpperCase().replace(/[^0-9A-Z]/g, '');
          const match = sanitized.match(/^(\d{2,3})([A-Z]{1,3})$/);
          return {
            raw,
            standardFrom: els.fromStandard.value,
            standardTo: els.toStandard.value,
            band: match ? Number(match[1]) : NaN,
            cup: match ? match[2] : ''
          };
        }

        function validate(inputs) {
          const errors = [];
          const canonicalCupLabel = canonicalCup(inputs.cup);
          const allowedStandards = new Set(['US', 'UK', 'EU']);
          if (!allowedStandards.has(inputs.standardFrom) || !allowedStandards.has(inputs.standardTo)) {
            errors.push('Select both source and target standards.');
          }
          if (!inputs.raw) {
            errors.push('Enter a bra size (e.g., 34B).');
          }
          if (!Number.isFinite(inputs.band)) {
            errors.push('Band size must include two or three digits.');
          }
          if (!canonicalCupLabel) {
            errors.push('Provide a recognizable cup letter.');
          }
          inputs.canonicalCupLabel = canonicalCupLabel;
          return { ok: errors.length === 0, errors };
        }

        function renderError(errors) {
          if (!errors || errors.length === 0) {
            els.errorBox.style.display = 'none';
            els.errorBox.textContent = '';
            return;
          }
          els.errorBox.style.display = 'block';
          els.errorBox.textContent = errors.join(' ');
        }

        function compute(inputs) {
          const measurement = measurementForBand(inputs.standardFrom, inputs.band);
          if (!Number.isFinite(measurement)) return null;
          const targetBand = bandForMeasurement(inputs.standardTo, measurement);
          const masterIndex = cupIndexForLabel(inputs.canonicalCupLabel);
          if (masterIndex < 0) return null;
          const targetList = cupOrders[inputs.standardTo] || cupMasterOrder;
          const targetCupLabel = targetList.includes(inputs.canonicalCupLabel)
            ? inputs.canonicalCupLabel
            : cupLabelForStandard(masterIndex, inputs.standardTo);
          const clamped = targetCupLabel !== inputs.canonicalCupLabel;
          return {
            measurement,
            targetBand,
            targetCupLabel,
            masterIndex,
            targetCupIndex: targetList.indexOf(targetCupLabel),
            clamped,
            canonicalCupLabel: inputs.canonicalCupLabel
          };
        }

        function format(result, inputs) {
          const bandAnchor = sanitizeBand(inputs.standardFrom, inputs.band);
          const convertedLabel = `${inputs.standardTo} ${result.targetBand}${result.targetCupLabel}`;
          const measurementText = `${Math.round((result.measurement + Number.EPSILON) * 100) / 100} cm anchor`;
          const cupShiftText = result.clamped ? 'Adjusted to the closest available cup' : 'Matches original cup level';
          const narrative = result.clamped
            ? `Target standard lacks ${result.canonicalCupLabel}; using ${result.targetCupLabel} as the nearest option.`
            : 'Cup sequence matches the target standard list.';
          return {
            convertedLabel,
            narrative,
            measurementText,
            cupShiftText,
            originalLabel: `${bandAnchor}${result.canonicalCupLabel}`,
            targetStandardLabel: convertedLabel
          };
        }

        function render(formatted, errors) {
          renderError(errors);
          const placeholder = '—';
          if (!formatted) {
            els.convertedSize.textContent = placeholder;
            els.convertedNarrative.textContent = 'Enter a size and hit Convert to see the equivalent.';
            els.bandAnchorValue.textContent = placeholder;
            els.cupShiftValue.textContent = placeholder;
            els.originalSizeValue.textContent = placeholder;
            els.targetStandardValue.textContent = placeholder;
            return;
          }
          els.convertedSize.textContent = formatted.convertedLabel;
          els.convertedNarrative.textContent = formatted.narrative;
          els.bandAnchorValue.textContent = formatted.measurementText;
          els.cupShiftValue.textContent = formatted.cupShiftText;
          els.originalSizeValue.textContent = formatted.originalLabel;
          els.targetStandardValue.textContent = formatted.targetStandardLabel;
        }

        function update() {
          const inputs = parseInputs();
          const validation = validate(inputs);
          if (!validation.ok) {
            render(null, validation.errors);
            return;
          }
          const result = compute(inputs);
          if (!result) {
            render(null, ['Unable to map that size.']);
            return;
          }
          const formatted = format(result, inputs);
          render(formatted, []);
        }

        const debouncedUpdate = debounce(update, 100);
        [els.currentSize, els.fromStandard, els.toStandard].forEach((el) => {
          el.addEventListener('input', debouncedUpdate);
          el.addEventListener('change', debouncedUpdate);
        });

        els.calcBtn.addEventListener('click', update);
        els.resetBtn.addEventListener('click', () => {
          els.currentSize.value = defaults.currentSize;
          els.fromStandard.value = defaults.fromStandard;
          els.toStandard.value = defaults.toStandard;
          update();
        });

        update();
      })();
    </script>
