              <div>
                <p><strong>Core tension equation:</strong></p>
                <div class="formula-box">\[ T = \frac{F}{2 \pi r} \]</div>
                <p class="text-gray-600">T is belt tension, F is the tangential force delivered by the driver, and r is the pulley radius (half the diameter). This calculator echoes the requested tension so you can validate the assumed operating load.</p>

          tension: parseFloatSafe(elements.tension.value)
        };
      }

      function parseFloatSafe(value) {
        const parsed = parseFloat(value);
        return Number.isFinite(parsed) ? parsed : 0;
      }

      function validate(inputs) {
        const errors = [];
        if (inputs.pulleyDiameter <= 0) {
          errors.push('Pulley diameter must be greater than zero.');
        }
        if (inputs.beltLength <= 0) {
          errors.push('Belt length must be greater than zero.');
        }
        if (inputs.tension <= 0) {
          errors.push('Tension must be a positive value.');
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const tension = inputs.tension;
        const beltLength = inputs.beltLength;
        const segmentLength = beltLength / scheduleRowsCount;
        const schedule = Array.from({ length: scheduleRowsCount }, (_, index) => {
          return {
            segment: index + 1,
            length: segmentLength,
            tension: tension,
            notes: index === 0 ? 'Start node' : index === scheduleRowsCount - 1 ? 'End node' : 'Evenly distributed'
          };
        });

        return {
          tension,
          beltLength,
          pulleyDiameter: inputs.pulleyDiameter,
          segmentCount: schedule.length,
          schedule
        };
      }

      function formatRawInput(element, fallback) {
        const trimmed = String(element.value).trim();
        return trimmed.length ? trimmed : fallback;
      }

      function format(outputs) {
        const tensionLabel = formatRawInput(elements.tension, formatNumber(outputs.tension, 1));
        return {
          tension: `${tensionLabel} N`,
          beltLength: `${formatNumber(outputs.beltLength, 1)} mm`,
          pulleyDiameter: `${formatNumber(outputs.pulleyDiameter, 1)} mm`,
          inputTension: `${tensionLabel} N`,
          segmentCount: `${outputs.segmentCount} segments`,
          scheduleRows: outputs.schedule.map(row => ({
            segment: row.segment,
            length: formatNumber(row.length, 1),
            tension: formatNumber(row.tension, 1),
            notes: row.notes
          }))
        };
      }

      function formatNumber(value, decimals = 2) {
        const options = { minimumFractionDigits: decimals, maximumFractionDigits: decimals };
        if (!Number.isFinite(value)) {
          return '0';
        }
        return value.toLocaleString('en-US', options);
      }

      function render(formatted, validation) {
        if (!validation.ok) {
          elements.errorBox.style.display = 'block';
          elements.errorBox.textContent = validation.errors.join(' ');
          pushPlaceholders();
          renderSchedule([]);
          elements.downloadCsv.disabled = true;
          state.valid = false;
          return;
        }

        elements.errorBox.style.display = 'none';
        elements.errorBox.textContent = '';
        elements.totalMonthly.textContent = formatted.tension;
        elements.piMonthly.textContent = formatted.beltLength;
        elements.tiMonthly.textContent = formatted.pulleyDiameter;
        elements.totalInterest.textContent = formatted.inputTension;
        elements.payoffTime.textContent = formatted.segmentCount;
        state.lastSchedule = formatted.scheduleRows;
        renderSchedule(formatted.scheduleRows);
        elements.downloadCsv.disabled = false;
        state.valid = true;
      }

      function pushPlaceholders() {
        elements.totalMonthly.textContent = '—';
        elements.piMonthly.textContent = '—';
        elements.tiMonthly.textContent = '—';
        elements.totalInterest.textContent = '—';
        elements.payoffTime.textContent = '—';
      }

      function renderSchedule(rows) {
        elements.scheduleBody.innerHTML = '';
        if (!rows.length) {
          const row = document.createElement('tr');
          row.innerHTML = '<td colspan="4" class="text-center text-gray-500">Provide valid inputs to populate the tension segments.</td>';
          elements.scheduleBody.appendChild(row);
          return;
        }
        rows.forEach(entry => {
          const row = document.createElement('tr');
          row.innerHTML = `<td>${entry.segment}</td><td>${entry.length}</td><td>${entry.tension}</td><td>${entry.notes}</td>`;
          elements.scheduleBody.appendChild(row);
        });
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          render(null, validation);
          return;
        }
        const outputs = compute(inputs);
        const formatted = format(outputs);
        render(formatted, validation);
      }

      function reset() {
        elements.pulleyDiameter.value = defaults.pulleyDiameter;
        elements.beltLength.value = defaults.beltLength;
        elements.tension.value = defaults.tension;
        state.scheduleVisible = false;
        elements.scheduleWrap.style.display = 'none';
        elements.toggleSchedule.textContent = 'View tension breakdown';
        update();
      }

      let debounce;
      function scheduleDebouncedUpdate() {
        clearTimeout(debounce);
        debounce = setTimeout(update, 100);
      }

      function toggleScheduleHandler() {
        state.scheduleVisible = !state.scheduleVisible;
        if (state.scheduleVisible) {
          elements.scheduleWrap.style.display = 'block';
          elements.toggleSchedule.textContent = 'Hide tension breakdown';
        } else {
          elements.scheduleWrap.style.display = 'none';
          elements.toggleSchedule.textContent = 'View tension breakdown';
        }
      }

      function downloadCsvHandler() {
        if (!state.valid || !state.lastSchedule.length) {
          return;
        }
        const header = ['Segment', 'Length (mm)', 'Tension (N)', 'Notes'];
        const rows = [header.join(',')];
        state.lastSchedule.forEach(row => {
          rows.push([row.segment, row.length, row.tension, row.notes].join(','));
        });
        const csv = rows.join('\\n');
        const blob = new Blob([csv], { type: 'text/csv' });
        const url = URL.createObjectURL(blob);
        const link = document.createElement('a');
        link.href = url;
        link.download = 'belt-drive-schedule.csv';
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        URL.revokeObjectURL(url);
      }

      function init() {
        elements.scheduleWrap.style.display = 'none';
        elements.toggleSchedule.addEventListener('click', toggleScheduleHandler);
        elements.downloadCsv.addEventListener('click', downloadCsvHandler);
        elements.calcBtn.addEventListener('click', update);
        elements.resetBtn.addEventListener('click', reset);
        [elements.pulleyDiameter, elements.beltLength, elements.tension].forEach(input => {
          input.addEventListener('input', scheduleDebouncedUpdate);
        });
        reset();
      }

      init();
    })();
    </script>
