        resetBtn: document.getElementById('resetBtn')
      };

      const state = {
        scheduleVisible: false,
        valid: false,
        rawSchedule: [],
        formattedScheduleRows: []
      };

      const roundTo = (value, decimals = 6) => {
        if (!Number.isFinite(value)) return 0;
        const factor = Math.pow(10, decimals);
        return Math.round((value + Number.EPSILON) * factor) / factor;
      };

      const formatNumber = (value, decimals = 1) => {
        if (!Number.isFinite(value)) return '0';
        return value.toLocaleString('en-US', { minimumFractionDigits: decimals, maximumFractionDigits: decimals });
      };

      const formatRawInput = (raw, fallback) => {
        const trimmed = String(raw ?? '').trim();
        return trimmed.length ? trimmed : fallback;
      };

      const parseFloatSafe = (value) => {
        const parsed = parseFloat(value);
        return Number.isFinite(parsed) ? parsed : 0;
      };

      function parseInputs() {
        return {
          pulleyDiameter: parseFloatSafe(elements.pulleyDiameter.value),
          beltLength: parseFloatSafe(elements.beltLength.value),
          tension: parseFloatSafe(elements.tension.value),
          raw: {
            pulleyDiameter: elements.pulleyDiameter.value,
            beltLength: elements.beltLength.value,
            tension: elements.tension.value
          }
        };
      }

      function validate(inputs) {
        const errors = [];
        if (!Number.isFinite(inputs.pulleyDiameter) || inputs.pulleyDiameter <= 0) {
          errors.push('Pulley diameter must be greater than zero.');
        }
        if (!Number.isFinite(inputs.beltLength) || inputs.beltLength <= 0) {
          errors.push('Belt length must be greater than zero.');
        }
        if (!Number.isFinite(inputs.tension) || inputs.tension <= 0) {
          errors.push('Tension must be a positive value.');
        }
        return { ok: errors.length === 0, errors };
      }

      function compute(inputs) {
        const segmentLength = roundTo(inputs.beltLength / SEGMENT_COUNT, 6);
        const schedule = Array.from({ length: SEGMENT_COUNT }, (_, index) => {
          const notes = index === 0 ? 'Start node' : index === SEGMENT_COUNT - 1 ? 'End node' : 'Evenly distributed';
          return {
            segment: index + 1,
            length: segmentLength,
            tension: roundTo(inputs.tension, 6),
            notes
          };
        });
        return {
          tension: roundTo(inputs.tension, 6),
          beltLength: roundTo(inputs.beltLength, 6),
          pulleyDiameter: roundTo(inputs.pulleyDiameter, 6),
          segmentCount: schedule.length,
          schedule
        };
      }

      function format(outputs, inputs) {
        const tensionLabel = formatRawInput(inputs.raw.tension, formatNumber(outputs.tension, 1));
        return {
          tension: `${tensionLabel} N`,
          beltLength: `${formatNumber(outputs.beltLength, 1)} mm`,
          pulleyDiameter: `${formatNumber(outputs.pulleyDiameter, 1)} mm`,
          requestedTension: `${tensionLabel} N`,
          segmentCount: `${outputs.segmentCount} segments`,
          scheduleRows: outputs.schedule.map(row => ({
            segment: row.segment,
            length: formatNumber(row.length, 1),
            tension: formatNumber(row.tension, 1),
            notes: row.notes
          })),
          scheduleData: outputs.schedule
        };
      }

      function renderSchedule(rows) {
        elements.scheduleBody.innerHTML = '';
        if (!rows.length) {
          const placeholder = document.createElement('tr');
          placeholder.innerHTML = '<td colspan="4" class="text-center text-gray-500">Provide valid inputs to populate the tension segments.</td>';
          elements.scheduleBody.appendChild(placeholder);
          return;
        }
        const fragment = document.createDocumentFragment();
        rows.forEach(row => {
          const tr = document.createElement('tr');
          tr.innerHTML = `<td>${row.segment}</td><td>${row.length}</td><td>${row.tension}</td><td>${row.notes}</td>`;
          fragment.appendChild(tr);
        });
        elements.scheduleBody.appendChild(fragment);
      }

      function render(formatted, errors = []) {
        if (errors.length) {
          elements.errorBox.style.display = 'block';
          elements.errorBox.textContent = errors.join(' ');
          elements.totalMonthly.textContent = '—';
          elements.piMonthly.textContent = '—';
          elements.tiMonthly.textContent = '—';
          elements.totalInterest.textContent = '—';
          elements.payoffTime.textContent = '—';
          state.valid = false;
          state.rawSchedule = [];
          state.formattedScheduleRows = [];
          elements.downloadCsv.disabled = true;
          renderSchedule([]);
          return;
        }

        elements.errorBox.style.display = 'none';
        elements.errorBox.textContent = '';
        elements.totalMonthly.textContent = formatted.tension;
        elements.piMonthly.textContent = formatted.beltLength;
        elements.tiMonthly.textContent = formatted.pulleyDiameter;
        elements.totalInterest.textContent = formatted.requestedTension;
        elements.payoffTime.textContent = formatted.segmentCount;
        state.rawSchedule = formatted.scheduleData;
        state.formattedScheduleRows = formatted.scheduleRows;
        state.valid = true;
        elements.downloadCsv.disabled = false;
        if (state.scheduleVisible) {
          renderSchedule(state.formattedScheduleRows);
        }
      }

      function update() {
        const inputs = parseInputs();
        const validation = validate(inputs);
        if (!validation.ok) {
          render(null, validation.errors);
          return;
        }
        const outputs = compute(inputs);
        const formatted = format(outputs, inputs);
        render(formatted);
      }

      function reset() {
        elements.pulleyDiameter.value = defaults.pulleyDiameter;
        elements.beltLength.value = defaults.beltLength;
        elements.tension.value = defaults.tension;
        state.scheduleVisible = false;
        elements.scheduleWrap.style.display = 'none';
        elements.toggleSchedule.textContent = 'View tension breakdown';
        update();
      }

      function toggleScheduleHandler() {
        state.scheduleVisible = !state.scheduleVisible;
        const showing = state.scheduleVisible;
        elements.scheduleWrap.style.display = showing ? 'block' : 'none';
        elements.toggleSchedule.textContent = showing ? 'Hide tension breakdown' : 'View tension breakdown';
        if (showing && state.formattedScheduleRows.length) {
          renderSchedule(state.formattedScheduleRows);
        }
      }

      function formatCsvNumber(value, decimals = 1) {
        if (!Number.isFinite(value)) return '0';
        return value.toFixed(decimals);
      }

      function downloadCsvHandler() {
        if (!state.valid || !state.rawSchedule.length) return;
        const header = 'Segment,Length (mm),Tension (N),Notes';
        const rows = state.rawSchedule.map(row => [row.segment, formatCsvNumber(row.length), formatCsvNumber(row.tension), row.notes].join(','));
        const csv = [header].concat(rows).join('\n');
        const blob = new Blob([csv], { type: 'text/csv' });
        const url = URL.createObjectURL(blob);
        const link = document.createElement('a');
        link.href = url;
        link.download = 'belt-drive-schedule.csv';
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        URL.revokeObjectURL(url);
      }

      function debounce(fn, delay = 100) {
        let timer;
        return (...args) => {
          clearTimeout(timer);
          timer = setTimeout(() => fn(...args), delay);
        };
      }

      const debouncedUpdate = debounce(update, 100);

      document.querySelectorAll('#inputsCard input, #inputsCard select, #inputsCard textarea').forEach((el) => {
        el.addEventListener('input', debouncedUpdate);
        el.addEventListener('change', debouncedUpdate);
      });

      elements.calcBtn.addEventListener('click', update);
      elements.resetBtn.addEventListener('click', reset);
      elements.toggleSchedule.addEventListener('click', toggleScheduleHandler);
      elements.downloadCsv.addEventListener('click', downloadCsvHandler);

      reset();
    })();
    </script>
  </body>
</html>
