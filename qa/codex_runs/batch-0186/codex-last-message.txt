    <script>
        (function () {
            'use strict';

            const els = {
                totalInvestment: document.getElementById('totalInvestment'),
                numberOfRungs: document.getElementById('numberOfRungs'),
                investmentPerRung: document.getElementById('investmentPerRung'),
                displayTotal: document.getElementById('displayTotal'),
                displayRungs: document.getElementById('displayRungs'),
                errorBox: document.getElementById('errorBox'),
                calcBtn: document.getElementById('calcBtn'),
                resetBtn: document.getElementById('resetBtn')
            };

            const round2 = (value) => Math.round((Number(value) + Number.EPSILON) * 100) / 100;

            const formatCurrency = (value) => {
                const safe = Number.isFinite(value) ? value : 0;
                try {
                    return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(safe);
                } catch (err) {
                    return '$' + round2(safe).toFixed(2);
                }
            };

            function parseInputs() {
                const totalInvestment = parseFloat(els.totalInvestment.value);
                const numberOfRungs = parseInt(els.numberOfRungs.value, 10);
                return { totalInvestment, numberOfRungs };
            }

            function validate(inputs) {
                const errors = [];
                if (!Number.isFinite(inputs.totalInvestment) || inputs.totalInvestment <= 0) {
                    errors.push('Total investment must be greater than 0.');
                }
                if (!Number.isFinite(inputs.numberOfRungs) || inputs.numberOfRungs <= 0 || !Number.isInteger(inputs.numberOfRungs)) {
                    errors.push('Number of rungs must be a whole number greater than 0.');
                }
                return { ok: errors.length === 0, errors };
            }

            function compute(inputs) {
                if (inputs.numberOfRungs <= 0 || !Number.isFinite(inputs.totalInvestment)) {
                    return null;
                }
                const investmentPerRung = inputs.totalInvestment / inputs.numberOfRungs;
                if (!Number.isFinite(investmentPerRung)) {
                    return null;
                }
                return { investmentPerRung: round2(investmentPerRung) };
            }

            function format(outputs, inputs) {
                return {
                    investmentPerRung: outputs ? formatCurrency(outputs.investmentPerRung) : '—',
                    totalInvestment: formatCurrency(inputs.totalInvestment),
                    numberOfRungs: Number.isFinite(inputs.numberOfRungs) ? inputs.numberOfRungs : '—'
                };
            }

            function render(formatted, errors) {
                if (errors && errors.length) {
                    els.errorBox.style.display = 'block';
                    els.errorBox.textContent = errors.join(' ');
                    els.investmentPerRung.textContent = '—';
                } else {
                    els.errorBox.style.display = 'none';
                    els.errorBox.textContent = '';
                }
                els.displayTotal.textContent = formatted.totalInvestment;
                els.displayRungs.textContent = formatted.numberOfRungs;
                if (!errors || !errors.length) {
                    els.investmentPerRung.textContent = formatted.investmentPerRung;
                }
            }

            function update() {
                const inputs = parseInputs();
                const validation = validate(inputs);
                const preview = format(null, inputs);
                if (!validation.ok) {
                    render(preview, validation.errors);
                    return;
                }
                const outputs = compute(inputs);
                if (!outputs) {
                    render(preview, ['Unable to compute results with the current inputs.']);
                    return;
                }
                const formatted = format(outputs, inputs);
                render(formatted, []);
            }

            function debounce(fn, wait) {
                let timeout;
                return function () {
                    clearTimeout(timeout);
                    timeout = setTimeout(() => fn(), wait);
                };
            }

            const debouncedUpdate = debounce(update, 100);

            document.querySelectorAll('#inputsCard input').forEach((el) => {
                el.addEventListener('input', debouncedUpdate);
                el.addEventListener('change', debouncedUpdate);
